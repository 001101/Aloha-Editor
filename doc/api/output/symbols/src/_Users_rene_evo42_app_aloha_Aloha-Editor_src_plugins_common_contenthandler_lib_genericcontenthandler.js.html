  7 define(
  8 ['aloha', 'aloha/jquery', 'aloha/contenthandlermanager'],
  9 function(Aloha, jQuery, ContentHandlerManager) {
 10 	"use strict";
 11 
 12 	var
 13 		GENTICS = window.GENTICS;
 14 
 15 	/**
 16 	 * Register the generic content handler
 17 	 */
 18 	var GenericContentHandler = ContentHandlerManager.createHandler({
 19 		/**
 20 		 * Handle the pasting. Remove all unwanted stuff.
 21 		 * @param content
 22 		 */
 23 		handleContent: function( content ) {
 24 			if ( typeof content === 'string' ){
 25 				content = jQuery( '<div>' + content + '</div>' );
 26 			} else if ( content instanceof jQuery ) {
 27 				content = jQuery( '<div>' ).append(content);
 28 			}
 29 
 30 			// If we find an aloha-block inside the pasted content,
 31 			// we do not modify the pasted stuff, as it most probably
 32 			// comes from Aloha and not from other sources, and does
 33 			// not need to be cleaned up.
 34 			if (content.find('.aloha-block').length > 0) {
 35 				return;
 36 			}
 37 
 38 			// clean lists
 39 			this.cleanLists(content);
 40 
 41 			// transform tables
 42 			this.transformTables(content);
 43 
 44 			// remove comments
 45 			this.removeComments(content);
 46 
 47 			// unwrap font and span tags
 48 			this.unwrapTags(content);
 49 
 50 			// remove styles
 51 			this.removeStyles(content);
 52 
 53 			// remove namespaced elements
 54 			this.removeNamespacedElements(content);
 55 
 56 			// transform formattings
 57 			this.transformFormattings(content);
 58 
 59 			// transform links
 60 			//this.transformLinks(content);
 61 
 62 			return content.html();
 63 		},
 64 
 65 		/**
 66 		 * Clean lists: The only allowed children of ol or ul elements are li's. Everything else will be removed
 67 		 * @param content
 68 		 */
 69 		cleanLists: function(content) {
 70 			content.find('ul,ol').each(function() {
 71 				var $list = jQuery(this);
 72 				$list.contents(':not(li,ul,ol)').each(function() {
 73 					jQuery(this).remove();
 74 				});
 75 				// for all li's, trim the text contents
 76 				$list.children('li').each(function() {
 77 					var $li = jQuery(this);
 78 					$li.contents().each(function() {
 79 						if (this.nodeType === 3) {
 80 							this.data = jQuery.trim(this.data);
 81 						}
 82 					});
 83 				});
 84 			});
 85 		},
 86 
 87 		/**
 88 		 * Transform tables which were pasted
 89 		 * @param content
 90 		 */
 91 		transformTables: function(content) {
 92 			// remove border, cellspacing, cellpadding from all tables
 93 			// @todo what about width, height?
 94 			content.find('table').each(function() {
 95 				jQuery(this).removeAttr('border').removeAttr('cellspacing').removeAttr('cellpadding');
 96 			});
 97 			
 98 			// remove unwanted attributes and cleanup single empty p-tags
 99 			content.find('td').each(function() {
100 				// remove width, height and valign from all table cells
101 				jQuery(this).removeAttr('width').removeAttr('height').removeAttr('valign');
102 				
103 				if ( this.innerHTML.replace(/[\s\xA0]+/g,'') === '<p><br></p>' ) {
104 					this.innerHTML = ' ';
105 				}
106 				
107 				if ( jQuery(this).find('p').length == 1) {
108 					jQuery(this).find('p').contents().unwrap();
109 				}
110 			});
111 			
112 			// remove unwanted attributes from tr also? (tested with paste from open/libre office)
113 			// @todo or do this all via sanitize.js 
114 			content.find('tr').each(function() {
115 				// remove width, height and valign from all table cells
116 				jQuery(this).removeAttr('width').removeAttr('height').removeAttr('valign');
117 			});
118 			
119 			// completely colgroup tags
120 			// @TODO should we remove colgroup? use sanitize for that?
121 			content.find('colgroup').remove();
122 		},
123 
124 		/**
125 		 * Transform formattings
126 		 * @param content
127 		 */
128 		transformFormattings: function( content ) {
129 			// find all formattings we will transform
130 			content.find('strong,em,s,u').each(function() {
131 				if (this.nodeName.toLowerCase() == 'strong') {
132 					// transform strong to b
133 					Aloha.Markup.transformDomObject(jQuery(this), 'b');
134 				} else if (this.nodeName.toLowerCase() == 'em') {
135 					// transform em to i
136 					Aloha.Markup.transformDomObject(jQuery(this), 'i');
137 				} else if (this.nodeName.toLowerCase() == 's') {
138 					// transform s to del
139 					Aloha.Markup.transformDomObject(jQuery(this), 'del');
140 				} else if (this.nodeName.toLowerCase() == 'u') {
141 					// transform u?
142 					jQuery(this).contents().unwrap();
143 				}
144 			});
145 		},
146 
147 		/**
148 		 * Transform links
149 		 * @param content
150 		 */
151 		transformLinks: function( content ) {
152 			// find all links and remove the links without href (will be destination anchors from word table of contents)
153 			// aloha is not supporting anchors at the moment -- maybe rewrite anchors in headings to "invisible"
154 			// in the test document there are anchors for whole paragraphs --> the whole P appear as link
155 			content.find('a').each(function() {
156 				if ( typeof jQuery(this).attr('href') === 'undefined' ) {
157 					jQuery(this).contents().unwrap();
158 				}
159 			});
160 		},
161 
162 		/**
163 		 * Remove all comments
164 		 * @param content
165 		 */
166 		removeComments: function( content ) {
167 			var that = this;
168 
169 			// ok, remove all comments
170 			content.contents().each(function() {
171 				if (this.nodeType == 8) {
172 					jQuery(this).remove();
173 				} else {
174 					// do recursion
175 					that.removeComments(jQuery(this));
176 				}
177 			});
178 		},
179 
180 		/**
181 		 * Remove some unwanted tags from content pasted
182 		 * @param content
183 		 */
184 		unwrapTags: function( content ) {
185 			var that = this;
186 
187 			content.children('span,font,div').filter(function() {
188 				return this.contentEditable != 'false';
189 			}).each(function() {
190 				if (this.nodeName == 'DIV') {
191 					// safari and chrome cleanup for plain text paste with working linebreaks
192 					if (this.innerHTML == '<br>') {
193 						jQuery(this).contents().unwrap();
194 					} else {
195 						jQuery( Aloha.Markup.transformDomObject(jQuery(this), 'p').append('<br>') ).contents().unwrap();
196 					}
197 				} else {
198 					jQuery(this).contents().unwrap();
199 				}
200 
201 				that.unwrapTags(jQuery(this));
202 			});
203 		},
204 
205 		/**
206 		 * Remove styles
207 		 * @param content
208 		 */
209 		removeStyles: function( content ) {
210 			var that = this;
211 
212 			// completely remove style tags
213 			content.children('style').filter(function() {
214 				return this.contentEditable != 'false';
215 			}).remove();
216 
217 			// remove style attributes and classes
218 			content.children().filter(function() {
219 				return this.contentEditable != 'false';
220 			}).each(function() {
221 				jQuery(this).removeAttr('style').removeClass();
222 				that.removeStyles(jQuery(this));
223 			});
224 		},
225 
226 		/**
227 		 * Remove all elements which are in different namespaces
228 		 * @param content
229 		 */
230 		removeNamespacedElements: function( content ) {
231 			// get all elements
232 			content.find('*').each(function() {
233 				// try to determine the namespace prefix ('prefix' works for W3C
234 				// compliant browsers, 'scopeName' for IE)
235 
236 				var nsPrefix = this.prefix ? this.prefix
237 						: (this.scopeName ? this.scopeName : undefined);
238 				// when the prefix is set (and different from 'HTML'), we remove the
239 				// element
240 				if ((nsPrefix && nsPrefix != 'HTML') || this.nodeName.indexOf(':') >= 0 ) {
241 					var $this = jQuery(this), $contents = $this.contents();
242 					if ($contents.length) {
243 						// the element has contents, so unwrap the contents
244 						$contents.unwrap();
245 					} else {
246 						// the element is empty, so remove it
247 						$this.remove();
248 					}
249 				}
250 			});
251 		}
252 	});
253 
254 	return GenericContentHandler;
255 });
