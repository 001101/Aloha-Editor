 11 	/**
 12 	 * Helper function to create an array of "selected" td's, which are marked
 13 	 * with the class "aloha-cell-selected"
 14 	 *
 15 	 * @param {Table} table
 16 	 * @return {Array} set of selected td's
 17 	 */
 18 	function getSelectedCells ( table ) {
 19 		var cells = [],
 20 		    $cells = table.obj.find( '.aloha-cell-selected' );
 21 		$cells.each( function () { cells.push( this ); } );
 22 		return cells;
 23 	};
