  4 	/**
  5 	 * Constructs a TableCell.
  6 	 *
  7 	 * @param {DomNode} cell
  8 	 *        A td/th which will be represente by this TableCell.
  9 	 * @param {Table} tableObj
 10 	 *        The Table which contains the cell. The cell will be
 11 	 *        activated/dactivated with the table.
 12 	 */
 13 	var TableCell = function(originalTd, tableObj) {
 14         if (null == originalTd) {
 15             originalTd = '<td> </td>';
 16         }
 17 
 18         //original Td must be a DOM node so that the this.obj.context property is available
 19         //this transformation will properly handle jQuery objects as well as DOM nodes
 20         originalTd = jQuery( originalTd ).get( 0 );
 21 
 22         this.obj = jQuery(originalTd);
 23         this.tableObj = tableObj;
 24 
 25         tableObj.cells.push(this);
 26 	};

 50 	TableCell.prototype.activate = function () {
 51 		// wrap the created div into the contents of the cell
 52 		this.obj.wrapInner( '<div/>' );
 53 
 54 		// create the editable wrapper for the cells
 55 		var wrapper = this.obj.children( 'div' ).eq( 0 );
 56 
 57 		wrapper.contentEditable( true );
 58 		wrapper.addClass( 'aloha-table-cell-editable' );
 59 
 60 		var that = this;
 61 		
 62 		// attach events to the editable div-object
 63 		wrapper.bind( 'focus', function ( jqEvent ) {
 64 			// ugly workaround for ext-js-adapter problem in ext-jquery-adapter-debug.js:1020
 65 			if ( jqEvent.currentTarget ) {
 66 				jqEvent.currentTarget.indexOf = function () {
 67 					return -1;
 68 				};
 69 			}
 70 			that._editableFocus( jqEvent );
 71 		} );
 72 		
 73 		wrapper.bind( 'mousedown', function ( jqEvent ) {
 74 			// ugly workaround for ext-js-adapter problem in ext-jquery-adapter-debug.js:1020
 75 			if ( jqEvent.currentTarget ) {
 76 				jqEvent.currentTarget.indexOf = function () {
 77 					return -1;
 78 				};
 79 			}
 80 			
 81 			that._editableMouseDown( jqEvent );
 82 
 83 			that._startCellSelection();
 84 		} );
 85 		wrapper.bind( 'blur',      function ( jqEvent ) { that._editableBlur( jqEvent );    });
 86 		wrapper.bind( 'keyup',     function ( jqEvent ) { that._editableKeyUp( jqEvent );   });
 87 		wrapper.bind( 'keydown',   function ( jqEvent ) { that._editableKeyDown( jqEvent ); });
 88 		wrapper.bind( 'mouseover', function ( jqEvent ) { that._selectCellRange();          });
 89 
 90 		// we will treat the wrapper just like an editable
 91 		wrapper.contentEditableSelectionChange( function ( event ) {
 92 			Aloha.Selection.onChange( wrapper, event );
 93 			return wrapper;
 94 		} );
 95 
 96 		this.obj.bind( 'mousedown', function ( jqEvent ) {
 97 			setTimeout( function () {
 98 				that.wrapper.trigger( 'focus' );
 99 			}, 1 );
100 			that.tableObj.selection.unselectCells();
101 	        that._startCellSelection();       
102 			jqEvent.stopPropagation();
103 		} );
104 
105 		if ( this.obj.get( 0 ) ) {
106 			this.obj.get( 0 ).onselectstart = function ( jqEvent ) { return false; };
107 		}
108 
109 		// set contenteditable wrapper div
110 		this.wrapper = this.obj.children();
111 		if ( this.wrapper.get( 0 ) ) {
112 			this.wrapper.get( 0 ).onselectstart = function () {
113 				window.event.cancelBubble = true;
114 			};
115 			// Disabled the dragging of content, since it makes cell selection difficult
116 			this.wrapper.get( 0 ).ondragstart = function () { return false };
117 		}
118 
119 		return this;
120 	};

122 	/**
123 	 * The deactivate method removes the contenteditable helper div within the
124 	 * table-data field and wraps the innerHtml to the outerHTML
125 	 *
126 	 * @return void
127 	 */
128 	TableCell.prototype.deactivate = function() {
129 		var wrapper = this.obj.children('.aloha-table-cell-editable');
130 
131 		if (wrapper.length) {
132 			// get the inner html of the contenteditable div
133 			var innerHtml = wrapper.html();
134 
135 			// remove the contenteditable div and its attached events
136 			wrapper.remove();
137 
138 			// remove the click event of the
139 			this.obj.unbind('click');
140 
141 			if (jQuery.trim(this.obj.attr('class')) == '') {
142 				this.obj.removeAttr('class');
143 			}
144 
145 			// set the inner html of the contenteditable div as html for the table-data
146 			// field
147 			this.obj.html(innerHtml);
148 		}
149 	}

160 	/**
161 	 * Focus method for the contentediable div within a table data-field. The method
162 	 * requires the event-property Cell as a Cell object. If the
163 	 * Cell wasn't activated yet it does all relevant actions to activate the cell.
164 	 *
165 	 * @param e
166 	 *            the jquery event object
167 	 * @return void
168 	 */
169 	TableCell.prototype._editableFocus = function(e) {
170 		// only do activation stuff if the cell don't has the focus
171 		if (!this.hasFocus) {
172 			// set an internal flag to focus the table
173 			this.tableObj.focus();
174 
175 			// add an active-class
176 			this.obj.addClass('aloha-table-cell_active');
177 
178 			// set the focus flag
179 			this.hasFocus = true;
180 
181 			// select the whole content in the table-data field
182 			this._selectAll(this.wrapper.get(0));
183 
184 			// unset the selection type
185 			this.tableObj.selection.selectionType = 'cell';
186 
187 		}
188 	};

226 	/**
227 	 * Starts the cell selection mode
228 	 */
229 	TableCell.prototype._startCellSelection = function(){
230 		if(!this.tableObj.selection.cellSelectionMode){
231 
232 			//unselect currently selected cells
233 			this.tableObj.selection.unselectCells();
234 
235 			// activate cell selection mode
236 			this.tableObj.selection.cellSelectionMode = true; 
237 
238 			//bind a global mouseup event handler to stop cell selection
239 			var that = this;
240 			jQuery('body').bind('mouseup.cellselection', function(){
241 				that._endCellSelection();
242 			});
243 
244 			this.tableObj.selection.baseCellPosition = [this._virtualY(), this._virtualX()];
245 		}
246 	};

248 	/**
249 	 * Ends the cell selection mode
250 	 */
251 	TableCell.prototype._endCellSelection = function(){
252 		if(this.tableObj.selection.cellSelectionMode){
253 			this.tableObj.selection.cellSelectionMode = false; 
254 			this.tableObj.selection.baseCellPosition = null;
255 			this.tableObj.selection.lastSelectionRange = null; 
256 
257 			this.tableObj.selection.selectionType = 'cell';
258 
259 			//unbind the global cell selection event
260 			jQuery('body').unbind('mouseup.cellselection');
261 		}
262 	};

264 	TableCell.prototype._getSelectedRect = function () {
265 		var right = this._virtualX();
266 		var bottom = this._virtualY();
267 		var topLeft = this.tableObj.selection.baseCellPosition;
268 		var left = topLeft[1];
269 		if (left > right) {
270 			left = right;
271 			right = topLeft[1];
272 		}
273 		var top = topLeft[0];
274 		if (top > bottom) {
275 			top = bottom;
276 			bottom = topLeft[0];
277 		}
278 		return {"top": top, "right": right, "bottom": bottom, "left": left};
279 	};

281 	/**
282 	 * Toggles selection of cell.
283 	 * This works only when cell selection mode is active. 
284 	 */
285 	TableCell.prototype._selectCellRange = function(){
286 		if(!this.tableObj.selection.cellSelectionMode) {
287 			return;
288 		}
289 
290 		var rect = this._getSelectedRect();
291 
292 		var table = this.tableObj;
293 		var $rows = table.obj.children().children('tr');
294 		var grid = Utils.makeGrid($rows);
295 		
296 		table.selection.selectedCells = [];
297 		var selectClass = table.get('classCellSelected');
298 		Utils.walkGrid(grid, function (cellInfo, j, i) {
299 			if ( Utils.containsDomCell(cellInfo) ) {
300 				if (i >= rect.top && i <= rect.bottom && j >= rect.left && j <= rect.right) {
301 					jQuery( cellInfo.cell ).addClass(selectClass);
302 					table.selection.selectedCells.push(cellInfo.cell);
303 				} else {
304 					jQuery( cellInfo.cell ).removeClass(selectClass);
305 				}
306 			}
307 		});
308 
309 		table.selection.notifyCellsSelected();
310 	};

312 	/**
313 	 * Selects all inner-contens of an contentEditable-object
314 	 *
315 	 * @param editableNode dom-representation of the editable node (div-element)
316 317 	 * @return void
318 	 */
319 	TableCell.prototype._selectAll = function(editableNode) {
320 		var e = (editableNode.jquery) ? editableNode.get(0) : editableNode;
321 
322 		// Not IE
323 		if (!jQuery.browser.msie) {
324 			var s = window.getSelection();
325 			// WebKit
326 			if ( s.setBaseAndExtent /*&& e> 0 */ ) {
327 				s.setBaseAndExtent( e, 0, e, Math.max( 0, e.innerText.length - 1 ) );
328 			}
329 			// Firefox and Opera
330 			else {
331 				// workaround for bug # 42885
332 				if (window.opera
333 					&& e.innerHTML.substring(e.innerHTML.length - 4) == '<BR>') {
334 					e.innerHTML = e.innerHTML + ' ';
335 				}
336 
337 				var r = document.createRange();
338 				r.selectNodeContents(e);
339 				s.removeAllRanges();
340 				s.addRange(r);
341 			}
342 		}
343 		// Some older browsers
344 		else if (document.getSelection) {
345 			var s = document.getSelection();
346 			var r = document.createRange();
347 			r.selectNodeContents(e);
348 			s.removeAllRanges();
349 			s.addRange(r);
350 		}
351 		// IE
352 		else if (document.selection) {
353 			var r = document.body.createTextRange();
354 			r.moveToElementText(e);
355 			r.select();
356 		}
357 	};

389 	/**
390 	 * The key-down event for the ediable-div in the td-field. Check if the the div
391 	 * is empty and insert an  . Furthermore if cells are selected, unselect
392 	 * them.
393 	 *
394 	 * @param jqEvent
395 	 *            the jquery-event object
396 	 * @return void
397 	 */
398 	TableCell.prototype._editableKeyDown = function(jqEvent) {
399 		var KEYCODE_TAB = 9;
400 
401 		this._checkForEmptyEvent(jqEvent);
402 		
403 		if ( this.obj[0] === this.tableObj.obj.find('tr:last td:last')[0] ) {
404 			// only add a row on a single key-press of tab (so check
405 			// that alt-, shift- or ctrl-key are NOT pressed)
406 			if (KEYCODE_TAB == jqEvent.keyCode && !jqEvent.altKey && !jqEvent.shiftKey && !jqEvent.ctrlKey) {
407 				// add a row after the current row
408 				this.tableObj.addRow(this.obj.parent().index() + 1);
409 
410 				// firefox needs this for the first cell of the new row
411 				// to be selected (.focus() doesn't work reliably in
412 				// IE7)
413 				this.tableObj.cells[this.tableObj.cells.length - 1]._selectAll(this.wrapper.get(0));
414 
415 				jqEvent.stopPropagation();
416 				return;
417 			}
418 		}
419 	};

421 	/**
422 	 * The custom keyup event for a table-cell Checks if the cell is empty and
423 	 * inserts a space (\u00a0)
424 	 *
425 	 * @param e
426 	 *            the event object which is given by jquery
427 	 * @return void
428 	 */
429 	TableCell.prototype._checkForEmptyEvent = function(jqEvent) {
430 		var $wrapper = jQuery(this.wrapper),
431 		    text = $wrapper.text();
432 
433 		if ( $wrapper.children().length > 0) {
434 			return;
435 		}
436 
437 		// if empty insert a blank space and blur and focus the wrapper
438 		if ( text === '' ){
439 			this.wrapper.text('\u00a0');
440 			this.wrapper.get(0).blur();
441 			this.wrapper.get(0).focus();
442 		}
443 	};
