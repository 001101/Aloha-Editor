188 	/**
189 	 * Given an unbalanced table structure, pad it with the necessary cells to
190 	 * make it perfectly rectangular
191 	 *
192 	 * @param {Aloha.Table} tableObj
193 	 */
194 	function correctTableStructure ( tableObj ) {
195 		var table = tableObj.obj,
196 			
197 			i,
198 		    row,
199 		    rows = tableObj.getRows(),
200 		    rowsNum = rows.length,
201 			
202 			cols,
203 			colsNum,
204 			
205 		    colsCount,
206 		    maxColsCount = 0,
207 		    cachedColsCounts = [],
208 		    colsCountDiff,
209 		    colSpan;
210 		
211 		for ( i = 0; i < rowsNum; i++ ) {
212 			row = jQuery( rows[ i ] );
213 			cols = row.find( '>td' );
214 			colsNum = cols.length;
215 			colsCount = Utils.cellIndexToGridColumn( rows, i, colsNum - 1 ) + 1;
216 			
217 			// Check if the last cell in this row has a col span, to account
218 			// for it in the total number of colums in this row
219 			
220 			colSpan = parseInt( cols.last().attr( 'colspan' ), 10 );
221 			
222 			if ( colSpan == 0 ) {
223 				// TODO: support colspan=0
224 				// http://dev.w3.org/html5/markup/td.html#td.attrs.colspan
225 				// http://www.w3.org/TR/html401/struct/tables.html#adef-colspan
226 				// The value zero ("0") means that the cell spans all columns
227 				// from the current column to the last column of the column
228 				// group (COLGROUP) in which the cel
229 			} else if ( !isNaN( colSpan ) ) {
230 				// The default value of this attribute is one ("1"), so where this
231 				// is the case, we will remove such superfluous colspan attributes
232 				if ( colSpan == 1 ) {
233 					cols.last().removeAttr( 'colspan' );
234 				}
235 				
236 				colsCount += ( colSpan - 1 );
237 			}
238 			
239 			cachedColsCounts.push( colsCount );
240 			
241 			if ( colsCount > maxColsCount ) {
242 				maxColsCount = colsCount;
243 			}
244 		}
245 		
246 		for ( i = 0; i < rowsNum; i++ ) {
247 			colsCountDiff = maxColsCount - cachedColsCounts[ i ];
248 			if ( colsCountDiff > 0 ) {
249 				// Create as many td's as we need to complete the row
250 				jQuery( rows[ i ] ).append(
251 					( new Array( colsCountDiff + 1 ) ).join( '<td></td>' )
252 				);
253 			}
254 		}
255 	};

484 	/**
485 	 * Mouse-Down event for the selection-cells on the left side of the table
486 	 *
487 	 * @param jqEvent
488 	 *            the jquery-event object
489 	 * @return void
490 	 */
491 	Table.prototype.rowSelectionMouseDown = function ( jqEvent ) {
492 		// focus the table (if not already done)
493 		this.focus();
494 
495 		// if no cells are selected, reset the selection-array
496 		if ( this.selection.selectedCells.length == 0 ) {
497 			this.rowsToSelect = [];
498 		}
499 
500 		// set the origin-rowId of the mouse-click
501 		this.clickedRowId = jqEvent.currentTarget.parentNode.rowIndex;
502 
503 		// set single column selection
504 		if ( jqEvent.metaKey ) {
505 			var arrayIndex = jQuery.inArray( this.clickedRowId, this.rowsToSelect );
506 			if ( arrayIndex >= 0 ) {
507 				this.rowsToSelect.splice( arrayIndex, 1 );
508 			} else {
509 				this.rowsToSelect.push( this.clickedRowId );
510 			}
511 		// block of columns selection
512 		} else if ( jqEvent.shiftKey ) {
513 			this.rowsToSelect.sort( function( a, b ) { return a - b; } );
514 			var start = this.rowsToSelect[ 0 ];
515 			var end = this.clickedRowId;
516 			if ( start > end ) {
517 				start = end;
518 				end = this.rowsToSelect[ 0 ];
519 			}
520 			this.rowsToSelect = [];
521 			for ( var i = start; i <= end; i++ ) {
522 				this.rowsToSelect.push( i );
523 			}
524 		// single column
525 		} else {
526 			this.rowsToSelect = [ this.clickedRowId ];
527 		}
528 
529 		// mark the selection visual
530 		this.selectRows();
531 
532 		// prevent browser from selecting the table
533 		jqEvent.preventDefault();
534 
535 		// stop bubble, otherwise the mousedown of the table is called ...
536 		jqEvent.stopPropagation();
537 
538 		// prevent ff/chrome/safare from selecting the contents of the table
539 		return false;
540 	};

542 	/**
543 	 * The mouse-over event for the selection-cells on the left side of the table.
544 	 * On mouse-over check which column was clicked, calculate the span between
545 	 * clicked and mouse-overed cell and mark them as selected
546 	 *
547 	 * @param jqEvent
548 	 *            the jquery-event object
549 	 * @return void
550 	 */
551 	Table.prototype.rowSelectionMouseOver = function (jqEvent) {
552 		var rowIndex = jqEvent.currentTarget.parentNode.rowIndex,
553 			indexInArray, start, end, i;
554 
555 		// only select the row if the mouse was clicked and the clickedRowId isn't
556 		// from the selection-row (row-id = 0)
557 		if (this.mousedown && this.clickedRowId >= 0) {
558 
559 			// select first cell
560 	//		var firstCell = this.obj.find('tr:nth-child(2) td:nth-child(2)').children('div[contenteditable=true]').get(0);
561 	//		jQuery(firstCell).get(0).focus();
562 
563 			indexInArray = jQuery.inArray(rowIndex, this.rowsToSelect);
564 
565 			start = (rowIndex < this.clickedRowId) ? rowIndex : this.clickedRowId;
566 			end = (rowIndex < this.clickedRowId) ? this.clickedRowId : rowIndex;
567 
568 			this.rowsToSelect = new Array();
569 			for ( i = start; i <= end; i++) {
570 				this.rowsToSelect.push(i);
571 			}
572 
573 			// this actually selects the rows
574 			this.selectRows();
575 
576 			// prevent browser from selecting the table
577 			jqEvent.preventDefault();
578 
579 			// stop bubble, otherwise the mousedown of the table is called ...
580 			jqEvent.stopPropagation();
581 
582 			// prevent ff/chrome/safare from selecting the contents of the table
583 			return false;
584 		}
585 	};

1149 	 * Inserts new columns into the table. Either on the right or left side. If
1150 	 * columns are selected, the amount of selected columns will be inserted on the
1151 	 * 'right' or 'left' side. If no cells are selected, 1 new column will be
1152 	 * inserted before/after the column of the last active cell.
1153 	 * As well all column-selection events must be bound to the firsts row-cell.
1154 	 *
1155 	 * @param position
1156 	 *            could be 'left' or 'right'. defines the position where the new
1157 	 *            columns should be inserted
1158 	 * @return void
1159 	 */
1160 	Table.prototype.addColumns = function( position ) {
1161 		var 
1162 			that = this,
1163 			emptyCell = jQuery( '<td>' ),
1164 		    rows = this.getRows(),
1165 			cell,
1166 			currentColIdx,
1167 			columnsToSelect = [],
1168 			selectedColumnIdxs = this.selection.selectedColumnIdxs;
1169 		
1170 		if ( 0 === selectedColumnIdxs.length ) {
1171 			return;
1172 		}
1173 		
1174 		selectedColumnIdxs.sort( function ( a, b ) { return a - b; } );
1175 		
1176 		// refuse to insert a column unless a consecutive range has been selected
1177 		if ( ! Utils.isConsecutive( selectedColumnIdxs ) ) {
1178 			Aloha.showMessage( new Aloha.Message( {
1179 				title : i18n.t( 'Table' ),
1180 				text  : i18n.t( 'table.addColumns.nonConsecutive' ),
1181 				type  : Aloha.Message.Type.ALERT
1182 			} ) );
1183 			return;
1184 		}
1185 		
1186 		if ( 'left' === position ) {
1187 			currentColIdx = selectedColumnIdxs[ 0 ];
1188 			// inserting a row before the selected column indicies moves
1189 			// all selected columns one to the right
1190 			for ( var i = 0; i < this.selection.selectedColumnIdxs.length; i++ ) {
1191 				this.selection.selectedColumnIdxs[ i ] += 1;
1192 			}
1193 		} else {//"right" == position
1194 			currentColIdx = selectedColumnIdxs[ selectedColumnIdxs.length - 1 ];
1195 		}
1196 		
1197 		var grid = Utils.makeGrid( rows );
1198 		
1199 		for ( var i = 0; i < rows.length; i++ ) {
1200 			// prepare the cell to be inserted
1201 			cell = emptyCell.clone();
1202 			cell.html( '\u00a0' );
1203 
1204 			// on first row correct the position of the selected columns
1205 			if ( i == 0 ) {
1206 				// this is the first row, so make a column-selection cell
1207 				this.attachColumnSelectEventsToCell( cell );
1208 			} else {
1209 				// activate the cell for this table
1210 				cellObj = this.newActiveCell( cell.get(0) );
1211 				cell = cellObj.obj;
1212 			}
1213 
1214 			var leftCell = Utils.leftDomCell( grid, i, currentColIdx );
1215 			if ( null == leftCell ) {
1216 				jQuery( rows[i] ).prepend( cell );
1217 			} else {
1218 				if ( 'left' === position && Utils.containsDomCell( grid[ i ][ currentColIdx ] ) ) {
1219 					jQuery( leftCell ).before( cell );
1220 				} else {//right
1221 					jQuery( leftCell ).after( cell );
1222 				}
1223 			}
1224 
1225 			this.numCols++;
1226 		}
1227 	};
