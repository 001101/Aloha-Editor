 66 	//					+ stylePath + '"></script>');

127 		prepareFileUpload: function(file) {
128 			var 
129 				reader = new FileReader(),
130 				fileObj,
131 				that = this;
132 			reader.file = file;
133             reader.onloadend = function() {
134                 var currentFile = {
135                     name: this.file.name,
136                     type: this.file.type,
137                     fileSize: this.file.fileSize,
138                     fileName: this.file.fileName,
139                     data: reader.result
140                 };
141                 that.filesObjs.push(that.uploader.addFileUpload(currentFile));
142                 that.processedFiles++;
143                 Aloha.trigger('aloha-file-upload-prepared',fileObj);
144             };
145             reader.readAsDataURL(file);

147 		/**
148 		 * Our drop event Handler
149 		 */
150 		dropEventHandler: function(event) {
151 			var 
152 				that = this, edConfig, len, target,
153 				files = event.dataTransfer.files, dropimg;
154 			this.targetEditable = undefined;
155 			this.droppedFilesCount = files.length;
156 			this.processedFiles = 0;
157 			Aloha.Log.info(that, this.droppedFilesCount + " files have been dropped on the page");
158 			
159 //			if (jQuery.browser.msie) {
160 //				var textdata = event.dataTransfer.getData('Text');
161 //				var urldata = event.dataTransfer.getData('URL');
162 //				var imagedataW = window.event.dataTransfer.getData('URL');
163 //				var textdataW = window.event.dataTransfer.getData('Text');
164 //				var x = textdataW;
165 //			}
166 			// if no files where dropped, use default handler
167 			if (!event.dataTransfer && !event.dataTransfer.files) {
168 				event.sink = false;
169 				return true;
170 			}
171 			if (this.droppedFilesCount < 1) {
172 				event.sink = false;
173 				return true;
174 			}
175 			if (event.preventDefault) {
176 				event.preventDefault();
177 			} else {
178 				event.cancelBubble = true;
179 			}
180 			if (this.droppedFilesCount > that.settings.max_file_count) {
181 				Aloha.Log.warn(that,"too much files dropped");
182 				if (event.stopPropagation) {
183 					event.stopPropagation();
184 				} else {
185 					event.returnValue = false;
186 				}
187 				return true;
188 			}
189 			target = jQuery(event.target);
190 			//If drop in editable
191 			if (target.hasClass('aloha-editable')) {
192 				this.targetEditable = target;
193 				target = this.targetEditable.children(':last');
194 				if (target.hasClass('aloha-editable')) {
195 					//nested space is needed in this tag, otherwise select won't success...
196 					this.targetEditable.append('<span> </span>');
197 					target = this.targetEditable.children(':last');
198 				}
199 			} else {
200 				this.targetEditable = target.parents('.aloha-editable');
201 			}
202 			this.filesObjs = [];
203 			this.dropInEditable = false;
204 			len = this.droppedFilesCount;
205 			if (this.targetEditable[0] === null) { // Process files out of editables
206 				while(--len >= 0) {
207 					if  ( // Set of conditions, can we resize the image, and do we have a conf to do it
208 							!(!!document.createElement('canvas').getContext &&
209 							  files[len].type.match(/image\//) &&
210 							  edConfig.upload.config.image)
211 						) {
212 						if (files[len].size <= that.settings.max_file_size) {
213 							that.prepareFileUpload(files[len]);
214 						} else {
215 							this.processedFiles++;
216 							Aloha.Log.warn(that,"max_file_size exeeded, upload of " + files[len].name + " aborted");
217 						}
218 					} else {
219 						that.prepareFileUpload(files[len]);
220 					}
221 				}
222 			} else {
223 				Aloha.getEditableById(this.targetEditable.attr('id')).activate();
224 				that.targetRange = that.initializeRangeForDropEvent(event, this.targetEditable);
225 				edConfig = that.getEditableConfig(this.targetEditable);
226 				if (edConfig) {
227 					that.dropInEditable = true;
228 				}
229 				while(--len >= 0) {
230 					try {
231 						dropimg = edConfig.upload.config.image;
232 					} catch (e) {
233 						dropimg = false;
234 					}
235 					if  ( // Set of conditions, can we resize the image, and do we have a conf to do it
236 							!(!!document.createElement('canvas').getContext &&
237 							  files[len].type.match(/image\//) &&
238 							  dropimg)
239 						) {
240 						if (files[len].size <= edConfig.max_file_size) {
241 							that.prepareFileUpload(files[len]);
242 						} else {
243 							this.processedFiles++;
244 							Aloha.Log.warn(that,"max_file_size exeeded, upload of " + files[len].name + " aborted");
245 						}
246 					} else {
247 						that.prepareFileUpload(files[len]);
248 					}
249 				} //while
250 			}
251 			
252 			if (event.stopPropagation) {
253 				event.stopPropagation();
254 			} else {
255 				event.returnValue = false;
256 			}
257 			return false;
258 		},
259 		/**
260 		 *  Attach drag and drop listeners to document body (Native JS way)
261 		 *
262 		 */
263 		setBodyDropHandler: function() {
264 			var that = this;
265 			if (!document.body.BodyDragSinker){
266 				document.body.BodyDragSinker = true;
267 				this.onstr = "";
268 				this.mydoc = document;
269 				this.methodName = "addEventListener";
270 				if (jQuery.browser.msie) {
271 					this.onstr = "on";
272 					this.methodName = "attachEvent";
273 					this.mydoc = document.body;
274 				}
275 
276 				// sets the default handler
277 				this.mydoc[this.methodName](this.onstr+"drop", function(event) {that.dropEventHandler(event)} , false);
278 			// TODO: improve below to allow default comportment behaviour if drop event is not a files drop event
279 			this.mydoc[this.methodName](this.onstr+"dragenter", function(event) {
280 				if (event.preventDefault)
281 					event.preventDefault();
282 				else
283 					event.cancelBubble = true;
284 				if (event.stopPropagation)
285 					event.stopPropagation();
286 				else
287 					event.returnValue = false;
288 				return false;
289 			}, false);
290 			this.mydoc[this.methodName](this.onstr+"dragleave", function(event) {
291 				if (event.preventDefault)
292 					event.preventDefault();
293 				else
294 					event.cancelBubble = true;
295 				if (event.stopPropagation)
296 					event.stopPropagation();
297 				else
298 					event.returnValue = false;
299 				return false;
300 			}, false);
301 			this.mydoc[this.methodName](this.onstr+"dragover", function(event) {
302 				if (event.preventDefault)
303 					event.preventDefault();
304 				else
305 					event.cancelBubble = true;
306 				if (event.stopPropagation)
307 					event.stopPropagation();
308 				else
309 					event.returnValue = false;
310 				//return false;
311 			}, false);
312 
313 
314 
315 			} // if
316 			// end body events
317 			//==================
318 		},

326 			var target = jQuery(event.target);
327 //			if (target.textNodes().length == 0 && target.html().length == 0) {
328 //				target.html(" ");

332 			if (target.textNodes().length == 0) {
333 				range.startContainer = target[0].childNodes[0];
334 				range.endContainer = target[0].childNodes[0];
335 			} else {
336 				range.startContainer = target.textNodes()[0];
337 				range.endContainer = target.textNodes()[0];
338 			}
339 		//
340 				range.startOffset = 0;
341 				range.endOffset = 0;
342 			try {
343 				range.select();
344 			} catch (error) {
345 				Aloha.Log.error(this,error);
346 			}
347 			return range;
348 		}
