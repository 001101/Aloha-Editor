 10 	/**
 11 	 * Do an markup test
 12 	 * @param editable the editable
 13 	 * @param startContainer
 14 	 * @param startOffset
 15 	 * @param endContainer
 16 	 * @param endOffset
 17 	 * @param markup markup to be applied
 18 	 * @param original reference
 19 	 * @param reference result selector
 20 	 * @param nesting true if nesting of markup shall be allowed
 21 	 */
 22 	function doMarkupTest(editable, startContainer, startOffset, endContainer, endOffset, markup, original, reference, nesting) {
 23 		// generate the range
 24 		var range = TestUtils.generateRange(startContainer, startOffset, endContainer, endOffset);
 25 	
 26 		// apply the markup
 27 		TestUtils.applyMarkup(editable, range, markup, nesting);
 28 	
 29 		// get the result
 30 		var result = Aloha.editables[0].getContents(true);
 31 	
 32 		// get the expected results
 33 		var expected = Aloha.jQuery(reference).contents();
 34 	
 35 		// compare the result with the expected result
 36 		deepEqual(result.extractHTML(), expected.extractHTML(), 'Check Operation Result');
 37 	
 38 	/*
 39 		// now apply the markup a second time
 40 		TestUtils.applyMarkup(editable, range, markup, nesting);
 41 	
 42 		// get the expected results
 43 		expected = Aloha.jQuery(original).contents();
 44 	
 45 		// compare the result with the expected result
 46 		deepEqual(result.extractHTML(), expected.extractHTML(), 'Check Double Operation Result');
 47 	*/
 48 	};
