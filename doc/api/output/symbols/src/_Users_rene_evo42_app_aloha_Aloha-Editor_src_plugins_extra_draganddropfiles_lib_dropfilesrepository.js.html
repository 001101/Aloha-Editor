201 		UploadFile: Aloha.RepositoryDocument.extend({
202 			_constructor: function(properties) {
203 				var xhr = this.xhr,
204 				that = this;
205 				this._super(properties);
206 				xhr.upload['onprogress'] = function(rpe) {
207 					that.loaded = rpe.loaded;
208 					that.total = rpe.total;
209 					that.ulProgress = rpe.loaded / rpe.total;
210 					Aloha.trigger('aloha-upload-progress',that);
211 					xhr.onload = function(load) {
212 						try {
213 							that.src = that.upload_config.callback(xhr.responseText);
214 							Aloha.trigger('aloha-upload-success',that);
215 						} catch(e) {
216 							Aloha.trigger('aloha-upload-failure', that);
217 						}
218 //						if (that.delegateUploadEvent(xhr.responseText)) {
219 	//
220 //						} else {
221 					};
222 					xhr.onabort = function() {
223 						Aloha.trigger('aloha-upload-abort', that);
224 					};
225 					xhr.onerror = function(e) {
226 						Aloha.trigger('aloha-upload-error', that);

232 			/**
233 			 * Process upload of a file
234 			 */
235 			startUpload: function() {
236 				//if ()
237 				var xhr = this.xhr, options = this.upload_config, that = this, data;
238 
239 				xhr.open(options.method, typeof(options.url) == "function" ? options.url(number) : options.url, true);
240 				xhr.setRequestHeader("Cache-Control", "no-cache");
241 				xhr.setRequestHeader("X-Requested-With", "XMLHttpRequest");
242 				xhr.setRequestHeader(options.file_name_header, this.file.fileName);
243 				xhr.setRequestHeader("X-File-Size", this.file.fileSize);
244 				xhr.setRequestHeader("Accept", options.accept);
245 //			l
246 				if (!options.send_multipart_form) {
247 					xhr.setRequestHeader("Content-Type", this.file.type + ";base64");
248 					xhr.overrideMimeType(this.file.type);
249 					var canvas = $('<canvas>').first(),
250 						targetsize = {},
251 						tempimg = new Image();
252 					Aloha.Log.debug(Aloha,"Original Data (length:" + this.file.data.length + ") = " + this.file.data.substring(0,30));
253 					tempimg.onload = function() {
254 						targetsize = {
255 							height: tempimg.height,
256 							width: tempimg.width
257 						};
258 
259 
260 						if (tempimg.width > tempimg.height) {
261 							if (tempimg.width > options.image.max_width) {
262 								targetsize.width = options.image.max_width;
263 								targetsize.height = tempimg.height * options.image.max_width / tempimg.width;
264 							}
265 						} else {
266 							if (tempimg.height > options.image.max_height) {
267 								targetsize.height = options.image.max_height;
268 								targetsize.width = tempimg.width * options.image.max_height / tempimg.height;
269 							}
270 
271 						}
272 
273 						var canvas = document.createElement('canvas');
274 						canvas.setAttribute('width', targetsize.width);
275 						canvas.setAttribute('height', targetsize.height);
276 						canvas.getContext('2d').drawImage(
277 							tempimg,
278 							0,
279 							0,
280 							tempimg.width,
281 							tempimg.height,
282 							0,
283 							0,
284 							targetsize.width,
285 							targetsize.height
286 						);
287 						data = canvas.toDataURL(that.file.type);
288 						Aloha.Log.debug(Aloha,"Sent Data (length:" + data.length + ") = " + data.substring(0,30));
289 						xhr.send(data);
290 					};
291 					tempimg.src = this.file.data;
292 				} else {
293 					if (window.FormData) {//Many thanks to scottt.tw
294 						var f = new FormData();
295 						f.append(typeof(options.fieldName) == "function" ? options.fieldName() : options.fieldName, this.file);
296 						xhr.send(f);
297 					}
298 					else if (this.file.getAsBinary) {//Thanks to jm.schelcher
299 						var boundary = (1000000000000+Math.floor(Math.random()*8999999999998)).toString();
300 						var dashdash = '--';
301 302 						var crlf     = '\r\n';
303 
304 						/* Build RFC2388 string. */
305 						var builder = '';
306 
307 						builder += dashdash;
308 						builder += boundary;
309 						builder += crlf;
310 
311 						builder += 'Content-Disposition: form-data; name="'+(typeof(options.fieldName) == "function" ? options.fieldName() : options.fieldName)+'"';
312 						builder += '; filename="' + this.file.fileName + '"';
313 						builder += crlf;
314 
315 						builder += 'Content-Type: application/octet-stream';
316 						builder += crlf;
317 						builder += crlf;
318 
319 						/* Append binary data. */
320 						builder += this.file.getAsBinary();
321 						builder += crlf;
322 
323 						/* Write boundary. */
324 						builder += dashdash;
325 						builder += boundary;
326 						builder += dashdash;
327 						builder += crlf;
328 
329 						xhr.setRequestHeader('content-type', 'multipart/form-data; boundary=' + boundary);
330 						xhr.sendAsBinary(builder);
331 					}
332 					else {
333 						options.onBrowserIncompatible();
334 					}
335 				}
336 
337 
338 			}
