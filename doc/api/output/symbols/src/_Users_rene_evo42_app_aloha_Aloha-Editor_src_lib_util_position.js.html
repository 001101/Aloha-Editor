 20 /**
 21  * position utility, which will provide scroll and mouse positions
 22  * please note that the positions provided by this class are not
 23  * realtime - instead they are calculated with a 0.5 second delay
 24  */
 25 GENTICS.Utils.Position = {};
 26 
 27 /**
 28  * jquery reference to the window object
 29  */
 30 GENTICS.Utils.Position.w = jQuery(window);
 31 
 32 /**
 33  * contains the current scroll top and left position, and indicates if the user is currently scrolling
 34  * @api
 35  */
 36 GENTICS.Utils.Position.Scroll = {
 37 		top : 0,
 38 		left : 0,
 39 		isScrolling : false
 40 };
 41 
 42 /**
 43  * contains the scroll corrections to apply on special cases (ribbon for example)
 44  * @api
 45  */
 46 GENTICS.Utils.Position.ScrollCorrection = {
 47 		top : 0,
 48 		left : 0
 49 };
 50 
 51 /**
 52  * contains the current mouse position (x,y) as well as an indicator if the mouse is moving
 53  * @api
 54  */
 55 GENTICS.Utils.Position.Mouse = {
 56 		x : 0,
 57 		y : 0,
 58 		oldX : 0,
 59 		oldY : 0,
 60 		isMoving : false,
 61 		triggeredMouseStop : true
 62 };
 63 
 64 /**
 65  * contains all mousestop callbacks
 66  */
 67 GENTICS.Utils.Position.mouseStopCallbacks = [];
 68 
 69 /**
 70  * contains all mousemove callbacks
 71  */
 72 GENTICS.Utils.Position.mouseMoveCallbacks = [];
 73 
 74 /**
 75  * updates scroll position and the scrolling status
 76  */
 77 GENTICS.Utils.Position.update = function () {
 78 	// update scroll position
 79 	var
 80 		st = this.w.scrollTop(),
 81 		sl = this.w.scrollLeft(),
 82 		i;
 83 
 84 	if (this.Scroll.isScrolling) {
 85 		if (this.Scroll.top == st && this.Scroll.left == sl) {
 86 			// stopped scrolling
 87 			this.Scroll.isScrolling = false;
 88 		}
 89 	} else {
 90 		if (this.Scroll.top != st || this.Scroll.left != sl) {
 91 			// started scrolling
 92 			this.Scroll.isScrolling = true;
 93 		}
 94 	}
 95 
 96 	// update scroll positions
 97 	this.Scroll.top = st;
 98 	this.Scroll.left = sl;
 99 
100 	// check wether the user has stopped moving the mouse
101 	if (this.Mouse.x == this.Mouse.oldX && this.Mouse.y == this.Mouse.oldY) {
102 		this.Mouse.isMoving = false;
103 		// now check if we've triggered the mousestop event
104 		if (!this.Mouse.triggeredMouseStop) {
105 			this.Mouse.triggeredMouseStop = true;
106 			// iterate callbacks
107 			for (i=0; i<this.mouseStopCallbacks.length; i++) {
108 				this.mouseStopCallbacks[i].call();
109 			}
110 		}
111 	} else {
112 		this.Mouse.isMoving = true;
113 		this.Mouse.triggeredMouseStop = false;
114 		// iterate callbacks
115 		for (i=0; i<this.mouseMoveCallbacks.length; i++) {
116 			this.mouseMoveCallbacks[i].call();
117 		}
118 	}
119 
120 	// update mouse positions
121 	this.Mouse.oldX = this.Mouse.x;
122 	this.Mouse.oldY = this.Mouse.y;
123 };
124 
125 /**
126  * adds a callback method which is invoked when the mouse has stopped moving
127  * @param	callback	the callback method to be invoked
128  * @return	index of the callback
129  */
130 GENTICS.Utils.Position.addMouseStopCallback = function (callback) {
131 	this.mouseStopCallbacks.push(callback);
132 	return (this.mouseStopCallbacks.length - 1);
133 };
134 
135 /**
136  * adds a callback method which is invoked when the mouse is moving
137  * @param	callback	the callback method to be invoked
138  * @return	index of the callback
139  */
140 GENTICS.Utils.Position.addMouseMoveCallback = function (callback) {
141 	this.mouseMoveCallbacks.push(callback);
142 	return (this.mouseMoveCallbacks.length - 1);
143 };

147 jQuery(function () {
148 	setInterval(function (){
149 		GENTICS.Utils.Position.update();
