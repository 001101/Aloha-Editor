  9  * Provides a set of language codes and images

 11 
 12 define(
 13 [ 'aloha', 'aloha/jquery' ],
 14 function( Aloha, jQuery ) {
 15 	'use strict';
 16 
 17 	return new ( Aloha.AbstractRepository.extend( {
 18 
 19 		/**
 20 		 * Set of language codes
 21 		 */
 22 		languageCodes: [],
 23 
 24 		_constructor: function() {
 25 			this._super( 'wai-languages' );
 26 		},
 27 
 28 		/**
 29 		 * Initialize WAI Languages, load the language file and prepare the data.
 30 		 */
 31 		init: function() {
 32 			// Load the language codes
 33 			jQuery.ajax( {
 34 				url      : Aloha.getPluginUrl( 'wai-lang' ) + '/lib/language-codes.json',
 35 				dataType : 'json',
 36 				success  : jQuery.proxy( this.storeLanguageCodes, this ),
 37 				error    : this.errorHandler
 38 			} );
 39 
 40 		    this.repositoryName = 'WaiLanguages';
 41  42 		},
 43 
 44 		markObject: function( obj, item ) {
 45 			//copied from wai-lang-plugin makeVisible to avoid a circular dependency
 46 			// We do not need to add this class here since it already being
 47 			// done in the wai-lang plugin
 48 			// jQuery( obj ).addClass( 'aloha-wai-lang' );
 49 		},
 50 
 51 		/**
 52 		 * This method will invoked if a error occurres while loading data via ajax
 53 		 */
 54 		errorHandler: function( text, error ) {
 55 			//TODO log error here
 56 		},
 57 
 58 		/**
 59 		 * Stores the retrieved language code data in this object
 60 		 */
 61 		storeLanguageCodes: function( data ) {
 62 			var that = this,
 63 			    flagsIconsPath = Aloha.getPluginUrl( 'flag-icons' ),
 64 			    el;
 65 
 66 			// Transform loaded json into a set of repository documents
 67 			jQuery.each( data, function( key, value ) {
 68 				el = value;
 69 				el.id = key;
 70 				el.repositoryId = that.repositoryId;
 71 				el.type = 'language';
 72 				el.url =  flagsIconsPath + '/img/flags/' + el.id + '.png';
 73 				// el.renditions.url = "img/flags/" + e.id + ".png";
 74 				// el.renditions.kind.thumbnail = true;
 75 				that.languageCodes.push( new Aloha.RepositoryDocument( el ) );
 76 			} );
 77 		},
 78 
 79 		/**
 80 		 * Searches a repository for object items matching query if objectTypeFilter.
 81 		 * If none found it returns null.
 82 		 * Not supported: filter, orderBy, maxItems, skipcount, renditionFilter
 83 		 */
 84 		query: function( p, callback ) {
 85 			var query = new RegExp( '^' + p.queryString, 'i' ),
 86 			    i,
 87 			    d = [],
 88 			    matchesName,
 89 			    matchesType,
 90 			    currentElement;
 91 
 92 			for ( i = 0; i < this.languageCodes.length; ++i ) {
 93 				currentElement = this.languageCodes[ i ];
 94 				matchesName = ( !p.queryString || currentElement.name.match( query )  || currentElement.nativeName.match( query ) );
 95 				matchesType = ( !p.objectTypeFilter || ( !p.objectTypeFilter.length ) || jQuery.inArray( currentElement.type, p.objectTypeFilter ) > -1 );
 96 
 97 				if ( matchesName && matchesType ) {
 98 					d.push( currentElement );
 99 				}
100 			}
101 102 
			callback.call( this, d );
103 		}
104 
105 	} ) )();
106 } );
