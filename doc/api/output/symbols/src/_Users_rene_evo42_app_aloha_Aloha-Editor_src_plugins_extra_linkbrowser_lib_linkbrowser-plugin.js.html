 24 	
 25 	'use strict';
 26 	
 27 	var LinkBrowser = Browser.extend( {
 28 		
 29 		init: function ( config ) {
 30 			this._super( config );
 31 			
 32 			var that = this;
 33 			var repositoryButton = new Aloha.ui.Button( {
 34 				iconClass : 'aloha-button-big aloha-button-tree',
 35 				size      : 'large',
 36 				onclick   : function () { that.show(); },
 37 				tooltip   : i18n.t( 'button.addlink.tooltip' ),
 38 				toggle    : false
 39 			} );
 40 			
 41 			FloatingMenu.addButton(
 42 				Links.name,
 43 				repositoryButton,
 44 				i18n.t( 'floatingmenu.tab.link' ),
 45 				1
 46 			);
 47 			
 48 			this.url = Aloha.getAlohaUrl() + '/../plugins/extra/linkbrowser/';
 49 		},
 50 		
 51 		onSelect: function ( item ) {
 52 			Links.hrefField.setItem( item );
 53 			
 54 			// Now create a selection within the editable since the user should be able to type once the link has been created.
 55 			// 1. We need to save the current cursor position since the a activate editable event will be fired and this will 
 56 			// set the cursor in the upper left cornor of the editable.
 57 			var	range = Aloha.Selection.getRangeObject();
 58 			var currentStartContainer = range.startContainer = range.endContainer; 
 59 			var currentStartOffset = range.startOffset = range.endOffset;
 60 			
 61 			// 2. Do the first select - this will invoke the activate editable event
 62 			range.select();
 63 
 64 			// 3. Restore the range
 65 			range.startContainer = range.endContainer = currentStartContainer; 
 66 			range.startOffset = range.endOffset = currentStartOffset;
 67 			// 4. Invoke the final selection
 68 			range.select();
 69 						
 70 			// Close the browser lightbox
 71 			this.close();
 72 		},
 73 		
 74 		renderRowCols: function ( item ) {
 75 			var row = {},
 76 			    pluginUrl = this.url,
 77 			    icon = '__page__',
 78 			    idMatch = item.id.match( /(\d+)\./ );
 79 			
 80 			jQuery.each( this.columns, function ( colName, v ) {
 81 				switch ( colName ) {
 82 				case 'icon':
 83 					row.icon = '<div class="aloha-browser-icon\
 84 						aloha-browser-icon-' + icon + '"></div>';
 85 					break;
 86 				case 'translations':
 87 					if ( !item.renditions ) {
 88 						break;
 89 					}
 90 					
 91 					var rends = item.renditions,
 92 					    i = rends.length,
 93 					    strBldr = [],
 94 					    r;
 95 					
 96 					for ( ; i > 0; --i ) {
 97 						r = rends[ i ];
 98 						if ( r.kind == 'translation' ) {
 99 							strBldr.push(
100 								//'<a href="' + repo_host + r.url + '">' +
101 								'<img src="'   + pluginUrl  + 'img/flags/' + r.language + '.png"\
102 									  alt="'   + r.language + '"\
103 									  title="' + r.filename + '" />'
104 								//'</a>'
105 							);
106 						}
107 					}
108 					
109 					row.translations = strBldr.join( '' );
110 					break;
111 				case 'language':
112 					row.language = item.language;
113 					break;
114 				default:
115 					row[ colName ] = item[ colName ] || '--';
116 				}
117 			} );
118 			
119 			return row;
120 		}
121 	
122 	} );

146 			};
