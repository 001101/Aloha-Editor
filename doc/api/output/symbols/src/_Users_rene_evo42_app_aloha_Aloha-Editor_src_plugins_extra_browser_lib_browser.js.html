 60 /**
 61  * Simple templating
 62  *
 63  * @param {String} str - The string containing placeholder keys in curly
 64  *                       brackets
 65  * @param {Object} obj - Associative array of replacing placeholder keys
 66  *                       with corresponding values
 67  */
 68 function supplant (str, obj) {
 69 	 return str.replace(/\{([a-z0-9\-\_]+)\}/ig, function (str, p1, offset, s) {
 70 		 var replacement = obj[p1] || str;
 71 		 return (typeof replacement === 'function') ? replacement() : replacement;
 72 	 });
 73 };
 74 
 75 /**
 76  * Wrapper to call the supplant method on a given string, taking the
 77  * nsClasses object as the associative array containing the replacement
 78  * pairs
 79  *
 80  * @param {String} str
 81  * @return {String}
 82  */
 83 function renderTemplate (str) {
 84 	return (typeof str === 'string') ? supplant(str, nsClasses) : str;
 85 };

264 	/**
265 	 * TODO: Is there a way we can guard against potential infinite loops?
266 	 *
267 	 * @param {Associative Array Object} fn - Name of any Browser method
268 	 */
269 	trigger: function (fn, returned) {
270 		var cb = this._callbacks;
271 		var func = cb[fn];
272 		
273 		if (typeof func === 'object') {
274 			for (var i = 0, l = func.length; i < l; ++i) {
275 				if (typeof func[i] === 'function') {
276 					func[i].call(this, returned);
277 				}
278 			}
279 		}

282 	/**
283 	 * Handles the binding of callbacks to any method in the browser.
284 	 * Removes the necessity for functions to manually trigger callbacks
285 	 * events within themselves by wrapping them within an function that
286 	 * does it on their behalf when necessary.
287 	 * A user can simply do the following:
288 	 *		browser.callback('anyFunction', function () { alert('called back'); });
289 	 * This will work regardless of whether browser.anyFunction manually
290 	 * triggers any events.
291 	 *
292 	 * this.enableCallbacks actually does most of the heavy lifting.
293 	 *
294 	 * @param {String} fn - Name of any browser method
295 	 * @param {Function} cb - Callback function to invoke
296 	 */
297 	callback: function (fn, cb) {
298 		if (typeof this[fn] != 'function') {
299 			Console.warn(
300 				'Unable to add a callback to "' + fn +
301 				'" because it is not a method in Aloha.Browser.'
302 			);
303 			
304 			return this;
305 		}
306 		
307 		if (typeof cb !== 'function') {
308 			Console.warn(
309 				'Unable to add a callback to "' + fn + '" because '	+
310 				'the callback object that was given is of type "'	+
311 				(typeof cb) + '". '									+
312 				'The callback object needs to be of type "function".'
313 			);
314 			
315 			return this;
316 		}
317 		
318 		if (this._callbacks[fn] == undefined) {
319 			if (this.enableCallbacks(fn)) {
320 				this._callbacks[fn] = [cb];
321 			}
322 		} else {
323 			this._callbacks[fn].push(cb);
324 		}
325 		
326 		return this;
327 	},

329 	/**
330 	 * Work-horse for this.callback method
331 	 */
332 	enableCallbacks: function (fn) {
333 		var browser_inst = this;
334 		var func = this[fn];
335 		
336 		if (typeof func === 'function') {
337 			this[fn] = function () {
338 				var returned = func.apply(browser_inst, arguments);
339 				browser_inst.trigger.call(browser_inst, fn, returned);
340 				return returned;
341 			};
342 			
343 			return true;
344 		} else {
345 			Console.warn(
346 				'Cannot enable callbacks for function "' + fn +
347 				'" because no such method was found in Aloha.Browser.'
348 			);
349 			
350 			return false;
351 		}
352 	},

397 	 * Convert a repository object into an object that can be used with our
398 	 * tree component. Also add a reference to this object in our objs hash.
399 	 * According to the Repository specification, each object will at least
400 	 * have the following properties at least: id, name, url, and type. Any
401 	 * and all other attributes are optional.

403 	harvestRepoObject: function (obj) {

418 		var icon = '', attr;
419 		
420 		switch (obj.baseType) {
421 		case 'folder':
422 			icon = 'folder';
423 			break;
424 		case 'document':
425 			icon = 'document';
426 			break;
427 		}
428 
429 		// if the object has a type set, we set it as type to the node
430 		if (obj.type) {
431 			attr = {rel: obj.type};
432 		}
433 
434 		return {
435 			data: {
436 				title : obj.name, 
437 				attr  : {'data-rep-oobj': obj.uid}, 
438 				icon  : icon
439 			},
440 			attr : attr,
441 			state: (obj.hasMoreItems || obj.baseType === 'folder') ? 'closed' : null,
442 			resource: obj
443 		};
444 	},

816 	/**
817 	 * This function adds new sort fields into the _orderBy array.
818 	 * If a field already exists, it will be spliced from where it is and
819 	 * unshifted to the end of the array
820 	 */
821 	setSortOrder: function (by, order) {
822 		var sortItem = {};
823 		sortItem[by] = order || 'asc';
824 		
825 		var orderBy = this._orderBy || [];
826 		var field;
827 		var orderItem;
828 		var found = false;
829 		
830 		for (var i = 0, j = orderBy.length; i < j; ++i) {
831 			orderItem = orderBy[i];
832 			
833 			for (field in orderItem) {
834 				if (field === by) {
835 					orderBy.splice(i, 1);
836 					orderBy.unshift(sortItem);
837 					found = true;
838 					break;
839 				}
840 			}
841 			
842 			if (found) {
843 				break;
844 			}
845 		}
846 		
847 		found || orderBy.unshift(sortItem);
848 		
849 		this._orderBy = orderBy;
850 		
851 		return this;
852 	},

858 	doPaging: function (dir) {
859 		switch (dir) {
860 			case 'first':
861 				this._pagingOffset = 0;
862 				break;
863 			case 'end':
864 				if ((this._pagingCount % this.pageSize) === 0) {
865 					// item count is exactly divisible by page size
866 					this._pagingOffset = this._pagingCount - this.pageSize;
867 				} else {
868 					this._pagingOffset = this._pagingCount - (this._pagingCount % this.pageSize);
869 				}
870 				break;
871 			case 'next':
872 				this._pagingOffset += this.pageSize;
873 				break;
874 			case 'prev':
875 				this._pagingOffset -= this.pageSize;
876 				break;
877 		}
878 		
879 		// TODO: animate
880 881 		//var grid = this.grid.find('.ui-jqgrid-bdiv');
882 		//grid.animate({marginLeft: -grid.width()}, 500);
883 		
884 		this.fetchItems(this._currentFolder, this.processItems);
885 	},

948 	processItems: function (data, metainfo) {
949 		var btns = this._pagingBtns;
950 		var disabled = 'ui-state-disabled';
951 
952 		// if the total number of items is known, we can calculate the number of pages
953 		if (metainfo && jQuery.isNumeric(metainfo.numItems)) {
954 			this._pagingCount = metainfo.numItems;
955 		} else {
956 			this._pagingCount = undefined;
957 		}
958 
959 		this.grid.find('.loading').hide();
960 		this.list.show();
961 		this.listItems(data);
962 		
963 		if (this._pagingOffset <= 0) {
964 			btns.first.add(btns.prev).addClass(disabled);
965 		} else {
966 			btns.first.add(btns.prev).removeClass(disabled);
967 		}
968 		
969 		if (isNaN(this._pagingCount)) {
970 			btns.end.addClass(disabled);
971 			
972 			if (data.length < this.pageSize) {
973 				btns.next.addClass(disabled);
974 			} else {
975 				btns.next.removeClass(disabled);
976 			}
977 		} else if (this._pagingOffset + this.pageSize >= this._pagingCount) {
978 			btns.end.add(btns.next).addClass(disabled);
979 		} else {
980 			btns.end.add(btns.next).removeClass(disabled);
981 		}
982 		
983 		var from, to;
984 		
985 		if (data.length == 0 && this._pagingOffset == 0) {
986 			from = 0;
987 			to = 0;
988 		} else {
989 			from = this._pagingOffset + 1;
990 			to = from + data.length - 1;
991 		}
992 		
993 		this.grid.find('.ui-paging-info').html(
994 			i18n.t('Viewing') + ' ' +	 (from)
995 					          + ' - '  + (to)
996 					          + ' ' + i18n.t('of') + ' ' + (jQuery.isNumeric(this._pagingCount) ? this._pagingCount : i18n.t('numerous'))
997 		);
998 	},
