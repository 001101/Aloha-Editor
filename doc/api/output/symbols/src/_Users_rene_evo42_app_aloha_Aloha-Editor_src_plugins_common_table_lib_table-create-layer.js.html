1 define(
  2 ['aloha/jquery'],
  3 function (jQuery) {
  4 	/**
  5   6 	 * Initialize of the CreateLayer object
  7 	 */
  8 	CreateLayer = function(TablePlugin){
  9 		this.TablePlugin = TablePlugin;
 10 	};
 11 
 12 	/**
 13 	 * Internal configuration of the create-table panel
 14 	 */
 15 	CreateLayer.prototype.parameters = {
 16 		elemId: 'aloha-table-createLayer', // id of the create-table panel
 17 		className: 'aloha-table-createdialog',   // class-name of the create-table panel
 18 		numX: 10,	         // Number of cols in the create-layer
 19 		numY: 10,            // Number of rows in the create-layer vertically
 20 		layer: undefined,    // Attribute holding the create-layer
 21 		target: undefined    // the clicktarget which was clicked on (mostly the button of the floatingmenu)
 22 	};
 23 
 24 	/**
 25 	 * The configuration-object for the implementer of the plugin. All keys of
 26 	 * the "parameters" object could be overwritten within this object and will
 27 	 * simply be used instead.
 28 	 */
 29 	CreateLayer.prototype.config = new Object();
 30 
 31 	/**
 32 	 * Flag wether the CreateLayer is currently visble or not
 33 	 */
 34 	CreateLayer.prototype.visible = false;
 35 
 36 	/**
 37 	 * This function checks if there is an create-table-layer. If no layer exists, it creates one and puts it into the configuration.
 38 	 * If the layer was already created it sets the position of the panel and shows it.
 39 	 *
 40 	 * @return void
 41 	 */
 42 	CreateLayer.prototype.show = function(){
 43 		var layer = this.get('layer');
 44 
 45 		// create the panel if the layer doesn't exist
 46 		if (layer == null) {
 47 			this.create();
 48 		}else {
 49 			// or reposition, cleanup and show the layer
 50 			this.setPosition(layer);
 51 			layer.find('td').removeClass('hover');
 52 			layer.show();
 53 		}
 54 		this.visible = true;
 55 	};
 56 
 57 	/**
 58 	 * Creates the div-layer which holds a table with the given number of rows and cols.
 59 	 * It sets click and mouseover-events to the table data fields
 60 	 *
 61 	 * @return void
 62 	 */
 63 	CreateLayer.prototype.create = function () {
 64 		var that = this;
 65 		var layer = jQuery('<div></div>');
 66 		layer.id = this.get('elemId');
 67 		layer.addClass(this.get('className'));
 68 
 69 		var table = jQuery('<table></table>');
 70 		table.css('width', (this.get('numX') + 6) * 15);
 71 		var tr;
 72 		var td;
 73 
 74 		for (var i = 0; i < this.get('numY'); i++) {
 75 			tr = jQuery('<tr></tr>');
 76 
 77 			for (var j = 0; j < this.get('numX'); j++) {
 78 				td = jQuery('<td>\u00a0</td>');
 79 
 80 				if (i == 0 && j == 0) {
 81 					td.addClass('hover');
 82 				}
 83 
 84 				td.bind('mouseover', {rowId: i, colId: j}, function(e) {
 85 					that.handleMouseOver(e, table);
 86 				});
 87 
 88 				td.bind('click', {rowId: i, colId: j}, function(e){
 89 					var rows = e.data.rowId + 1;
 90 					var cols = e.data.colId + 1;
 91 
 92 					that.TablePlugin.createTable(cols, rows);
 93 					that.hide();
 94 				});
 95 
 96 				tr.append(td);
 97 			}
 98 			table.append(tr);
 99 		}
100 		layer.append(table);
101 
102 		// set attributes
103 		this.set('layer', layer);
104 		this.setPosition();
105 
106 		// stop bubbling the click on the create-dialog up to the body event
107 		layer.bind('click', function(e) {
108 			e.stopPropagation();
109 		}).mousedown(function(e) {
110 			e.stopPropagation();
111 		});
112 
113 		// append layer to body and
114 		// hide the create layer if user clicks anywhere in the body
115 		jQuery('body').append(layer).bind('click', function(e) {
116 			if (e.target != that.get('target') && that.visible) {
117 				that.hide();
118 			}
119 		});
120 	};
121 
122 	/**
123 	 * handles the mose over state for a cell
124 	 * @param e event object
125 	 * @param table the aeffected table
126 	 * @return void
127 	 */
128 	CreateLayer.prototype.handleMouseOver = function(e, table) {
129 		var rowId = e.data.rowId;
130 		var colId = e.data.colId;
131 		var innerRows = table.find('tr');
132 
133 		for (var n = 0; n <= innerRows.length; n++) {
134 			var innerCells = jQuery(innerRows[n]).find('td');
135 
136 			for (var k = 0; k <= innerCells.length; k++) {
137 				if (n <= rowId && k <= colId) {
138 					jQuery(innerCells[k]).addClass('hover');
139 				} else {
140 					jQuery(innerCells[k]).removeClass('hover');
141 				}
142 			}
143 		}
144 	};
145 
146 	/**
147 	 * Sets the "left" and "top" style-attributes according to the clicked target-button
148 	 *
149 	 *  @return void
150 	 */
151 	CreateLayer.prototype.setPosition = function() {
152 		var targetObj = jQuery(this.get('target'));
153 		var pos = targetObj.offset();
154 		this.get('layer').css('left', pos.left + 'px');
155 		this.get('layer').css('top', (pos.top + targetObj.height()) + 'px');
156 	};
157 
158 	/**
159 	 * Hides the create-table panel width the jQuery-method hide()
160 	 *
161 	 * @see jQuery().hide()
162 	 * @return void
163 	 */
164 	CreateLayer.prototype.hide = function() {
165 		this.get('layer').hide();
166 		this.visible = false;
167 	};
168 
169 	/**
170 	 * The "get"-method returns the value of the given key. First it searches in the
171 	 * config for the property. If there is no property with the given name in the
172 	 * "config"-object it returns the entry associated with in the parameters-object
173 	 *
174 	 * @param property
175 	 * @return void
176 	 */
177 	CreateLayer.prototype.get = function(property) {
178 		// return param from the config
179 		if (this.config[property]) {
180 			return this.config[property];
181 		}
182 		// if config-param was not found return param from the parameters-object
183 		if (this.parameters[property]) {
184 			return this.parameters[property];
185 		}
186 		return undefined;
187 	};
188 
189 	/**
190 	 * The "set"-method takes a key and a value. It checks if there is a key-value
191 	 * pair in the config-object. If so it saves the data in the config-object. If
192 	 * not it saves the data in the parameters-object.
193 	 *
194 	 * @param key
195 	 *            the key which should be set
196 	 * @param value
197 	 *            the value which should be set for the associated key
198 	 */
199 	CreateLayer.prototype.set = function (key, value) {
200 		// if the key already exists in the config-object, set it to the config-object
201 		if (this.config[key]) {
202 			this.config[key] = value;
203 
204 			// otherwise "add" it to the parameters-object
205 		}else{
206 			this.parameters[key] = value;
207 		}
208 	};
209 
210 	return CreateLayer;
211 });
