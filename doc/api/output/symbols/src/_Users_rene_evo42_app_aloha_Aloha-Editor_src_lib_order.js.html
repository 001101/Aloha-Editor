 10  11 (function () {
 12     //Sadly necessary browser inference due to differences in the way
 13     //that browsers load and execute dynamically inserted javascript
 14     //and whether the script/cache method works.
 15     //Currently, Gecko and Opera do not load/fire onload for scripts with
 16     //type="script/cache" but they execute injected scripts in order
 17     //unless the 'async' flag is present.
 18     //However, this is all changing in latest browsers implementing HTML5
 19     //spec. Firefox nightly supports using the .async true by default, and
 20     //if false, then it will execute in order. Favor that test first for forward
 21     //compatibility. However, it is unclear if webkit/IE will follow suit.
 22     //Latest webkit breaks the script/cache trick.
 23     //Test for document and window so that this file can be loaded in
 24     //a web worker/non-browser env. It will not make sense to use this
 25     //plugin in a non-browser env, but the file should not error out if included
 26     //in a file, then loaded in a non-browser env.
 27     var supportsInOrderExecution = typeof document !== "undefined" &&
 28                                    typeof window !== "undefined" &&
 29                                    (document.createElement("script").async ||
 30                                (window.opera && Object.prototype.toString.call(window.opera) === "[object Opera]") ||
 31                                //If Firefox 2 does not have to be supported, then
 32                                //a better check may be:
 33                                //('mozIsLocallyAvailable' in window.navigator)
 34                                ("MozAppearance" in document.documentElement.style)),
 35         readyRegExp = /^(complete|loaded)$/,
 36         waiting = [],
 37         cached = {};
 38 
 39     function loadResource(name, req, onLoad) {
 40         req([name], function (value) {
 41             //The value may be a real defined module. Wrap
 42             //it in a function call, because this function is used
 43             //as the factory function for this ordered dependency.
 44             onLoad(function () {
 45                 return value;
 46             });
 47         });
 48     }
 49 
 50     //Callback used by the type="script/cache" callback that indicates a script
 51     //has finished downloading.
 52     function scriptCacheCallback(evt) {
 53         var node = evt.currentTarget || evt.srcElement, i,
 54             moduleName, resource;
 55 
 56         if (evt.type === "load" || readyRegExp.test(node.readyState)) {
 57             //Pull out the name of the module and the context.
 58             moduleName = node.getAttribute("data-requiremodule");
 59 
 60             //Mark this cache request as loaded
 61             cached[moduleName] = true;
 62 
 63             //Find out how many ordered modules have loaded
 64             for (i = 0; (resource = waiting[i]); i++) {
 65                 if (cached[resource.name]) {
 66                     loadResource(resource.name, resource.req, resource.onLoad);
 67                 } else {
 68                     //Something in the ordered list is not loaded,
 69                     //so wait.
 70                     break;
 71                 }
 72             }
 73 
 74             //If just loaded some items, remove them from waiting.
 75             if (i > 0) {
 76                 waiting.splice(0, i);
 77             }
 78 
 79             //Remove this script tag from the DOM
 80             //Use a setTimeout for cleanup because some older IE versions vomit
 81             //if removing a script node while it is being evaluated.
 82             setTimeout(function () {
 83                 node.parentNode.removeChild(node);
 84             }, 15);
 85         }
 86     }
 87 
 88     define({
 89         version: '0.26.0',
 90 
 91         load: function (name, req, onLoad, config) {
 92             var url = req.nameToUrl(name, null);
 93 
 94             //If a build, just load the module as usual.
 95             if (config.isBuild) {
 96                 loadResource(name, req, onLoad);
 97                 return;
 98             }
 99 
100             //Make sure the async attribute is not set for any pathway involving
101             //this script.
102             require.s.skipAsync[url] = true;
103             if (supportsInOrderExecution) {
104                 //Just a normal script tag append, but without async attribute
105                 //on the script.
106                 req([name], function (value) {
107                     //The value may be a real defined module. Wrap
108                     //it in a function call, because this function is used
109                     //as the factory function for this ordered dependency.
110                     onLoad(function () {
111                         return value;
112                     });
113                 });
114             } else {
115                 //Credit to LABjs author Kyle Simpson for finding that scripts
116                 //with type="script/cache" allow scripts to be downloaded into
117                 //browser cache but not executed. Use that
118                 //so that subsequent addition of a real type="text/javascript"
119                 //tag will cause the scripts to be executed immediately in the
120                 //correct order.
121                 if (req.specified(name)) {
122                     req([name], function (value) {
123                         //The value may be a real defined module. Wrap
124                         //it in a function call, because this function is used
125                         //as the factory function for this ordered dependency.
126                         onLoad(function () {
127                             return value;
128                         });
129                     });
130                 } else {
131                     waiting.push({
132                         name: name,
133                         req: req,
134                         onLoad: onLoad
135                     });
136                     require.attach(url, null, name, scriptCacheCallback, "script/cache");
137                 }
138             }
139         }
140     });
141 }());
