  7 define(
  8 [
  9 'aloha', 
 10 'aloha/jquery', 
 11 'aloha/plugin', 
 12 'aloha/floatingmenu', 
 13 'i18n!zemanta/nls/i18n', 
 14 'i18n!aloha/nls/i18n', 
 15 'aloha/console',
 16 'css!zemanta/css/zemanta-widget-alohaeditor.css', 
 17 ],
 18 function(Aloha, jQuery, Plugin, FloatingMenu, i18n, i18nCore, console) {
 19 	"use strict";
 20 
 21 	var
 22 		GENTICS = window.GENTICS,
 23 		active = false,
 24 		editableId = false,
 25 		settings = false;
 26 
 27 		// check for API key settings
 28 		if ( typeof Aloha.settings.plugins.zemanta === 'undefined' ) {
 29 			settings = {};
 30 		} else {
 31 			settings = Aloha.settings.plugins.zemanta;
 32 		}
 33 
 34 		if ( typeof settings.apiKey === 'undefined' ) {
 35 			Aloha.Log.warn( 'Plugin.Zemanta', 'Add your own Zemanta API Key (Aloha.settings.plugins.zemanta.apiKey). Register one here: http://developer.zemanta.com/apps/register/' );
 36 			settings.apiKey = '8n7tl9nikmdps2rhkpusihnt';
 37 		}
 38 
 39 
 40 	/**
 41 	 * Zemanta API Key for SDK
 42 	*/
 43 	window.ZemantaGetAPIKey = function () {
 44 		return settings.apiKey || false;
 45 	}
 46  47 
	/**
 48 	 * register the plugin with unique name
 49 	 */
 50 	return Plugin.create('zemanta', {
 51 		/**
 52 		 * Configure the available languages
 53 		 */
 54 		languages: ['en'],
 55 
 56 		/**
 57 		 * Initialize the plugin and set initialize flag on true
 58 		 */
 59 		init: function(){
 60 			// load Zemanta SDK
 61 			var staticDomain = 'http://static.zemanta.com/';
 62 			Aloha.require([
 63 				'css!' + staticDomain + 'core/zemanta-widget.css',
 64 				staticDomain + 'core/jquery.js',
 65 				staticDomain + 'core/jquery.zemanta.js'
 66 			]);
 67 
 68 			this._initButtons();
 69 			//this._initEvents();
 70 		},
 71 
 72 		_initButtons: function() {
 73 			var self = this;
 74 
 75 			// zemanta button
 76 			FloatingMenu.createScope('Aloha.Zemanta', 'Aloha.continuoustext');
 77 			this.zemantaButton = new Aloha.ui.Button({
 78 				'name' : 'zemanta',
 79 				'iconClass' : 'aloha-button aloha-button-zemanta',
 80 				'size' : 'small',
 81 				'onclick' : function () { self.suggestions(); },
 82 				'tooltip' : i18n.t('button.zemanta.tooltip'),
 83 				'toggle' : true
 84 			});
 85 
 86 			FloatingMenu.addButton(
 87 				'Aloha.continuoustext',
 88 				this.zemantaButton,
 89 				i18n.t('floatingmenu.tab.related'),
 90 				1
 91 			);
 92 		},
 93 
 94 		_initEvents: function() {
 95 			var self = this;
 96 			// update suggestions with smart content change ...
 97 			Aloha.bind('aloha-editable-deactivated', function(event, rangeObject) {
 98 				// do something ... 
 99 			});
100 		},
101 		
102 		suggestions: function() {
103 			var self = this,
104 				widget = document.createElement('div'),
105 				tags = document.createElement('div'),
106 				links = document.createElement('div'),
107 				insertionSpaceId = false,
108 				insertionSpace = false,
109 				t0 = this.now();
110 
111 				this.editableId = Aloha.activeEditable.obj[0].id;
112 
113 				if (this.active == true) {
114 					// remove zemanta widgets (button click to activate/deactivate)
115 					$('#zemanta-sidebar').remove();
116 					$('#zemanta-suggested-tags').remove();
117 					$('#zemanta-suggested-links').remove();
118 					this.active = false;
119 					return;
120 				}
121 				
122 				if ( Aloha.activeEditable.getContents().length < 140 ) {
123 					self.zemantaButton.setPressed(false);
124 					
125 					Aloha.showMessage( new Aloha.Message( {
126 						title : i18n.t( 'Information' ),
127 						text  : i18n.t( 'zemanta.message.shorttext' ),
128 						type  : Aloha.Message.Type.ALERT
129 					} ) );
130 					return;
131 				}
132 
133 				widget.setAttribute('id', 'zemanta-sidebar');
134 				widget.innerHTML = '<div id="zemanta-message" class="zemanta">Loading Zemanta...</div><div id="zemanta-gallery" class="zemanta"></div><div id="zemanta-articles" class="zemanta"></div>';
135 
136 				tags.setAttribute('id', 'zemanta-suggested-tags');
137 				tags.innerHTML = '<div id="zemanta-tags" class="zemanta"><div id="zemanta-tags-div"><ul id="zemanta-tags-div-ul"><li class="zemanta-title">« Tags</li></ul><p class="zem-clear"> </p></div></div>';
138 
139 				links.setAttribute('id', 'zemanta-suggested-links');
140 				links.innerHTML = '<div id="zemanta-links"><ul id="zemanta-links-div-ul"><li class="zemanta-title"><span>Link recommendations will appear here</span> « Links</li></ul><p class="zem-clear"> </p></div>';
141 
142 				// if not set via settings insert the widgets below the current editable
143 				insertionSpaceId = this.editableId;
144 				if ( typeof this.settings.insertionSpaceId !== 'undefined' ) {
145 					insertionSpaceId = this.settings.insertionSpaceId;
146 				}
147 
148 				// insert zemanta widgets after this dom object
149 				// @nicetohave different IDs all available widgets and switch to insert before/after that tag
150 				insertionSpace = document.getElementById(insertionSpaceId);
151 				if ( !insertionSpace ) {
152 					// fallback: if the dom object does not exist (from config) insert it after the current editable
153 					insertionSpaceId = this.editableId;
154 					insertionSpace = document.getElementById(insertionSpaceId);
155 				}
156 
157 				if ( insertionSpace ) {
158 					insertionSpace.parentNode.insertBefore(widget, insertionSpace.nextSibling);
159 					insertionSpace.parentNode.insertBefore(links, insertionSpace.nextSibling);
160 					insertionSpace.parentNode.insertBefore(tags, insertionSpace.nextSibling);
161 				} else {
162 					Aloha.Log.warn( 'Plugin.Zemanta', 'There was a problem inserting the Zemanta widgets.' );
163 				}
164 
165 			try {
166 				$ = window.zQuery;
167 				if (!$) {
168 					throw 'Plugin.Zemanta: No zQuery available.';
169 				}
170 				if ($('#zemanta-message').html() === 'Loading...') {
171 					$('#zemanta-message').html('Preparing...');
172 				}
173 
174 				this.waitForLoad();
175 				this.active = true;
176 			} catch ( er ) {
177 				Aloha.Log.error( 'Plugin.Zemanta', er );
178 			}
179 		},
180 		
181 		/**
182 		 * Get current timestamp
183 		*/
184 		now: function() {
185 			return new Date().getTime();
186 		},
187 
188 		setPlatform: function($, p) {
189 			var editableId = Aloha.activeEditable.obj[0].id;
190 			
191 			return $.zextend(p, {
192 				widget_version: 3,
193 				platform: {
194 					dnd_supported: true,
195 					get_editor: function () {
196 						var editor = {element: null, property: null, type: null, win: null};
197 
198 						try {
199 							editor = {
200 								element: document.getElementById(editableId),
201 								property: 'innerHTML', 
202 								type: 'div',
203 								win: null
204 							}
205 						} catch ( er ) {
206 							Aloha.Log.error( 'Plugin.Zemanta.setPlatform', er );
207 						}
208 						return editor;
209 					}
210 				}
211 			});
212 		},
213 
214 		waitForLoad: function() {
215 			var done = false, t0 = null;
216 
217 			if (typeof $.zemanta === "undefined") {
218 				$('#zemanta-message').html('Waiting...');
219 				return;
220 			}
221 
222 			t0 = this.now();
223 			$('#zemanta-message').html('Initializing...');
224 
225 			try {
226 				done = $.zemanta.initialize(this.setPlatform($, {
227 					interface_type: "alohaeditor",
228 					tags_target_id: "zemanta-tags"
229 				}));
230 			} catch ( er ) {
231 				Aloha.Log.error( 'Plugin.Zemanta.waitForLoad', er );
232 			}
233 
234 			if ( !done ) {
235 				$('#zemanta-message').html('There was a problem initialising the editor.');
236 			} else {
237 				$('#zemanta-control').remove(); // does not work via css
238 			}
239 		}
240 	});
241 });
