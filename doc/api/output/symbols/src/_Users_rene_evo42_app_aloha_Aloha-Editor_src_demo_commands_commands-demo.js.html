1 // Bind to Aloha Ready Event
  2 Aloha.ready( function() {
  3 	var $ = jQuery = Aloha.jQuery;
  4 	$('.edit').aloha();
  5 
  6 var button = jQuery('#bold');
  7  
  8 button.attr( 'disabled',
  9     ( Aloha.queryCommandSupported( 'bold' ) &&
 10     Aloha.queryCommandEnabled( 'bold' ) )
 11 );
 12  
 13 button.click( function() {
 14     Aloha.execCommand( 'bold', false, '' );
 15     updateBoldColor();
 16 });
 17  
 18 Aloha.bind('aloha-selection-changed aloha-command-executed', function() {
 19 	updateBoldColor();
 20 });
 21 
 22 function updateBoldColor() {
 23 	if ( Aloha.queryCommandIndeterm( 'bold' ) ) {
 24 		button.css( 'background-color', 'yellow' );
 25 		return;
 26 	}
 27     button.css( 'background-color', 
 28             Aloha.queryCommandState( 'bold' ) ? 'lightgreen' : 'orange'
 29     );
 30 }
 31 // update the color on startup
 32 updateBoldColor();
 33 
 34 var 
 35 range = Aloha.createRange(),
 36 begin = jQuery( 'p' ),
 37 end = jQuery( 'i' );
 38 
 39 //setStart and setEnd take dom node and the offset as parameters
 40 range.setStart( begin.get(0), 0);
 41 range.setEnd( end.get(0), 1);
 42  43 
//add the range to the selection
 44 Aloha.getSelection().removeAllRanges();
 45 Aloha.getSelection().addRange( range );
 46 
 47 Aloha.execCommand( 'bold', false, '' );
 48 });
