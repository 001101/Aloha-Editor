 11     findContainerProperties: function(element, allowPropertiesInProperties) {
 12         if (!element.attr('typeof')) {
 13             element = element.children('[typeof][about]');
 14         }
 15         return jQuery(element).find('[property]').add(jQuery(element).filter('[property]')).filter(function() {
 16             var closestRDFaEntity = jQuery(this).closest('[typeof][about]');
 17             if (   closestRDFaEntity.index(element) !== 0
 18                 && closestRDFaEntity.length !== 0) {
 19                 // The property is under another entity, skip
 20                 console.log("Wrong entity", element, closestRDFaEntity, this);
 21                 return false;
 22             }
 23 
 24             if (!allowPropertiesInProperties) {
 25                 if (!jQuery(this).parents('[property]').length) {
 26                     return true;
 27                 }
 28                 // This property is under another property, skip
 29                 return false;
 30             }
 31 
 32             return true;
 33         });
 34     },

 36     /**
 37      * @private
 38      */
 39     _getContainerProperties: function(element, emptyValues) {
 40         var containerProperties = {};
 41 
 42         VIE.ContainerManager.findContainerProperties(element, true).each(function() {
 43         	var propertyName;
 44             var objectProperty = jQuery(this);
 45             propertyName = objectProperty.attr('property');
 46 
 47             if (typeof containerProperties[propertyName] !== 'undefined') {
 48                 if (containerProperties[propertyName] instanceof Array) {
 49                     if (emptyValues) {
 50                         return;
 51                     }
 52                     containerProperties[propertyName].push(objectProperty.html());
 53                     return;
 54                 }
 55                 // Multivalued property, convert to Array
 56                 var previousValue = containerProperties[propertyName];
 57                 containerProperties[propertyName] = [];
 58 
 59                 if (emptyValues) {
 60                     return;
 61                 }
 62 
 63                 containerProperties[propertyName].push(previousValue);
 64                 containerProperties[propertyName].push(objectProperty.html());
 65                 return;
 66             }
 67 
 68             if (emptyValues) {
 69                 containerProperties[propertyName] = '';
 70                 return;
 71             }
 72 
 73             containerProperties[propertyName] = objectProperty.html();
 74         });
 75 
 76         return containerProperties;
 77     },

 79     /**
 80      * @private
 81      */
 82     _getContainerValue: function(element, propertyName) {
 83         element = jQuery(element);
 84 
 85         if (typeof element.attr(propertyName) !== 'undefined')
 86         {
 87             // Direct match with container
 88             return element.attr(propertyName);
 89         }
 90         return element.find('[' + propertyName + ']').attr(propertyName);
 91     },

 97     cloneContainer: function(element) {
 98         element = jQuery(element).clone(false);
 99 
100         if (typeof element.attr('about') !== 'undefined')
101         {
102             // Direct match with container
103             element.attr('about', '');
104         }
105         element.find('[about]').attr('about', '');
106 107         VIE.ContainerManager.findContainerProperties(element, false).html('');
108 
109         return element;
110     },

112     getViewForContainer: function(element) {
113         element = jQuery(element);
114         var type = VIE.ContainerManager._getContainerValue(element, 'typeof');
115 
116         if (typeof VIE.ContainerManager.views[type] !== 'undefined') {
117             // We already have a view for this type
118             return VIE.ContainerManager.views[type];
119         }
120 
121         var viewProperties = {};
122         viewProperties.initialize = function() {
123             _.bindAll(this, 'render');
124             this.model.bind('change', this.render);
125         };
126         viewProperties.tagName = element.get(0).nodeName;
127         viewProperties.make = function(tagName, attributes, content) { 
128             return VIE.ContainerManager.cloneContainer(element);
129         };
130         viewProperties.render = function() {
131             var model = this.model;
132             VIE.ContainerManager.findContainerProperties(this.el, true).each(function() {
133                 var propertyElement = jQuery(this);
134                 var propertyName = propertyElement.attr('property');
135 
136                 if (model.get(propertyName) instanceof Array) {
137                     // For now we don't deal with multivalued properties in Views
138                     return true;
139                 }
140 
141                 if (propertyElement.html() !== model.get(propertyName)) {
142                     propertyElement.html(model.get(propertyName));
143                 }
144             });
145             return this;
146         };
147 
148         VIE.ContainerManager.findAdditionalViewProperties(element, viewProperties);
149 
150         VIE.ContainerManager.views[type] = Backbone.View.extend(viewProperties);
151 
152         return VIE.ContainerManager.views[type];
153     },

155     getModelForContainer: function(element) {
156         var type = VIE.ContainerManager._getContainerValue(element, 'typeof');
157         
158 
159         if (typeof VIE.ContainerManager.models[type] !== 'undefined') {
160             // We already have a model for this type
161             return VIE.ContainerManager.models[type];
162         }
163 
164         // Parse the relevant properties from DOM
165         var modelPropertiesFromRdf = VIE.ContainerManager._getContainerProperties(element, true);
166         var modelProperties = jQuery.extend({}, modelPropertiesFromRdf);
167 
168         modelProperties.getType = function() {
169             return type;
170         }
171 
172         modelProperties.toJSONLD = function() {
173             var instance = this;
174             var instanceLD = {"@":"<" + instance.id + ">"};
175 
176             // This can have only one type here, in rdf more types can be allowed
177             instanceLD.a = instance.getType();
178             for (var property in instance.attributes) if(instance.attributes.hasOwnProperty(property)) { //  && typeof instance.attributes[property] != "function"
179                 if (["id"].indexOf(property) == -1)
180                     instanceLD[property] = instance.attributes[property];
181             }
182             return instanceLD;
183         }
184 
185         modelProperties.getInstanceForJSONLD = function(){}
186 
187         VIE.ContainerManager.findAdditionalModelProperties(element, modelProperties);
188 
189         VIE.ContainerManager.models[type] = Backbone.Model.extend(modelProperties);
190 
191         return VIE.ContainerManager.models[type];
192     },

212     registerInstance: function(modelInstance, element) {
213         if (modelInstance.views === undefined) {
214             modelInstance.views = [];
215         }
216 
217         var viewExists = false;
218         jQuery.each(modelInstance.views, function() {
219             // Check whether we already have this view instantiated for the element
220             if (this.el.get(0) == element.get(0)) {
221                 viewExists = true;
222                 return false;
223             }
224         });
225         if (!viewExists) {
226             var view = VIE.ContainerManager.getViewForContainer(element);
227             modelInstance.views.push(new view({model: modelInstance, el: element}));
228         }
229 
230         VIE.ContainerManager.findAdditionalInstanceProperties(element, modelInstance);
231 
232         if (modelInstance.id) {
233             VIE.ContainerManager.instanceSingletons[modelInstance.id] = modelInstance;
234         }
235 
236         if (jQuery.inArray(modelInstance, VIE.ContainerManager.instances) === -1) {
237             VIE.ContainerManager.instances.push(modelInstance);
238         }
239 
240         return modelInstance;
241     },

243     getInstanceForContainer: function(element) {
244         var model = VIE.ContainerManager.getModelForContainer(element);
245         var properties = VIE.ContainerManager._getContainerProperties(element, false);
246         properties.id = VIE.ContainerManager._getContainerValue(element, 'about');
247 
248         if (   !properties.id
249             || VIE.ContainerManager.instanceSingletons[properties.id] === undefined) {
250             var modelInstance = new model(properties);
251         }
252         else 
253         {
254             var modelInstance = VIE.ContainerManager.instanceSingletons[properties.id];
255             modelInstance.set(properties);
256         }
257 
258         return VIE.ContainerManager.registerInstance(modelInstance, element);
259     },
