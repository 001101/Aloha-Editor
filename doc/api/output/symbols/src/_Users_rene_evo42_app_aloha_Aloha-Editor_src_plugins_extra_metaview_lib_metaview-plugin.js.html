  7 define(
  8 ['aloha/plugin', 'aloha/floatingmenu', 'i18n!metaview/nls/i18n', 'i18n!aloha/nls/i18n', 'aloha/jquery', 'css!metaview/css/metaview.css'],
  9 function(Plugin, FloatingMenu, i18n, i18nCore, jQuery) {
 10 	"use strict";
 11 
 12 	var
 13 		$ = jQuery,
 14 		GENTICS = window.GENTICS,
 15 		Aloha = window.Aloha;
 16 
 17      return Plugin.create('metaview', {
 18 		_constructor: function(){
 19 			this._super('metaview');
 20 		},
 21 		
 22 		/**
 23 		 * Configure the available languages
 24 		 */
 25 		languages: ['en', 'de'],
 26 
 27 		
 28 		/**
 29 		 * Initialize the plugin
 30 		 */
 31 		init: function () {
 32 			var that = this;
 33 			
 34 			this.createButtons();
 35 	
 36 			// mark active Editable with a css class
 37 			Aloha.bind(
 38 					"aloha-editable-activated",
 39 					function (jEvent, aEvent) {
 40  41 						if(jQuery(Aloha.activeEditable.obj).hasClass('aloha-metaview')) {
 42 							that.button.setPressed(true);
 43 						} else {
 44 							that.button.setPressed(false);
 45 						}
 46 					}
 47 			);
 48 		},
 49 		
 50 		buttonClick: function() {
 51 			var that = this;
 52 			if(jQuery(Aloha.activeEditable.obj).hasClass('aloha-metaview')) {
 53 				jQuery(Aloha.activeEditable.obj).removeClass('aloha-metaview');
 54 				that.button.setPressed(false);
 55 			} else {
 56 				jQuery(Aloha.activeEditable.obj).addClass('aloha-metaview');
 57 				that.button.setPressed(true);
 58 			}
 59 		},
 60 		
 61 		/**
 62 		 * Initialize the buttons
 63 		 */
 64 		createButtons: function () {
 65 			var that = this;
 66  67 	
 68 			that.button = new Aloha.ui.Button({
 69 				'name' : 'meta',
 70 				'iconClass' : 'aloha-button aloha-button-metaview',
 71 				'size' : 'small',
 72 				'onclick' : function () { that.buttonClick(); },
 73 				'tooltip' : i18n.t('button.switch-metaview.tooltip'),
 74 				'toggle' : true
 75 			});
 76 			FloatingMenu.addButton(
 77 				'Aloha.continuoustext',
 78 				that.button,
 79 				i18nCore.t('floatingmenu.tab.format'),
 80 				1
 81 			);			
 82 		}
 83 	});
 84 });
