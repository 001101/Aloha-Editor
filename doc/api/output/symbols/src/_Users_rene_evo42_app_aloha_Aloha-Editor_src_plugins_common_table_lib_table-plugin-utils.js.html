1 define(
  2 ['aloha/jquery'],
  3 function ($) {
  4 	var Utils = {
  5 		/**
  6 		 * Translates the DOM-Element column offset of a table-cell to the
  7 		 * column offset of a grid-cell, which is the column index adjusted
  8 		 * by other cells' rowspan and colspan values.
  9 		 *
 10 		 * @param rows the rows of a table as an array or jQuery object
 11 		 * @param rowIdx the index in rows of the cell to get the grid column index of
 12 		 * @param colIdx the index in rows[row].cells of the cell to get the grid column index of
 13 		 * @return the grid column index of the cell at the given rowIdx and colIdx, or null
 14 		 *   if the given rowIdx and colIdx coordinates point to a cell outside of the table.
 15 		 */
 16 		'cellIndexToGridColumn': function (rows, rowIdx, colIdx) {
 17 			var gridColumn = null;
 18 			Utils.walkCells(rows, function(ri, ci, walkedGridColumn, rowspan, colspan) {
 19 				if (ri === rowIdx && ci === colIdx) {
 20 					gridColumn = walkedGridColumn;
 21 					return false;
 22 				}
 23 			});
 24 			return gridColumn;
 25 		},
 26 		/**
 27 		 * Walks the table-cells of the table represented by the given rows,
 28 		 * invoking the given callback for each.
 29 		 * @param callback will receive the following arguments
 30 		 *   o ri the row index of the table-cell
 31 		 *   o ci the column index of the table-cell as the offset of the DOM-Element
 32 		 *   o gridCi the column index of the table-cell in a virtual grid that overlays the table (see makeGrid())
 33 		 *   o colspan the colspan attribute of the table-cell (as a number)
 34 		 *   o rowspan the rowspan attribute of the table-cell (as a number)
 35 		 *   returning false from the callback will terminate the walk early.
 36 		 * @return void
 37 		 */
 38 		'walkCells': function (rows, callback) {
 39 			var adjust = [];
 40 			for (var ri = 0; ri < rows.length; ri++) {
 41 				var cells = rows[ri].cells;
 42 				var skip = 0;
 43 				for (var ci = 0; ci < cells.length; ci++) {
 44 					var cell = cells[ci];
 45 					var colspan = Utils.colspan(cell);
 46 					var rowspan = Utils.rowspan(cell);
 47 					
 48 					while (adjust[ci + skip]) {
 49 						adjust[ci + skip] -= 1;
 50 						skip += 1;
 51 					}
 52 
 53 					if (false === callback(ri, ci, ci + skip, colspan, rowspan)) {
 54 						return;
 55 					}
 56 					
 57 					for (var i = 0; i < colspan; i++) {
 58 						if (adjust[ci + skip + i] ) {
 59 							throw "an impossible case has occurred";
 60 						}
 61 						adjust[ci + skip + i] = rowspan - 1;
 62 					}
 63 					skip += colspan - 1;
 64 				}
 65 				for (; ci + skip < adjust.length; skip++) {
 66 					if (adjust[ci + skip]) {
 67 						adjust[ci + skip] -= 1;
 68 					}
 69 				}
 70 			}
 71 		},
 72 		/**
 73 		 * Makes a grid out of the table represented by the given rows.  A
 74 		 * grid will contain one or multiple grid-cells for each table-cell.
 75 		 * A table-cell that has a colspan or rowspan greater than one will
 76 		 * be represented by multiple cells (colspan*rowspan) in the
 77 		 * grid.
 78 		 * @parm rows either an array or an jQuery object holding the DOM
 79 		 *   rows of the table.
 80 		 * @return the table translated to a grid of the form
 81 		 *   [ [cell11, cell12, cell13, ...],
 82 		 *     [cell21, cell22, cell23, ...],
 83 		 *     ... ]
 84 		 *  where each grid-cell is an object containing the following members:
 85 		 *  cell: the DOM object in the table that is rendered into the grid-cell
 86 		 *  colspan: the colspan attribute of the DOM object (as a number)
 87 		 *  rowspan: the rowspan attribute of the DOM object (as a number)
 88 		 *  spannedX: the row offset of the grid-cell in the table-cell (0 based)
 89 		 *  spannedY: the column offset of the grid-cll in the table-cell (0 based)
 90 		 */
 91 		'makeGrid': function (rows) {
 92 			var grid = [];
 93  94 			Utils.walkCells(rows, function(ri, ci, gridCi, colspan, rowspan) {
 95 				var cell = rows[ri].cells[ci];
 96 				for (var spannedY = 0; spannedY < rowspan; spannedY++) {
 97 					grid[ri + spannedY] = grid[ri + spannedY] || [];
 98 					for (var spannedX = 0; spannedX < colspan; spannedX++) {
 99 						grid[ri + spannedY][gridCi + spannedX] = {
100 							'cell'    : cell,
101 							'colspan' : colspan,
102 							'rowspan' : rowspan,
103 							'spannedX': spannedX,
104 							'spannedY': spannedY
105 						};
106 					}
107 				}
108 			});
109 			return grid;
110 		},
111 		/**
112 		 * A grid-cell is said to contain a dom-cell if it is located in the
113 		 * upper-left corner of a table-cell. A table-cell may have a
114 		 * rowspan and colspan, and as such may be comprised of several
115 		 * grid-cells, of which only one (the upper-left corner one)
116 		 * contains a dom-cell.
117 		 * @param cellInfo a cell in the grid returned by makeGrid()
118 		 * @return whether the given grid-cell maps to a dom-cell
119 		 */
120 		'containsDomCell': function (cellInfo) {
121 			return 0 === cellInfo.spannedX && 0 === cellInfo.spannedY;
122 		},
123 		/**
124 		 * A grid-cell may not contain a dom-cell (due to rowspan and
125 		 * colspan). If this function is given the coordinates of such a
126 		 * grid-cell, it will look to the left of the grid-cell, until it
127 		 * finds a grid-cell that contains a dom-cell and returns the
128 		 * DOM-Element of it.
129 		 *
130 		 * This function is useful to insert something into the DOM next to
131 		 * or in place of the grid-cell.
132 		 *
133 		 * @param grid the grid of the table (see makeGrid())
134 		 * @param ri the row index into the grid
135 		 * @param ci the column index into the grid
136 		 * @return the DOM-Element either at or to the left of the grid-cell
137 		 *   a the given coordinates.
138 		 */
139 		'leftDomCell': function (grid, ri, gridCi) {
140 			do {
141 				var cellInfo = grid[ri][gridCi];
142 				if ( 0 === cellInfo.spannedY ) {
143 					return cellInfo.cell;
144 				} 
145 				gridCi -= cellInfo.spannedX + 1;
146 			} while (gridCi >= 0);
147 			return null;
148 		},
149 		/**
150 		 * Given a cell of a table (td/th) with a colspan or rowspan
151 		 * greater than one, will set the rowspan and colspan of the
152 		 * cell to one and insert empty cells where the original cell
153 		 * spanned (the number of cells allocated with createCell will
154 		 * be rowspan * colspan - 1).
155 		 *
156 		 * @param cell
157 		 *        the cell to split
158 		 * @param createCell
159  		 *        a callback that will be invoked rowspan * colspan - 1
160 		 *        times, and which must return a table cell (td/th) that
161 		 *        will be inserted into the table
162 		 */
163 		'splitCell': function (cell, createCell) {
164 			var $cell = $(cell);
165 			var colspan = Utils.colspan( cell );
166 			var rowspan = Utils.rowspan( cell );
167 
168 			//catch the most common case early
169 			if (1 === colspan && 1 === rowspan) {
170 				return;
171 			}
172 
173 			var $row  = $cell.parent();
174 			var $rows = $row.parent().children();
175 			var rowIdx = $row.index();
176 			var colIdx = $cell.index();
177 			var grid = Utils.makeGrid($rows);
178 			var gridColumn = Utils.cellIndexToGridColumn($rows, rowIdx, colIdx);
179 			for (var i = 0; i < rowspan; i++) {
180 				for (var j = (0 === i ? 1 : 0); j < colspan; j++) {
181 					var leftCell = Utils.leftDomCell(grid, rowIdx + i, gridColumn);
182 					if (null == leftCell) {
183 						$rows.eq(rowIdx + i).prepend(createCell());
184 					} else {
185 						$( leftCell ).after(createCell());
186 					}
187 				}
188 			}
189 			$cell.removeAttr('colspan');
190 			$cell.removeAttr('rowspan');
191 		},
192 		/**
193 		 * @param cell
194 		 *        the DOM node for a table cell (td/th)
195 		 * @return
196 		 *        a numeric value indicating the number of rows the cell spans
197 		 */
198 		'rowspan': function (cell) {
199 			return parseInt( $( cell ).attr('rowspan') ) || 1;
200 		},
201 		/**
202 		 * @param cell
203 		 *        the DOM node for a table cell (td/th)
204 		 * @return
205 		 *        a numeric value indicating the number of columns the cell spans
206 		 */
207 		'colspan': function (cell) {
208 			return parseInt( $( cell ).attr('colspan') ) || 1;
209 		},
210 		/**
211 		 * Calls the given callback with each object in the given
212 		 * two-dimensional array.
213 		 *
214 		 * @param grid
215 		 *        A two-dimensional array.
216 		 * @param callback
217 		 *        Invoked with each item in the given two-dimensional array.
218 		 *        Accepts the following parameters:
219 		 *        o item an item in the given two-dimensional array
220 		 *        o x the offset in the nested array (horizontal axis)
221 		 *        o y the offset in the outer array (veritcal axis)
222 		 *        If the callback returns a value identical to false,
223 		 *        the walk will be aborted early.
224 		 */
225 		'walkGrid': function (grid, callback) {
226 			for ( var i = 0; i < grid.length; i++ ) {
227 				for ( var j = 0; j < grid[i].length; j++ ) {
228 					if ( false === callback( grid[ i ][ j ], j, i ) ) {
229 						return;
230 					}
231 				}
232 			}
233 		},
234 		/**
235 		 * Walks the cells of the given grid inside the given
236 		 * coordinates.
237 		 *
238 		 * @param {array} grid
239 		 *        A two-dimensional array.
240 		 * @param {object} rect
241 		 *        Must have the properties top, left, bottom, right
242 		 *        each of which referring to a coordinate in the given grid.
243 		 * @param {function} callback
244 		 *        A callback to invoke for each item in the given
245 		 *        two-dimensional array. See walkGrid() for the
246 		 *        specification of this parameter.
247 		 */
248 		'walkGridInsideRect': function ( grid, rect, callback ) {
249 			Utils.walkGrid( grid, function ( cellInfo, x, y ) {
250 				if ( y >= rect.top && y < rect.bottom && x >= rect.left && x < rect.right ) {
251 					return callback( cellInfo, x, y );
252 				}
253 			});
254 		},
255 		/**
256 		 * Slices leading null or undefined items off of an array
257 		 *
258 		 * @param array
259 		 *        the array to slice null or undefined items off from
260 		 * @return
261 		 *        a new array with the remaining items
262 		 */
263 		'leftTrimArray': function ( array ) {
264 			for (var i = 0; i < array.length; i++) {
265 				if ( null != array[i] ) {
266 					return array.slice( i, array.length );
267 				}
268 			}
269 			return [];
270 		},
271 		/**
272 		 * Given a two-dimensional array, will determine the smallest
273 		 * possible contour that contains all items for which
274 		 * hasCountour returns true.
275 		 *
276 		 * @param grid
277 		 *        A two-dimensional array
278 		 * @param hasContour
279 		 *        Invoked with each item in the given two dimensional array.
280 		 *        Accepts the following parameters:
281 		 *        o item an item in the given two-dimensional array
282 		 *        o x the offset in the nested array (horizontal axis)
283 		 *        o y the offset in the outer array (veritcal axis)
284 		 *        Returns a boolean value indicating whether the item is
285 		 *        considered to have a contour.
286 		 * @return
287 		 *        A set of arrays that indicate a contour
288 		 *        top:    an array of the smallest vertical offsets
289 		 *        right:  an array of the greatest horizontal offsets
290 		 *        bottom: an array of the greatest vertical offsets
291 		 *        left:   an array of the smallest horizontal offsets
292 		 */
293 		'makeContour': function ( grid, hasContour ) {
294 			var left = [];
295 			var right = [];
296 			var top = [];
297 			var bottom = [];
298 			Utils.walkGrid( grid, function ( item, x, y ) {
299 				if ( hasContour( item, x, y ) ) {
300 					if ( null == left[ y ] || x < left[ y ] ) {
301 						left[ y ] = x;
302 					}
303 					if ( null == right[ y ] || x > right[ y ] ) {
304 						right[ y ] = x;
305 					}
306 					if ( null == top[ x ] || y < top[ x ] ) {
307 						top[ x ] = y;
308 					}
309 					if ( null == bottom[ x ] || y > bottom[ x ] ) {
310 						bottom[ x ] = y;
311 					}
312 				}
313 			});
314 			left   = Utils.leftTrimArray(left);
315 			right  = Utils.leftTrimArray(right);
316 			top    = Utils.leftTrimArray(top);
317 			bottom = Utils.leftTrimArray(bottom);
318 			return {'left': left, 'right': right, 'top': top, 'bottom': bottom};
319 		},
320 		/**
321 		 * Returns the index of the first item that doesn't match the given value
322 		 *
323 		 * @param array
324 		 *        An array that contains arbitrary items
325 		 * @param but
326 		 *        A value to ignore while searching in the given array
327 		 * @return
328 		 *        The offset of the first item in the given array that doesn't match the given value.
329 		 *        If no such item was found, -1 is returned.
330 		 */
331 		'indexOfAnyBut': function ( array, but ) {
332 			for ( var i = 0; i < array.length; i++ ) {
333 				if ( but !== array[ i ] ) {
334 					return i;
335 				}
336 			}
337 			return -1;
338 		},
339 		/**
340 		 * @param array
341 		 *        an array of integers
342 		 * @return
343 		 *        true if each item in the given array has a
344 		 *        difference to its neighbor of exactly 1
345 		 */
346 		'isConsecutive': function ( array ) {
347 			for ( var i = 1; i < array.length; i++ ) {
348 				if ( 1 !== Math.abs( array[ i ] - array[ i - 1 ] ) ) {
349 					return false;
350 				}
351 			}
352 			return true;
353 		}
354 	};
355 	return Utils;
356 });
