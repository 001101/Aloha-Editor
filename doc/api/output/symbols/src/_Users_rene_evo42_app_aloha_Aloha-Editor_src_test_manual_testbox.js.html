125 	function queryCommand( ) {
126 		
127 		var execCommand = $command.val();
128 		var	result;
129 		
130 		if ( !selectionRange ) {
131 			return;
132 		}
133 		
134 		if ( !execCommand ) {
135 			jQuery('#aloha-query').hide();
136 			return;
137 		}
138 
139 		jQuery('#aloha-query').show();
140 		jQuery('#aloha-indeterm').show();
141 		jQuery('#aloha-state').show();
142 		jQuery('#aloha-value').show();
143 		
144 		// Select the last range before applying a command
145 		var range = Aloha.createRange();
146 		range.setStart( selectionRange.startContainer, selectionRange.startOffset) ;
147 		range.setEnd( selectionRange.endContainer, selectionRange.endOffset);
148 		Aloha.getSelection().removeAllRanges();
149 		Aloha.getSelection().addRange(range);
150 		
151 		// "If command has no indeterminacy, raise an INVALID_ACCESS_ERR
152 		try {
153 			result = engine.queryCommandIndeterm( execCommand );
154 		} catch (e)	{
155 			if ( e === "INVALID_ACCESS_ERR" ) {
156 				jQuery('#aloha-indeterm').hide();
157 			}
158 		}
159 		jQuery('#aloha-indeterm-result').html( (result ? 'true' : 'false') );
160 
161 		// "If command has no state, raise an INVALID_ACCESS_ERR exception."
162 		try {
163 			result = engine.queryCommandState( execCommand );
164 		} catch (e)	{
165 			if ( e === "INVALID_ACCESS_ERR" ) {
166 				jQuery('#aloha-state').hide();
167 			}
168 		}
169 		jQuery('#aloha-state-result').html( (result ? 'true' : 'false') );
170 		
171 		// "If command has no value, raise an INVALID_ACCESS_ERR exception."
172 		try {
173 			result = engine.queryCommandValue( execCommand );
174 		} catch (e)	{
175 			if ( e === "INVALID_ACCESS_ERR" ) {
176 				jQuery('#aloha-value').hide();
177 			}
178 		}
179 		jQuery('#aloha-value-result').html( result );
180 		
181 	};

183 	/**
184 	 * If applyMarkupOnNextSelection is true: we will copy the value in
185 	 * fillArea into the testArea, and applySelection on it.
186 	 *
187 	 * Otherwise we will check to see if one of either the start of end
188 	 * container is within testArea. If so we will add markers on the current
189 	 * range object, 
190 	 */
191 	function onSelectionChanged ( e ) {
192 		
193 		clearTimeout( selectionTimeout );
194 
195 		// Don't read selection if shift is pressed
196 		if ( e && e.shiftKey ) {
197 			return;
198 		}
199 
200 		if ( applyMarkupOnNextSelection ) {
201 			$testArea[0].innerHTML = $fillArea.val();
202 			applySelection( $testArea );
203 			applyMarkupOnNextSelection = false;
204 		}
205 		
206 		var range = getSelectionRange();
207 		
208 		if ( range ) {
209 			// Check that at least one of either of the end or start containers
210 			// is within the "selectable" testArea
211 			var containers = jQuery( [ range.startContainer,
212 				range.endContainer ] );
213 			
214 			if ( containers.length == 0 ) {
215 				return;
216 			}
217 			
218 			if ( !containers.is( $testArea ) ) {
219 				var parent = containers.parent();
220 				
221 				if ( !parent.is( $testArea ) &&
222 					 parent.parents( '#' + $testArea.attr( 'id' ) )
223 						.length == 0 ) {
224 					return;
225 				}
226 			}
227 			
228 			var timeout = 0;
229 
230 			// For ie wait for double and triple clicks
231 			if( jQuery.browser.msie ) {
232 				timeout = 200;
233 			}
234 
235 			selectionTimeout = setTimeout( function() {
236 				TestUtils.addBrackets( range );
237 				applySelection( $testArea );
238  			}, timeout );
239 		}
240 	};

242 	/**
243 	 * Catches exceptions caused when invoking getRangeAt, without any ranges
244 	 * available.
245 	 *
246 	 * @return {Object:range}
247 	 */

249 		try {
250 			return Aloha.getSelection().getRangeAt( 0 );
251 		} catch ( ex ) {
252 			return null;
253 		}

256 	/**
257 	 * Remove all occurances of the following selection marker delimiters from
258 	 * the given element: {, }, [, ], data-start, data-end
259 	 *
260 	 * @param {DOMElement}
261 	 */
262 	function stripMarkers ( elem ) {
263 		if ( elem && elem.length ) {
264 			elem.html( elem.html().replace(
265 				/\{|\[|(data\-(start|end)\s*=\s*[\"\'][^\"\']*[\"\'])|\}|\]/g,
266 				''
267 			) );
268 		}
269 	};

271 	/**
272 	 * If the given element contains one start selection marker and one end
273 	 * selection marker, then we will attempt to apply the selection on the
274 	 * element.
275 	 *
276 	 * @param {DOMElement} elem - DOM Element whose innerHTML contains
277 	 *							  selection markers defining the selection that
278 	 *							  should be applied to it.
279 	 */
280 	 function applySelection ( elem ) {
281 		 
282 		// Display the current selection in the viewArea
283 		$sourceArea.val( $testArea.html() );
284 		
285 		// convert html for processing
286 		var html = jQuery('<div>').text( $testArea.html() ).html();
287 
288 		html = elem.html();
289 		
290 		var startMarkers = html.match( /\{|\[|data\-start/g ),
291 		    endMarkers = html.match( /\}|\]|data\-end/g ),
292 			numMarkers = 0;
293 		
294 		if ( startMarkers && startMarkers.length ) {
295 			numMarkers += startMarkers.length;
296 		}
297 		
298 		if ( endMarkers && endMarkers.length ) {
299 			numMarkers += endMarkers.length;
300 		}
301 		
302 		if ( numMarkers == 1 ) {
303 			Aloha.Console.warn('Collapsed selection at end of node: ');
304 			Aloha.Console.warn( getRange() );
305 			stripMarkers( elem );
306 		} else if ( numMarkers == 2 ) {
307 			
308 			// Identify the markers and add the new range to the element 
309 			var range = TestUtils.addRange( elem );
310 			var selection = Aloha.getSelection();
311 			selection.removeAllRanges();
312 			
313 			// Add the identified range
314 			selection.addRange( range );
315 			selectionRange = range;
316 			queryCommand();
317 
318 		} else {
319 			// if numMarkers is > 2 then we have a problem, and we will remove
320 			// all markers to create a "clean-sheet"
321 			stripMarkers( elem );
322 		}
323 	 };
