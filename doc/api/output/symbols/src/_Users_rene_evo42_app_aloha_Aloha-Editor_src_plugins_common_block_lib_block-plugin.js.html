  8 /**
  9  * @name block
 10  * @namespace Block plugin
 11  */
 12 define([
 13     'aloha',
 14 	'aloha/plugin',
 15 	'aloha/jquery',
 16 	'aloha/contenthandlermanager',
 17 	'block/blockmanager',
 18 	'block/sidebarattributeeditor',
 19 	'block/block',
 20 	'block/editormanager',
 21 	'block/blockcontenthandler',
 22 	'block/editor',
 23 	'css!block/css/block.css'
 24 ], function(Aloha, Plugin, jQuery, ContentHandlerManager, BlockManager, SidebarAttributeEditor, block, EditorManager, BlockContentHandler, editor) {
 25 	"use strict";
 26 
 27 	/**
 28 	 * Register the plugin with unique name
 29 	 */
 30 	var BlockPlugin = Plugin.create( 'block', {
 31 		settings: {},
 32 //		dependencies: [ 'contenthandler' ],
 33 
 34 		init: function () {
 35 			var that = this;
 36 			// Register default block types
 37 			BlockManager.registerBlockType('DebugBlock', block.DebugBlock);
 38 			BlockManager.registerBlockType('DefaultBlock', block.DefaultBlock);
 39 
 40 			// Register default editors
 41 			EditorManager.register('string', editor.StringEditor);
 42 			EditorManager.register('number', editor.NumberEditor);
 43 			EditorManager.register('url', editor.UrlEditor);
 44 			EditorManager.register('email', editor.EmailEditor);
 45 			
 46 			// register content handler for block plugin
 47 			ContentHandlerManager.register('block', BlockContentHandler);
 48 
 49 			BlockManager.registerEventHandlers();
 50 
 51 			Aloha.bind('aloha-ready', function() {
 52 				// When Aloha is fully loaded, we initialize the blocks.
 53 				that._createBlocks();
 54 				if (that.settings['sidebarAttributeEditor'] !== false) {
 55 					SidebarAttributeEditor.init();
 56 				}
 57 			});
 58 		},
 59 		_createBlocks: function() {
 60 			var defaultBlockSettings;
 61 
 62 			if (!this.settings.defaults) {
 63 				this.settings.defaults = {};
 64 			}
 65 			jQuery.each(this.settings.defaults, function(selector, instanceDefaults) {
 66 				jQuery(selector).alohaBlock(instanceDefaults);
 67 			});
 68 		}
 69 	});
 70 
 71 	/**
 72 	 * See (http://jquery.com/).
 73 	 * @name jQuery
 74 	 * @class
 75 	 * See the jQuery Library  (http://jquery.com/) for full details.  This just
 76 	 * documents the function and classes that are added to jQuery by this plug-in.
 77 	 */
 78 
 79 	/**
 80 	 * See (http://jquery.com/).
 81 	 * @name jQuery.fn
 82 	 * @class
 83 	 * See the jQuery Library  (http://jquery.com/) for full details.  This just
 84 	 * documents the function and classes that are added to jQuery by this plug-in.
 85 	 */
 86 
 87 	/**
 88 	 * Create Aloha blocks from the matched elements
 89 	 * @api
 90 	 * @param {Object} instanceDefaults
 91 	 */
 92 	jQuery.fn.alohaBlock = function(instanceDefaults) {
 93 		instanceDefaults = instanceDefaults || {};
 94 		jQuery(this).each(function(index, element) {
 95 			BlockManager._blockify(element, instanceDefaults);
 96 		});
 97 
 98 		// Chain
 99 		return jQuery(this);
100 	};
101 
102 	// jQuery.fn.mahaloBlock = TODO
103 	return BlockPlugin;
104 });
