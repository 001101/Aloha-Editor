  7 define(
  8 ['aloha/jquery','aloha/plugin', 'aloha/floatingmenu', 'i18n!headerids/nls/i18n', 'i18n!aloha/nls/i18n', 'css!headerids/css/headerids.css'],
  9 function(jQuery, Plugin, FloatingMenu, i18n, i18nCore) {
 10 	"use strict";
 11 
 12 	var
 13 		$ = jQuery,
 14 		GENTICS = window.GENTICS,
 15 		Aloha = window.Aloha;
 16 	
 17 	
 18 	
 19 	// namespace prefix for this plugin
 20     var ns = 'aloha-headerids';
 21     
 22     
 23     // ------------------------------------------------------------------------
 24     // Local (helper) functions
 25     // ------------------------------------------------------------------------
 26     
 27     // Creates a selector string with this component's namepsace prefixed the each classname
 28     function nsSel () {
 29         var strBldr = [], prx = ns;
 30         $.each(arguments, function () { strBldr.push('.' + (this == '' ? prx : prx + '-' + this)); });
 31         return strBldr.join(' ').trim();
 32     };
 33     
 34     // Creates string with this component's namepsace prefixed the each classname
 35     function nsClass () {
 36         var strBldr = [], prx = ns;
 37         $.each(arguments, function () { strBldr.push(this == '' ? prx : prx + '-' + this); });
 38         return strBldr.join(' ').trim();
 39     };
 40     
 41     
 42     
 43 
 44 	
 45 	
 46      return Plugin.create('headerids', {
 47 		_constructor: function(){
 48 			this._super('headerids');
 49 		},
 50 		
 51 		config: ['true'],
 52 				
 53 		/**
 54 		 * Initialize the plugin
 55 		 */
 56 		init: function () {
 57 			var that = this;
 58 	
 59 			// mark active Editable with a css class
 60 			Aloha.bind("aloha-editable-activated", function(jEvent, params) { that.check(params.editable.obj); });
 61 			Aloha.bind("aloha-editable-deactivated", function(jEvent, params) { that.check(params.editable.obj); });
 62 			Aloha.bind('aloha-ready', function (ev) { that.initSidebar(Aloha.Sidebar.right); });
 63 		},
 64 		
 65 		check: function(editable) {
 66 			var that = this;
 67 			var config = that.getEditableConfig(editable);
 68 			if(jQuery.inArray('true',config) == -1) {
 69 				// Return if the plugin should do nothing in this editable
 70 				return;
 71 			}		
 72 			jQuery(editable).find('h1, h2, h3, h4, h5, h6').not('.aloha-customized').each(function(){ 
 73 				that.processH(this); 
 74 			});
 75 		},
 76 		
 77 		processH: function(h) {
 78 			var that = this;
 79 			jQuery(h).attr('id',that.sanitize(jQuery(h).text()));
 80 		},
 81 				
 82 		sanitize: function(str) {
 83 			return (str.replace(/[^a-z0-9]+/gi,'_'));
 84 		},
 85 		
 86 		
 87 		//ns = headerids
 88 		initSidebar: function(sidebar) {
 89 			var pl = this;
 90 			pl.sidebar = sidebar;
 91 			sidebar.addPanel({
 92                     
 93                     id         : nsClass('sidebar-panel'),
 94                     title     : 'Sprungmarken',
 95                     content     : '',
 96                     expanded : true,
 97                     activeOn : 'h1, h2, h3, h4, h5, h6',
 98                     
 99                     onInit     : function () {
100                         var that = this,
101                             content = this.setContent('<label class="'+nsClass('label')+'" for="'+nsClass('input')+'">'+i18n.t('headerids.label.target')+'</label><input id="'+nsClass('input')+'" class="'+nsClass('input')+'" type="text" name="value"/> <button class="'+nsClass('reset-button')+'">'+i18n.t('headerids.button.reset')+'</button><button class="'+nsClass('set-button')+'">'+i18n.t('headerids.button.set')+'</button>').content;
102                         
103                         content.find(nsSel('set-button')).click(function () {
104                             var content = that.content;
105 							jQuery(that.effective).attr('id',jQuery(nsSel('input')).val());
106 							jQuery(that.effective).addClass('aloha-customized');
107                         });
108 						
109 						content.find(nsSel('reset-button')).click(function () {
110                             var content = that.content;
111                             pl.processH(that.effective);
112 							jQuery(that.effective).removeClass('aloha-customized');
113 							that.content.find(nsSel('input')).val(that.effective.attr('id'));
114                         });
115                     },
116                     
117                     onActivate: function (effective) {
118 						var that = this;
119 						that.effective = effective;
120 						that.content.find(nsSel('input')).val(effective.attr('id'));
121                     }
122                     
123                 });
124 			sidebar.show();
125 		}
126 	});
127 });
