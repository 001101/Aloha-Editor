  8 define(
  9 ['aloha/jquery', 'aloha/contenthandlermanager', 'block/blockmanager'],
 10 function(jQuery, ContentHandlerManager, BlockManager) {
 11 
 12 	/**
 13 	 * @name block.BlockContentHandler
 14 	 * @class Special block content handler
 15 	 *
 16 	 * The blog content handler handles pasting of blocks in editables. Pasted
 17 	 * block markup will be replaced by a freshly rendered block instance.
 18 	 */
 19 	var BlockContentHandler = ContentHandlerManager.createHandler(
 20 	/** @lends block.BlockContentHandler */
 21 	{
 22 		/**
 23 		 * Handle the pasting. Remove all unwanted stuff.
 24 		 * @param {jQuery} content
 25 		 */
 26 		handleContent: function( content ) {
 27 			if ( typeof content === 'string' ){
 28 				content = jQuery( '<div>' + content + '</div>' );
 29 			} else if ( content instanceof jQuery ) {
 30 				content = jQuery( '<div>' ).append(content);
 31 			}
 32 
 33 			content.find('.aloha-block').each(function() {
 34 				var oldBlock = jQuery(this);
 35 
 36 				// TODO: use block.serialize();
 37 
 38 				var dataAttributes = {};
 39 				jQuery.each(oldBlock.data(), function(k, v) {
 40 					dataAttributes['data-' + k] = v;
 41 				})
 42 
 43 				var newBlock = jQuery('<' + this.tagName + '/>')
 44 					.attr(jQuery.extend({
 45 							about: oldBlock.attr('about'),
 46 							'class': oldBlock.attr('class')
 47 						}, dataAttributes))
 48 					.removeClass('aloha-block-active');
 49 
 50 				oldBlock.replaceWith(newBlock);
 51 				BlockManager._blockify(newBlock);
 52 			});
 53 			
 54 			return content.html();
 55 		}
 56 	});
 57 	return BlockContentHandler;
 58 });
