  8 /**
  9  * @name contenthandler
 10  * @namespace Content handler plugin
 11  */
 12 define( [
 13 	'aloha',
 14 	'aloha/plugin',
 15 	'aloha/jquery',
 16 	'aloha/contenthandlermanager',
 17 	'contenthandler/wordcontenthandler',
 18 	'contenthandler/genericcontenthandler',
 19 	'contenthandler/oembedcontenthandler',
 20 	'contenthandler/sanitizecontenthandler'
 21 ], function( Aloha,
 22 			 Plugin,
 23 			 jQuery,
 24 			 ContentHandlerManager,
 25 			 WordContentHandler,
 26 			 GenericContentHandler, 
 27 			 OembedContentHandler,
 28 			 SanitizeContentHandler ) {
 29 	'use strict';
 30 
 31 	/**
 32 	 * Register the plugin with unique name
 33 	 */
 34 	var ContentHandlerPlugin = Plugin.create( 'contenthandler', {
 35 		settings : {},
 36 		dependencies : [],
 37 		init : function () {
 38 			var that = this,
 39 				handler, cc,
 40 				contentHandler = [ 'word', 'generic', 'sanitize' ], //  'oembed' deactivated
 41 				i, j = contentHandler.length;
 42 
 43 			// Register available content handler
 44 			for ( i = 0; i < j; i++ ) {
 45 				handler = contentHandler[ i ];
 46 				cc = handler.charAt( 0 ).toUpperCase() + handler.slice( 1 );
 47 				ContentHandlerManager
 48 					.register( handler, eval( cc + 'ContentHandler' ) );
 49 			}
 50 		}
 51 	} );
 52 
 53 	return ContentHandlerPlugin;
 54 });
