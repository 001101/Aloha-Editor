 13 /**********************************************************************************
 14  * INITIALIZE EXAMPLES:
 15  **********************************************************************************
 16  * 	// automatically detect best suited storage driver and use default serializers
 17  *	$.storage = new $.store();
 18  *	// optionally initialize with specific driver and or serializers
 19  *	$.storage = new $.store( [driver] [, serializers] );
 20  *		driver		can be the key (e.g. "windowName") or the driver-object itself
 21  *		serializers	can be a list of named serializers like $.store.serializers
 22  **********************************************************************************
 23  * USAGE EXAMPLES:
 24  **********************************************************************************
 25  *	$.storage.get( key );			// retrieves a value
 26  *	$.storage.set( key, value );	// saves a value
 27  *	$.storage.del( key );			// deletes a value
 28  *	$.storage.flush();				// deletes aall values
 29  **********************************************************************************
 30  */

 41 /**********************************************************************************
 42  * $.store base and convinience accessor
 43  **********************************************************************************/
 44 
 45 $.store = function( driver, serializers )
 46 {
 47 	var JSON = window.JSON
 48 		that = this;
 49 	
 50 	// IE fix
 51 	if (typeof JSON === 'undefined' || !JSON) {
 52 		/*
 53 		 * The GENTICS global namespace object. If GENTICS is already defined, the
 54 		 * existing GENTICS object will not be overwritten so that defined
 55 		 * namespaces are preserved.
 56 		 */
 57 		var JSON = function () {
 58 			this.stringigy = function () {};
 59 			this.parse = function () {};
 60 		};
 61 	}
 62 
 63 	if( typeof driver == 'string' )
 64 	{
 65 		if( $.store.drivers[ driver ] )
 66 			this.driver = $.store.drivers[ driver ];
 67 		else
 68 			throw new Error( 'Unknown driver '+ driver );
 69 	}
 70 	else if( typeof driver == 'object' )
 71 	{
 72 		var invalidAPI = !$.isFunction( driver.init )
 73 			|| !$.isFunction( driver.get )
 74 			|| !$.isFunction( driver.set )
 75 			|| !$.isFunction( driver.del )
 76 			|| !$.isFunction( driver.flush );
 77 
 78 		if( invalidAPI )
 79 			throw new Error( 'The specified driver does not fulfill the API requirements' );
 80 
 81 		this.driver = driver;
 82 	}
 83 	else
 84 	{
 85 		// detect and initialize storage driver
 86 		$.each( $.store.drivers, function()
 87 		{
 88 			// skip unavailable drivers
 89 			if( !$.isFunction( this.available ) || !this.available() )
 90 				return true; // continue;
 91 
 92 			that.driver = this;
 93 			if( that.driver.init() === false )
 94 			{
 95 				that.driver = null;
 96 				return true; // continue;
 97 			}
 98 
 99 			return false; // break;
100 		});
101 	}
102 
103 	// use default serializers if not told otherwise
104 	if( !serializers )
105 		serializers = $.store.serializers;
106 
107 	// intialize serializers
108 	this.serializers = {};
109 	$.each( serializers, function( key, serializer )
110 	{
111 		// skip invalid processors
112 		if( !$.isFunction( this.init ) )
113 			return true; // continue;
114 
115 		that.serializers[ key ] = this;
116 		that.serializers[ key ].init( that.encoders, that.decoders );
117 	});
118 };

121 /**********************************************************************************
122  * $.store API
123  **********************************************************************************/
124 
125 $.extend( $.store.prototype, {
126 	get: function( key )
127 	{
128 		var value = this.driver.get( key );
129 		return this.driver.encodes ? value : this.unserialize( value );
130 	},
131 	set: function( key, value )
132 	{
133 		this.driver.set( key, this.driver.encodes ? value : this.serialize( value ) );
134 	},
135 	del: function( key )
136 	{
137 		this.driver.del( key );
138 	},
139 	flush: function()
140 	{
141 		this.driver.flush();
142 	},
143 	driver : undefined,
144 	encoders : [],
145 	decoders : [],
146 	serialize: function( value )
147 	{
148 		var that = this;
149 
150 		$.each( this.encoders, function()
151 		{
152 			var serializer = that.serializers[ this + "" ];
153 			if( !serializer || !serializer.encode )
154 				return true; // continue;
155 			try
156 			{
157 				value = serializer.encode( value );
158 			}
159 			catch( e ){}
160 		});
161 
162 		return value;
163 	},
164 	unserialize: function( value )
165 	{
166 		var that = this;
167 		if( !value )
168 			return value;
169 
170 		$.each( this.decoders, function()
171 		{
172 			var serializer = that.serializers[ this + "" ];
173 			if( !serializer || !serializer.decode )
174 				return true; // continue;
175 
176 			value = serializer.decode( value );
177 		});
178 
179 		return value;
180 	}
181 });

184 /**********************************************************************************
185  * $.store drivers
186  **********************************************************************************/

188 $.store.drivers = {
189 	// Firefox 3.5, Safari 4.0, Chrome 5, Opera 10.5, IE8
190 	'localStorage': {
191 		// see https://developer.mozilla.org/en/dom/storage#localStorage
192 		ident: "$.store.drivers.localStorage",
193 		scope: 'browser',
194 		available: function()
195 		{
196 			try
197 			{
198 				return !!window.localStorage;
199 			}
200 			catch(e)
201 			{
202 				// Firefox won't allow localStorage if cookies are disabled
203 				return false;
204 			}
205 		},
206 		init: $.noop,
207 		get: function( key )
208 		{
209 			return window.localStorage.getItem( key );
210 		},
211 		set: function( key, value )
212 		{
213 			window.localStorage.setItem( key, value );
214 		},
215 		del: function( key )
216 		{
217 			window.localStorage.removeItem( key );
218 		},
219 		flush: function()
220 		{
221 			window.localStorage.clear();
222 		}
223 	},
224 
225 	// IE6, IE7
226 	'userData': {
227 		// see http://msdn.microsoft.com/en-us/library/ms531424.aspx
228 		ident: "$.store.drivers.userData",
229 		element: null,
230 		nodeName: 'userdatadriver',
231 		scope: 'browser',
232 		initialized: false,
233 		available: function()
234 		{
235 			try
236 237 			{
238 				return !!( document.documentElement && document.documentElement.addBehavior );
239 			}
240 			catch(e)
241 			{
242 				return false;
243 			}
244 		},
245 		init: function()
246 		{
247 			// $.store can only utilize one userData store at a time, thus avoid duplicate initialization
248 			if( this.initialized )
249 				return;
250 
251 			try
252 			{
253 				// Create a non-existing element and append it to the root element (html)
254 				this.element = document.createElement( this.nodeName );
255 				document.documentElement.insertBefore( this.element, document.getElementsByTagName('title')[0] );
256 				// Apply userData behavior
257 				this.element.addBehavior( "#default#userData" );
258 				this.initialized = true;
259 			}
260 			catch( e )
261 			{
262 				return false;
263 			}
264 		},
265 		get: function( key )
266 		{
267 			this.element.load( this.nodeName );
268 			return this.element.getAttribute( key );
269 		},
270 		set: function( key, value )
271 		{
272 			this.element.setAttribute( key, value );
273 			this.element.save( this.nodeName );
274 		},
275 		del: function( key )
276 		{
277 			this.element.removeAttribute( key );
278 			this.element.save( this.nodeName );
279 
280 		},
281 		flush: function()
282 		{
283 			// flush by expiration
284 			this.element.expires = (new Date).toUTCString();
285 			this.element.save( this.nodeName );
286 		}
287 	},
288 
289 	// most other browsers
290 	'windowName': {
291 		ident: "$.store.drivers.windowName",
292 		scope: 'window',
293 		cache: {},
294 		encodes: true,
295 		available: function()
296 		{
297 			return true;
298 		},
299 		init: function()
300 		{
301 			this.load();
302 		},
303 		save: function()
304 		{
305 			window.name = $.store.serializers.json.encode( this.cache );
306 		},
307 		load: function()
308 		{
309 			try
310 			{
311 				this.cache = $.store.serializers.json.decode( window.name + "" );
312 				if( typeof this.cache != "object" )
313 					this.cache = {};
314 			}
315 			catch(e)
316 			{
317 				this.cache = {};
318 				window.name = "{}";
319 			}
320 		},
321 		get: function( key )
322 		{
323 			return this.cache[ key ];
324 		},
325 		set: function( key, value )
326 		{
327 			this.cache[ key ] = value;
328 			this.save();
329 		},
330 		del: function( key )
331 		{
332 			try
333 			{
334 				delete this.cache[ key ];
335 			}
336 			catch(e)
337 			{
338 				this.cache[ key ] = undefined;
339 			}
340 
341 			this.save();
342 		},
343 		flush: function()
344 		{
345 			window.name = "{}";
346 		}
347 	}
348 };

350 /**********************************************************************************
351  * $.store serializers
352  **********************************************************************************/
353 
354 $.store.serializers = {
355 
356 	'json': {
357 		ident: "$.store.serializers.json",
358 		init: function( encoders, decoders )
359 		{
360 			encoders.push( "json" );
361 			decoders.push( "json" );
362 		},
363 		encode: JSON.stringify,
364 		decode: JSON.parse
365 	},
366 
367 	// TODO: html serializer
368 	// 'html' : {},
369 
370 	'xml': {
371 		ident: "$.store.serializers.xml",
372 		init: function( encoders, decoders )
373 		{
374 			encoders.unshift( "xml" );
375 			decoders.push( "xml" );
376 		},
377 
378 		// wouldn't be necessary if jQuery exposed this function
379 		isXML: function( value )
380 		{
381 			var documentElement = ( value ? value.ownerDocument || value : 0 ).documentElement;
382 			return documentElement ? documentElement.nodeName.toLowerCase() !== "html" : false;
383 		},
384 
385 		// encodes a XML node to string (taken from $.jStorage, MIT License)
386 		encode: function( value )
387 		{
388 			if( !value || value._serialized || !this.isXML( value ) )
389 				return value;
390 
391 			var _value = { _serialized: this.ident, value: value };
392 
393 			try
394 			{
395 				// Mozilla, Webkit, Opera
396 				_value.value = new XMLSerializer().serializeToString( value );
397 				return _value;
398 			}
399 			catch(E1)
400 			{
401 				try
402 				{
403 					// Internet Explorer
404 					_value.value = value.xml;
405 					return _value;
406 				}
407 				catch(E2){}
408 			}
409 
410 			return value;
411 		},
412 
413 		// decodes a XML node from string (taken from $.jStorage, MIT License)
414 415 		decode: function( value )
416 		{
417 			if( !value || !value._serialized || value._serialized != this.ident )
418 				return value;
419 
420 			var dom_parser = ( "DOMParser" in window && (new DOMParser()).parseFromString );
421 			if( !dom_parser && window.ActiveXObject )
422 			{
423 				dom_parser = function( _xmlString )
424 				{
425 					var xml_doc = new ActiveXObject( 'Microsoft.XMLDOM' );
426 					xml_doc.async = 'false';
427 					xml_doc.loadXML( _xmlString );
428 					return xml_doc;
429 				}
430 			}
431 432 
433 			if( !dom_parser )
434 			{
435 				return undefined;
436 			}
437 
438 			value.value = dom_parser.call(
439 				"DOMParser" in window && (new DOMParser()) || window,
440 				value.value,
441 				'text/xml'
442 			);
443 
444 			return this.isXML( value.value ) ? value.value : undefined;
445 		}
446 	}
447 };
