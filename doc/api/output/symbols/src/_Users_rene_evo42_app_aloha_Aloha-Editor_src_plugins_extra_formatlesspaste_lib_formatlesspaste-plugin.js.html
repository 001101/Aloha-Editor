  8 define(
  9 ['aloha/core', 'aloha/plugin', 'aloha/jquery', 'aloha/floatingmenu', 
 10  'formatlesspaste/formatlesshandler', 'aloha/contenthandlermanager',
 11  'i18n!formatlesspaste/nls/i18n', 'i18n!aloha/nls/i18n','css!formatlesspaste/css/formatless.css'],
 12 function(Aloha, Plugin, jQuery, FloatingMenu, FormatlessPasteHandler, ContentHandlerManager, i18n, i18nCore) {
 13 	"use strict";
 14 
 15 	
 16 
 17 	// Public Methods
 18 	return Plugin.create('formatlesspaste', {
 19 		
 20 		
 21 		/**
 22 		 * Configure Formatless pasting
 23 		 */
 24 		formatlessPasteOption: false, 
 25 		
 26 		//Here we removes the text-level semantic and edit elements (http://dev.w3.org/html5/spec/text-level-semantics.html#usage-summary)
 27 		strippedElements : [
 28 			"a",
 29 			"em",
 30 			"strong",
 31 			"small",
 32 			"s",
 33 			"cite",
 34 			"q",
 35 			"dfn",
 36 			"abbr",
 37 			"time",
 38 			"code",
 39 			"var",
 40 			"samp",
 41 			"kbd",
 42 			"sub",
 43 			"sup",
 44 			"i",
 45 			"b",
 46 			"u",
 47 			"mark",
 48 			"ruby",
 49 			"rt",
 50 			"rp",
 51 			"bdi",
 52 			"bdo",
 53 			"ins",
 54 			"del" 
 55 		],
 56 
 57 		/**
 58 		 * Initialize the PastePlugin
 59 		 */
 60 		init: function() {
 61 			var that = this;
 62 
 63 			if ( typeof this.settings.formatlessPasteOption !== 'undefined') {
 64 				this.formatlessPasteOption = this.settings.formatlessPasteOption;
 65 			}
 66 			
 67 			if ( typeof this.settings.strippedElements !== 'undefined') {
 68 				this.strippedElements = this.settings.strippedElements;
 69 			}
 70 
 71 			if ( this.formatlessPasteOption ) {
 72 				this.registerFormatlessPasteHandler(); 
 73 			};
 74 
 75 		},
 76 
 77 		/**
 78 		 * Register Formatless paste handler
 79 		 */
 80 		registerFormatlessPasteHandler: function(){
 81 		
 82 			ContentHandlerManager.register( 'formatless', FormatlessPasteHandler );
 83 			FormatlessPasteHandler.strippedElements = this.strippedElements;
 84 			// add button to toggle format-less pasting
 85 			this.formatlessPasteButton = new Aloha.ui.Button({
 86 					'iconClass' : 'aloha-button aloha-button-formatless-paste',
 87 					'size' : 'small',
 88 					'onclick' : function () { 
 89 						//toggle the value of allowFormatless
 90 						FormatlessPasteHandler.enabled = !FormatlessPasteHandler.enabled;
 91 					},
 92 					'tooltip' : i18n.t( 'button.formatlessPaste.tooltip' ),
 93 					'toggle' : true
 94 				});
 95 			FloatingMenu.addButton(
 96 				'Aloha.continuoustext',
 97 				this.formatlessPasteButton,
 98 				i18nCore.t( 'floatingmenu.tab.format' ),
 99 				1
100 			);
101 		},
102 	});
103 });
