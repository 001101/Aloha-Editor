  8 define(['aloha', 'aloha/jquery', 'aloha/floatingmenu', 'aloha/observable', 'aloha/registry'],
  9 function(Aloha, jQuery, FloatingMenu, Observable, Registry) {
 10 	"use strict";
 11 
 12 	var
 13 		GENTICS = window.GENTICS;
 14 
 15 	/**
 16 	 * @name block.blockmanager
 17 	 * @class Block manager singleton
 18 	 */
 19 	var BlockManager = new (Class.extend(Observable,
 20 	/** @lends block.blockmanager */
 21 	{
 22 
 23 		/**
 24 		 * @name block.blockmanager#block-selection-change
 25 		 * @event
 26 		 * @param {Array} selectedBlocks Array of AbstractBlock objects, containing selected blocks. The first element in the array is the innermost block, and the other elements are the parent blocks.
 27 		 */
 28 
 29 		/**
 30 		 * @name block.blockmanager#block-delete
 31 		 * @event fired directly before a block is deleted
 32 		 * @param {AbstractBlock} the block to be deleted
 33 		 */
 34 
 35 		defaults: {
 36 			'block-type': 'DefaultBlock'
 37 		},
 38 
 39 		/**
 40 		 * Registry of block types
 41 		 * @type Registry
 42 		 */
 43 		blockTypes: null,
 44 
 45 		/**
 46 		 * Registry of blocks
 47 		 * @type Registry
 48 		 */
 49 		blocks: null,
 50 
 51 		activeBlocks: null,
 52 
 53 		/**
 54 		 * @constructor
 55 		 */
 56 		_constructor: function() {
 57 			FloatingMenu.createScope('Aloha.Block');
 58 			this.blockTypes = new Registry();
 59 			this.blocks = new Registry();
 60 			this.activeBlocks = {};
 61 		},
 62 
 63 		/**
 64 		 * Register initial event handlers
 65 		 *
 66 		 * @private
 67 		 */
 68 		registerEventHandlers: function() {
 69 			var that = this;
 70 
 71 			// Register event handlers for deactivating an Aloha Block
 72 			jQuery(document).bind('click', function(event) {
 73 				if (that.activeBlocks == {}) return;
 74 				if (jQuery(event.target).parents('.aloha-sidebar-bar, .aloha-block-do-not-deactivate, .aloha-floatingmenu').length > 0
 75 					|| jQuery(event.target).is('.aloha-sidebar-bar, .aloha-block-do-not-deactivate, .aloha-floatingmenu')) {
 76 					// If we are inside the sidebar, the floating menu or other elements which should not trigger the block deactivation, we do an early return.
 77 					return;
 78 				}
 79 				BlockManager._deactivateActiveBlocks();
 80 			});
 81 
 82 			Aloha.bind('aloha-selection-changed', function(evt, selection, originalEvent) {
 83 				// the following line is needed to de-select blocks when navigating over them using the mouse cursors.
 84 				// We only want to execute it though, if we are not inside a block, as it would otherwise
 85 				// directly deselect the block we just selected. This is just a hotfix and not the final solution yet.
 86 				if (selection && jQuery(selection.getCommonAncestorContainer()).parents('.aloha-block').length > 0) {
 87 					return;
 88 				}
 89 				that._deactivateActiveBlocks();
 90 			});
 91 		},
 92 
 93 		/**
 94 		 * Blockify a given element with the instance defaults
 95 		 * Directly called when one does jQuery.alohaBlock(instanceDefaults)
 96 		 *
 97 		 * @private
 98 		 */
 99 		_blockify: function(element, instanceDefaults) {
100 			var attributes, block;
101 			element = jQuery(element);
102 
103 			// TODO: check if object is already Block-ified
104 
105 			attributes = this.getConfig(element, instanceDefaults);
106 
107 			element.contentEditable(false);
108 
109 			if (!this.blockTypes.has(attributes['block-type'])) {
110 				Aloha.Log.error('block/blockmanager', 'Block Type ' + attributes['block-type'] + ' not found!');
111 				return;
112 			}
113 
114 			block = new (this.blockTypes.get(attributes['block-type']))(element);
115 			block.element.addClass('aloha-block-' + attributes['block-type']);
116 
117 			// Save attributes on block, but ignore jquery attribute.
118 			jQuery.each(attributes, function(k, v) {
119 				if (k.indexOf('jQuery') === 0) return;
120 				block.attr(k, v, true);
121 			});
122 
123 			// Remove the attributes from the child element, as they have been moved to the parent element.
124 			jQuery.each(element.data(), function(k, v) {
125 				element.removeAttr('data-' + k);
126 			});
127 			element.removeAttr('about');
128 
129 			// Register block
130 			this.blocks.register(block.getId(), block);
131 
132 			block._renderAndSetContent();
133 		},
134 
135 		/**
136 		 * Deactivate all active blocks
137 		 *
138 		 * @private
139 		 */
140 		_deactivateActiveBlocks: function() {
141 			jQuery.each(jQuery.extend({}, this.activeBlocks), function(id) {
142 				var block = BlockManager.getBlock(id);
143 				if (block) {
144 					block.deactivate();
145 				}
146 			});
147 		},
148 
149 		/**
150 		 * Merges the config from different places, and return the merged config.
151 		 *
152 		 * @private
153 		 */
154 		getConfig: function(blockElement, instanceDefaults) {
155 			// TODO: merge from plugin settings
156 			// TODO: What about double matches / overrides / multiple selectors applying?
157 			var settingsDefaults = {};
158 
159 			return jQuery.extend(
160 				{},
161 				this.defaults,
162 				settingsDefaults,
163 				instanceDefaults,
164 				blockElement.data(),
165 				{	// Override the "about" property
166 					about: blockElement.attr('about')
167 				}
168 			);
169 		},
170 
171 		/**
172 		 * Get a Block instance by id or DOM node
173 		 *
174 		 * @param {String|DOMNode} idOrDomNode
175 		 * @return {block.block.AbstractBlock} Block instance
176 		 */
177 		getBlock: function(idOrDomNode) {
178 			var id;
179 			if (typeof idOrDomNode === 'object') {
180 				id = jQuery(idOrDomNode).attr('id');
181 			} else {
182 				id = idOrDomNode;
183 			}
184 
185 			return this.blocks.get(id);
186 		},
187 188 
		/**
189 		 * Unregister (e.g. remove) the given block
190 		 *
191 		 * @param {Object|String} blockOrBlockId Block or block id
192 		 */
193 		_unregisterBlock: function(blockOrBlockId) {
194 			var id;
195 			if (typeof blockOrBlockId === 'object') {
196 				id = blockOrBlockId.getId();
197 			} else {
198 				id = blockOrBlockId;
199 			}
200 			this.blocks.unregister(blockOrBlockId);
201 		},
202 
203 		/**
204 		 * Register the given block type
205 		 *
206 		 * @param {String} Identifier
207 		 * @param {Class} A class that extends block.block.AbstractBlock
208 		 */
209 		registerBlockType: function(identifier, blockType) {
210 			FloatingMenu.createScope('Aloha.Block.' + identifier, 'Aloha.Block');
211 			this.blockTypes.register(identifier, blockType);
212 		},
213 
214 		/**
215 		 * Get all active blocks indexed by block id
216 		 *
217 		 * @return {Object}
218 		 */
219 		getActiveBlocks: function() {
220 			var activeBlocks = {};
221 			jQuery.each(this.blocks.getEntries(), function(blockId, block) {
222 				if (block.isActive()) {
223 					activeBlocks[blockId] = block;
224 				}
225 			});
226 			return activeBlocks;
227 		},
228 
229 		_setActive: function(block) {
230 			this.activeBlocks[block.id] = true;
231 		},
232 
233 		_setInactive: function(block) {
234 			delete this.activeBlocks[block.id];
235 		}
236 	}))();
237 
238 	Aloha.Block = Aloha.Block || {};
239 	Aloha.Block.BlockManager = BlockManager;
240 
241 	return BlockManager;
242 });
