  8 define([
  9     'aloha/plugin',
 10 	'aloha/floatingmenu',
 11 	'i18n!aloha/nls/i18n'
 12 ], function ( Plugin, FloatingMenu, i18nCore ) {
 13 	'use strict';
 14 	
 15 	return Plugin.create('speak', {
 16 		
 17 		init: function () {
 18 			var that = this;
 19 			
 20 			Aloha.require(['speak/speak','css!speak/css/speak.css']);
 21 			
 22 			Aloha.jQuery('body').append('<div id="audio"></div>')
 23 			
 24 			var button = new Aloha.ui.Button({
 25 				name      : 'speak',
 26 				text      : 'Speak',					// that.i18n('button.' + button + '.text'),
 27 				iconClass : 'GENTICS_button_speak',
 28 				size      : 'small',
 29 				onclick   : function() {
 30 					var range = Aloha.getSelection().getRangeAt( 0 );
 31 					speak( Aloha.jQuery(range.startContainer.parentNode).text() );
 32 				}
 33 			});
 34 			FloatingMenu.addButton(
 35 				'Aloha.continuoustext',
 36 				button,
 37 				i18nCore.t('floatingmenu.tab.format'),
 38 				1
 39 			);
 40 		}
 41 	});
 42 	
 43 });
