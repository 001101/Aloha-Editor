 21 (function () {
 22 	/**
 23 	 * This requirejs module loader loads jquery plugins and wraps them 
 24 	 */
 25 	define({
 26 		load: function (name, require, load, config) {
 27 			var url = require.toUrl(name + '.js');
 28 
 29 			require(['aloha/jquery'], function(value) {
 30 				var $ = value;
 31 				$.ajax({
 32 					type: 'GET',
 33 				  	url: url,
 34 				  	cache: true,
 35 				  	dataType: 'text',
 36 				  	success: function(plugin) {
 37 //						plugin = '(function(jQuery) { var $ = jQuery;\n' + plugin + '}(window.alohaQuery));';
 38 						plugin = 'define([\'aloha/jquery\'], function(jQuery) { var $ = jQuery;\n' + plugin + '});'
 39 						load.fromText(name, plugin);
 40 						load(null);
 41 					}
 42 				});
 43 			});
 44 		}
 45 	});
 46 }());
