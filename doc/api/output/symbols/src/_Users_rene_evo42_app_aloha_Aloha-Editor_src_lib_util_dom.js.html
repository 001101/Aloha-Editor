255 	 * Splits a DOM element at the given position up until the limiting object(s), so that it is valid HTML again afterwards.
256 	 * @param {RangeObject} range Range object that indicates the position of the splitting.
257 	 *				This range will be updated, so that it represents the same range as before the split.
258 	 * @param {jQuery} limit Limiting node(s) for the split.
259 	 *				The limiting node will not be included in the split itself.
260 	 *				If no limiting object is set, the document body will be the limiting object.
261 	 * @param {boolean} atEnd If set to true, the DOM will be splitted at the end of the range otherwise at the start.
262 	 * @return {object} jQuery object containing the two root DOM objects of the split, true if the DOM did not need to be split or false if the DOM could not be split
263 	 * @method
264 	 */
265 	split: function (range, limit, atEnd) {
266 		var
267 			splitElement = jQuery(range.startContainer),
268 			splitPosition = range.startOffset,
269 			updateRange, path, parents,
270 			newDom, insertElement, secondPart,
271 			i, pathLength, element, jqelement, children, newElement,
272 			next, prev, offset;
273 
274 
275 		if (atEnd) {
276 			splitElement = jQuery(range.endContainer);
277 			splitPosition = range.endOffset;
278 		}
279 
280 		if (limit.length < 1) {
281 			limit = jQuery(document.body);
282 		}
283 
284 		// we may have to update the range if it is not collapsed and we are splitting at the start
285 		updateRange = (!range.isCollapsed() && !atEnd);
286 
287 		// find the path up to the highest object that will be splitted
288 		parents = splitElement.parents().get();
289 		parents.unshift(splitElement.get(0));
290 
291 		jQuery.each(parents, function(index, element) {
292 			var isLimit = limit.filter(
293 					function(){
294 						return this == element;
295 					}).length;
296 			if (isLimit) {

299 				}

301 			}

304 		// nothing found to split -> return here
305 		if (! path) {
306 307 			return true;
308 		}
309 
310 		path = path.reverse();
311 
312 		// iterate over the path, create new dom nodes for every element and move
313 		// the contents right of the split to the new element
314 		for( i=0, pathLength = path.length; i < pathLength; ++i) {
315 			element = path[i];
316 			if (i === pathLength - 1) {
317 				// last element in the path -> we have to split it
318 
319 				// split the last part into two parts
320 				if (element.nodeType === 3) {
321 					// text node
322 					secondPart = document.createTextNode(element.data.substring(splitPosition, element.data.length));
323 					element.data = element.data.substring(0, splitPosition);
324 				} else {
325 					// other nodes
326 					jqelement = jQuery(element);
327 					children = jqelement.contents();
328 					newElement = jqelement.clone(false).empty();
329 					secondPart = newElement.append(children.slice(splitPosition, children.length)).get(0);
330 				}
331 
332 				// update the range if necessary
333 				if (updateRange && range.endContainer === element) {
334 					range.endContainer = secondPart;
335 					range.endOffset -= splitPosition;
336 					range.clearCaches();
337 				}
338 
339 				// add the second part
340 				if (insertElement) {
341 					insertElement.prepend(secondPart);
342 				} else {
343 					jQuery(element).after(secondPart);
344 				}
345 			} else {
346 				// create the new element of the same type and prepend it to the previously created element
347 				newElement = jQuery(element).clone(false).empty();
348 
349 				if (!newDom) {
350 					newDom = newElement;
351 				} else {
352 					insertElement.prepend(newElement);
353 				}
354 				insertElement = newElement;
355 
356 				// move all contents right of the split to the new element
357 				while ( true ) {
358 					next = path[i+1].nextSibling;
359 					if ( !next ) { break; }
360 					insertElement.append(next);
361 				}
362 
363 				// update the range if necessary
364 				if (updateRange && range.endContainer === element) {
365 					range.endContainer = newElement.get(0);
366 					prev = path[i+1];
367 					offset = 0;
368 					while ( true ) {
369 						prev = prev.previousSibling;
370 						if ( !prev ) { break; }
371 						offset++;
372 					}
373 					range.endOffset -= offset;
374 					range.clearCaches();
375 				}
376 			}
377 		}
378 
379 		// append the new dom
380 		jQuery(path[0]).after(newDom);
381 
382 		return jQuery([path[0], newDom ? newDom.get(0) : secondPart]);

705 				} else {
