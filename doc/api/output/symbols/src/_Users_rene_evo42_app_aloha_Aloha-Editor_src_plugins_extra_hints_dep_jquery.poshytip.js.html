  9 	var tips = [],
 10 		reBgImage = /^url\(["']?([^"'\)]*)["']?\);?$/i,
 11 		rePNG = /\.png$/i,
 12 		ie6 = $.browser.msie && $.browser.version == 6;

 14 	// make sure the tips' position is updated on resize
 15 	function handleWindowResize() {
 16 		$.each(tips, function() {
 17 			this.refresh(true);
 18 		});
 19 	}
 20 	$(window).resize(handleWindowResize);

 22 	$.Poshytip = function(elm, options) {
 23 		this.$elm = $(elm);
 24 		this.opts = $.extend({}, $.fn.poshytip.defaults, options);
 25 		this.$tip = $(['<div class="',this.opts.className,'">',
 26 				'<div class="tip-inner tip-bg-image"></div>',
 27 				'<div class="tip-arrow tip-arrow-top tip-arrow-right tip-arrow-bottom tip-arrow-left"></div>',
 28 			'</div>'].join('')).appendTo(document.body);
 29 		this.$arrow = this.$tip.find('div.tip-arrow');
 30 		this.$inner = this.$tip.find('div.tip-inner');
 31 		this.disabled = false;
 32 		this.content = null;
 33 		this.init();
 34 	};
 35 
 36 	$.Poshytip.prototype = {
 37 		init: function() {
 38 			tips.push(this);
 39 
 40 			// save the original title and a reference to the Poshytip object
 41 			var title = this.$elm.attr('title');
 42 			this.$elm.data('title.poshytip', title !== undefined ? title : null)
 43 				.data('poshytip', this);
 44 
 45 			// hook element events
 46 			if (this.opts.showOn != 'none') {
 47 				this.$elm.bind({
 48 					'mouseenter.poshytip': $.proxy(this.mouseenter, this),
 49 					'mouseleave.poshytip': $.proxy(this.mouseleave, this)
 50 				});
 51 				switch (this.opts.showOn) {
 52 					case 'hover':
 53 						if (this.opts.alignTo == 'cursor')
 54 							this.$elm.bind('mousemove.poshytip', $.proxy(this.mousemove, this));
 55 						if (this.opts.allowTipHover)
 56 							this.$tip.hover($.proxy(this.clearTimeouts, this), $.proxy(this.mouseleave, this));
 57 						break;
 58 					case 'focus':
 59 						this.$elm.bind({
 60 							'focus.poshytip': $.proxy(this.show, this),
 61 							'blur.poshytip': $.proxy(this.hide, this)
 62 						});
 63 						break;
 64 				}
 65 			}
 66 		},
 67 		mouseenter: function(e) {
 68 			if (this.disabled)
 69 				return true;
 70 
 71 			this.$elm.attr('title', '');
 72 			if (this.opts.showOn == 'focus')
 73 				return true;
 74 
 75 			this.clearTimeouts();
 76 			this.showTimeout = setTimeout($.proxy(this.show, this), this.opts.showTimeout);
 77 		},
 78 		mouseleave: function(e) {
 79 			if (this.disabled || this.asyncAnimating && (this.$tip[0] === e.relatedTarget || jQuery.contains(this.$tip[0], e.relatedTarget)))
 80 				return true;
 81  82 
			var title = this.$elm.data('title.poshytip');
 83 			if (title !== null)
 84 				this.$elm.attr('title', title);
 85 			if (this.opts.showOn == 'focus')
 86 				return true;
 87 
 88 			this.clearTimeouts();
 89 			this.hideTimeout = setTimeout($.proxy(this.hide, this), this.opts.hideTimeout);
 90 		},
 91 		mousemove: function(e) {
 92 			if (this.disabled)
 93 				return true;
 94  95 
			this.eventX = e.pageX;
 96 			this.eventY = e.pageY;
 97 			if (this.opts.followCursor && this.$tip.data('active')) {
 98 				this.calcPos();
 99 				this.$tip.css({left: this.pos.l, top: this.pos.t});
100 				if (this.pos.arrow)
101 					this.$arrow[0].className = 'tip-arrow tip-arrow-' + this.pos.arrow;
102 			}
103 		},
104 		show: function() {
105 			if (this.disabled || this.$tip.data('active'))
106 				return;
107 
108 			this.reset();
109 			this.update();
110 			this.display();
111 			if (this.opts.timeOnScreen)
112 				setTimeout($.proxy(this.hide, this), this.opts.timeOnScreen);
113 		},
114 		hide: function() {
115 			if (this.disabled || !this.$tip.data('active'))
116 				return;
117 
118 			this.display(true);
119 		},
120 		reset: function() {
121 			this.$tip.queue([]).detach().css('visibility', 'hidden').data('active', false);
122 			this.$inner.find('*').poshytip('hide');
123 			if (this.opts.fade)
124 				this.$tip.css('opacity', this.opacity);
125 			this.$arrow[0].className = 'tip-arrow tip-arrow-top tip-arrow-right tip-arrow-bottom tip-arrow-left';
126 			this.asyncAnimating = false;
127 		},
128 		update: function(content, dontOverwriteOption) {
129 			if (this.disabled)
130 				return;
131 
132 			var async = content !== undefined;
133 			if (async) {
134 				if (!dontOverwriteOption)
135 					this.opts.content = content;
136 				if (!this.$tip.data('active'))
137 					return;
138 			} else {
139 				content = this.opts.content;
140 			}
141 
142 			// update content only if it has been changed since last time
143 			var self = this,
144 				newContent = typeof content == 'function' ?
145 					content.call(this.$elm[0], function(newContent) {
146 						self.update(newContent);
147 					}) :
148 					content == '[title]' ? this.$elm.data('title.poshytip') : content;
149 			if (this.content !== newContent) {
150 				this.$inner.empty().append(newContent);
151 				this.content = newContent;
152 			}
153 
154 			this.refresh(async);
155 		},
156 		refresh: function(async) {
157 158 			if (this.disabled)
159 				return;
160 
161 			if (async) {
162 				if (!this.$tip.data('active'))
163 					return;
164 				// save current position as we will need to animate
165 				var currPos = {left: this.$tip.css('left'), top: this.$tip.css('top')};
166 			}
167 
168 			// reset position to avoid text wrapping, etc.
169 			this.$tip.css({left: 0, top: 0}).appendTo(document.body);
170 
171 			// save default opacity
172 			if (this.opacity === undefined)
173 				this.opacity = this.$tip.css('opacity');
174 
175 			// check for images - this code is here (i.e. executed each time we show the tip and not on init) due to some browser inconsistencies
176 			var bgImage = this.$tip.css('background-image').match(reBgImage),
177 				arrow = this.$arrow.css('background-image').match(reBgImage);
178 
179 			if (bgImage) {
180 				var bgImagePNG = rePNG.test(bgImage[1]);
181 				// fallback to background-color/padding/border in IE6 if a PNG is used
182 				if (ie6 && bgImagePNG) {
183 					this.$tip.css('background-image', 'none');
184 					this.$inner.css({margin: 0, border: 0, padding: 0});
185 					bgImage = bgImagePNG = false;
186 				} else {
187 					this.$tip.prepend('<table border="0" cellpadding="0" cellspacing="0"><tr><td class="tip-top tip-bg-image" colspan="2"><span></span></td><td class="tip-right tip-bg-image" rowspan="2"><span></span></td></tr><tr><td class="tip-left tip-bg-image" rowspan="2"><span></span></td><td></td></tr><tr><td class="tip-bottom tip-bg-image" colspan="2"><span></span></td></tr></table>')
188 						.css({border: 0, padding: 0, 'background-image': 'none', 'background-color': 'transparent'})
189 						.find('.tip-bg-image').css('background-image', 'url("' + bgImage[1] +'")').end()
190 						.find('td').eq(3).append(this.$inner);
191 				}
192 				// disable fade effect in IE due to Alpha filter + translucent PNG issue
193 				if (bgImagePNG && !$.support.opacity)
194 					this.opts.fade = false;
195 			}
196 			// IE arrow fixes
197 			if (arrow && !$.support.opacity) {
198 				// disable arrow in IE6 if using a PNG
199 				if (ie6 && rePNG.test(arrow[1])) {
200 					arrow = false;
201 					this.$arrow.css('background-image', 'none');
202 				}
203 				// disable fade effect in IE due to Alpha filter + translucent PNG issue
204 				this.opts.fade = false;
205 			}
206 
207 			var $table = this.$tip.find('table');
208 			if (ie6) {
209 				// fix min/max-width in IE6
210 				this.$tip[0].style.width = '';
211 				$table.width('auto').find('td').eq(3).width('auto');
212 				var tipW = this.$tip.width(),
213 					minW = parseInt(this.$tip.css('min-width')),
214 					maxW = parseInt(this.$tip.css('max-width'));
215 				if (!isNaN(minW) && tipW < minW)
216 					tipW = minW;
217 				else if (!isNaN(maxW) && tipW > maxW)
218 					tipW = maxW;
219 				this.$tip.add($table).width(tipW).eq(0).find('td').eq(3).width('100%');
220 			} else if ($table[0]) {
221 				// fix the table width if we are using a background image
222 				// IE9, FF4 use float numbers for width/height so use getComputedStyle for them to avoid text wrapping
223 				// for details look at: http://vadikom.com/dailies/offsetwidth-offsetheight-useless-in-ie9-firefox4/
224 				$table.width('auto').find('td').eq(3).width('auto').end().end().width(document.defaultView && document.defaultView.getComputedStyle && parseFloat(document.defaultView.getComputedStyle(this.$tip[0], null).width) || this.$tip.width()).find('td').eq(3).width('100%');
225 			}
226 			this.tipOuterW = this.$tip.outerWidth();
227 			this.tipOuterH = this.$tip.outerHeight();
228 
229 			this.calcPos();
230 
231 			// position and show the arrow image
232 			if (arrow && this.pos.arrow) {
233 				this.$arrow[0].className = 'tip-arrow tip-arrow-' + this.pos.arrow;
234 				this.$arrow.css('visibility', 'inherit');
235 			}
236 
237 			if (async) {
238 				this.asyncAnimating = true;
239 				var self = this;
240 				this.$tip.css(currPos).animate({left: this.pos.l, top: this.pos.t}, 200, function() { self.asyncAnimating = false; });
241 			} else {
242 				this.$tip.css({left: this.pos.l, top: this.pos.t});
243 			}
244 		},
245 		display: function(hide) {
246 			var active = this.$tip.data('active');
247 			if (active && !hide || !active && hide)
248 				return;
249 
250 			this.$tip.stop();
251 			if ((this.opts.slide && this.pos.arrow || this.opts.fade) && (hide && this.opts.hideAniDuration || !hide && this.opts.showAniDuration)) {
252 				var from = {}, to = {};
253 				// this.pos.arrow is only undefined when alignX == alignY == 'center' and we don't need to slide in that rare case
254 				if (this.opts.slide && this.pos.arrow) {
255 					var prop, arr;
256 					if (this.pos.arrow == 'bottom' || this.pos.arrow == 'top') {
257 						prop = 'top';
258 						arr = 'bottom';
259 					} else {
260 						prop = 'left';
261 						arr = 'right';
262 					}
263 					var val = parseInt(this.$tip.css(prop));
264 					from[prop] = val + (hide ? 0 : (this.pos.arrow == arr ? -this.opts.slideOffset : this.opts.slideOffset));
265 					to[prop] = val + (hide ? (this.pos.arrow == arr ? this.opts.slideOffset : -this.opts.slideOffset) : 0) + 'px';
266 				}
267 				if (this.opts.fade) {
268 					from.opacity = hide ? this.$tip.css('opacity') : 0;
269 					to.opacity = hide ? 0 : this.opacity;
270 				}
271 				this.$tip.css(from).animate(to, this.opts[hide ? 'hideAniDuration' : 'showAniDuration']);
272 			}
273 			hide ? this.$tip.queue($.proxy(this.reset, this)) : this.$tip.css('visibility', 'inherit');
274 			this.$tip.data('active', !active);
275 		},
276 		disable: function() {
277 			this.reset();
278 			this.disabled = true;
279 		},
280 		enable: function() {
281 			this.disabled = false;
282 		},
283 		destroy: function() {
284 			this.reset();
285 			this.$tip.remove();
286 			delete this.$tip;
287 			this.content = null;
288 			this.$elm.unbind('.poshytip').removeData('title.poshytip').removeData('poshytip');
289 			tips.splice($.inArray(this, tips), 1);
290 		},
291 		clearTimeouts: function() {
292 			if (this.showTimeout) {
293 				clearTimeout(this.showTimeout);
294 				this.showTimeout = 0;
295 			}
296 			if (this.hideTimeout) {
297 				clearTimeout(this.hideTimeout);
298 				this.hideTimeout = 0;
299 			}
300 		},
301 		calcPos: function() {
302 			var pos = {l: 0, t: 0, arrow: ''},
303 				$win = $(window),
304 				win = {
305 					l: $win.scrollLeft(),
306 					t: $win.scrollTop(),
307 					w: $win.width(),
308 					h: $win.height()
309 				}, xL, xC, xR, yT, yC, yB;
310 			if (this.opts.alignTo == 'cursor') {
311 				xL = xC = xR = this.eventX;
312 				yT = yC = yB = this.eventY;
313 			} else { // this.opts.alignTo == 'target'
314 				var elmOffset = this.$elm.offset(),
315 					elm = {
316 						l: elmOffset.left,
317 						t: elmOffset.top,
318 						w: this.$elm.outerWidth(),
319 						h: this.$elm.outerHeight()
320 					};
321 				xL = elm.l + (this.opts.alignX != 'inner-right' ? 0 : elm.w);	// left edge
322 				xC = xL + Math.floor(elm.w / 2);				// h center
323 				xR = xL + (this.opts.alignX != 'inner-left' ? elm.w : 0);	// right edge
324 				yT = elm.t + (this.opts.alignY != 'inner-bottom' ? 0 : elm.h);	// top edge
325 				yC = yT + Math.floor(elm.h / 2);				// v center
326 				yB = yT + (this.opts.alignY != 'inner-top' ? elm.h : 0);	// bottom edge
327 			}
328 
329 			// keep in viewport and calc arrow position
330 			switch (this.opts.alignX) {
331 				case 'right':
332 				case 'inner-left':
333 					pos.l = xR + this.opts.offsetX;
334 					if (pos.l + this.tipOuterW > win.l + win.w)
335 						pos.l = win.l + win.w - this.tipOuterW;
336 					if (this.opts.alignX == 'right' || this.opts.alignY == 'center')
337 						pos.arrow = 'left';
338 					break;
339 				case 'center':
340 					pos.l = xC - Math.floor(this.tipOuterW / 2);
341 					if (pos.l + this.tipOuterW > win.l + win.w)
342 						pos.l = win.l + win.w - this.tipOuterW;
343 					else if (pos.l < win.l)
344 						pos.l = win.l;
345 					break;
346 				default: // 'left' || 'inner-right'
347 					pos.l = xL - this.tipOuterW - this.opts.offsetX;
348 					if (pos.l < win.l)
349 						pos.l = win.l;
350 					if (this.opts.alignX == 'left' || this.opts.alignY == 'center')
351 						pos.arrow = 'right';
352 			}
353 			switch (this.opts.alignY) {
354 				case 'bottom':
355 				case 'inner-top':
356 					pos.t = yB + this.opts.offsetY;
357 					// 'left' and 'right' need priority for 'target'
358 					if (!pos.arrow || this.opts.alignTo == 'cursor')
359 						pos.arrow = 'top';
360 					if (pos.t + this.tipOuterH > win.t + win.h) {
361 						pos.t = yT - this.tipOuterH - this.opts.offsetY;
362 						if (pos.arrow == 'top')
363 							pos.arrow = 'bottom';
364 					}
365 					break;
366 				case 'center':
367 					pos.t = yC - Math.floor(this.tipOuterH / 2);
368 					if (pos.t + this.tipOuterH > win.t + win.h)
369 						pos.t = win.t + win.h - this.tipOuterH;
370 					else if (pos.t < win.t)
371 						pos.t = win.t;
372 					break;
373 				default: // 'top' || 'inner-bottom'
374 					pos.t = yT - this.tipOuterH - this.opts.offsetY;
375 					// 'left' and 'right' need priority for 'target'
376 					if (!pos.arrow || this.opts.alignTo == 'cursor')
377 						pos.arrow = 'bottom';
378 					if (pos.t < win.t) {
379 						pos.t = yB + this.opts.offsetY;
380 						if (pos.arrow == 'bottom')
381 							pos.arrow = 'top';
382 					}
383 			}
384 			this.pos = pos;
385 		}

388 	$.fn.poshytip = function(options) {
389 		if (typeof options == 'string') {
390 			var args = arguments,
391 				method = options;
392 			Array.prototype.shift.call(args);
393 			// unhook live events if 'destroy' is called
394 			if (method == 'destroy')
395 				this.die('mouseenter.poshytip').die('focus.poshytip');
396 			return this.each(function() {
397 				var poshytip = $(this).data('poshytip');
398 				if (poshytip && poshytip[method])
399 					poshytip[method].apply(poshytip, args);
400 			});
401 		}
402 
403 		var opts = $.extend({}, $.fn.poshytip.defaults, options);
404 
405 		// generate CSS for this tip class if not already generated
406 		if (!$('#poshytip-css-' + opts.className)[0])
407 			$(['<style id="poshytip-css-',opts.className,'" type="text/css">',
408 				'div.',opts.className,'{visibility:hidden;position:absolute;top:0;left:0;}',
409 				'div.',opts.className,' table, div.',opts.className,' td{margin:0;font-family:inherit;font-size:inherit;font-weight:inherit;font-style:inherit;font-variant:inherit;}',
410 				'div.',opts.className,' td.tip-bg-image span{display:block;font:1px/1px sans-serif;height:',opts.bgImageFrameSize,'px;width:',opts.bgImageFrameSize,'px;overflow:hidden;}',
411 				'div.',opts.className,' td.tip-right{background-position:100% 0;}',
412 				'div.',opts.className,' td.tip-bottom{background-position:100% 100%;}',
413 				'div.',opts.className,' td.tip-left{background-position:0 100%;}',
414 				'div.',opts.className,' div.tip-inner{background-position:-',opts.bgImageFrameSize,'px -',opts.bgImageFrameSize,'px;}',
415 				'div.',opts.className,' div.tip-arrow{visibility:hidden;position:absolute;overflow:hidden;font:1px/1px sans-serif;}',
416 			'</style>'].join('')).appendTo('head');
417 
418 		// check if we need to hook live events
419 		if (opts.liveEvents && opts.showOn != 'none') {
420 			var deadOpts = $.extend({}, opts, { liveEvents: false });
421 			switch (opts.showOn) {
422 				case 'hover':
423 					this.live('mouseenter.poshytip', function() {
424 						var $this = $(this);
425 						if (!$this.data('poshytip'))
426 							$this.poshytip(deadOpts).poshytip('mouseenter');
427 					});
428 					break;
429 				case 'focus':
430 					this.live('focus.poshytip', function() {
431 						var $this = $(this);
432 						if (!$this.data('poshytip'))
433 							$this.poshytip(deadOpts).poshytip('show');
434 					});
435 					break;
436 			}
437 			return this;
438 		}
439 
440 		return this.each(function() {
441 			new $.Poshytip(this, opts);
442 		});
443 	}

445 	// default settings
446 	$.fn.poshytip.defaults = {
447 		content: 		'[title]',	// content to display ('[title]', 'string', element, function(updateCallback){...}, jQuery)
448 		className:		'tip-yellow',	// class for the tips
449 		bgImageFrameSize:	10,		// size in pixels for the background-image (if set in CSS) frame around the inner content of the tip
450 		showTimeout:		500,		// timeout before showing the tip (in milliseconds 1000 == 1 second)
451 		hideTimeout:		100,		// timeout before hiding the tip
452 		timeOnScreen:		0,		// timeout before automatically hiding the tip after showing it (set to > 0 in order to activate)
453 		showOn:			'hover',	// handler for showing the tip ('hover', 'focus', 'none') - use 'none' to trigger it manually
454 		liveEvents:		false,		// use live events
455 		alignTo:		'cursor',	// align/position the tip relative to ('cursor', 'target')
456 		alignX:			'right',	// horizontal alignment for the tip relative to the mouse cursor or the target element
457 							// ('right', 'center', 'left', 'inner-left', 'inner-right') - 'inner-*' matter if alignTo:'target'
458 		alignY:			'top',		// vertical alignment for the tip relative to the mouse cursor or the target element
459 							// ('bottom', 'center', 'top', 'inner-bottom', 'inner-top') - 'inner-*' matter if alignTo:'target'
460 		offsetX:		-22,		// offset X pixels from the default position - doesn't matter if alignX:'center'
461 		offsetY:		18,		// offset Y pixels from the default position - doesn't matter if alignY:'center'
462 		allowTipHover:		true,		// allow hovering the tip without hiding it onmouseout of the target - matters only if showOn:'hover'
463 		followCursor:		false,		// if the tip should follow the cursor - matters only if showOn:'hover' and alignTo:'cursor'
464 		fade: 			true,		// use fade animation
465 		slide: 			true,		// use slide animation
466 		slideOffset: 		8,		// slide animation offset
467 		showAniDuration: 	300,		// show animation duration - set to 0 if you don't want show animation
468 		hideAniDuration: 	300		// hide animation duration - set to 0 if you don't want hide animation
469 	};
