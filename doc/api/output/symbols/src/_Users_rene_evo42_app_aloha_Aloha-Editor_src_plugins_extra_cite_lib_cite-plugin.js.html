 45 	/**
 46 	 * Simple templating
 47 	 *
 48 	 * @param {String} str - The string containing placeholder keys in curly
 49 	 *                       brackets
 50 	 * @param {Object} obj - Associative array of replacing placeholder keys
 51 	 *                       with corresponding values
 52 	 */
 53 	function supplant (str, obj) {
 54 		 return str.replace(/\{([a-z0-9\-\_]+)\}/ig, function (str, p1, offset, s) {
 55 			 var replacement = obj[p1] || str;
 56 			 return (typeof replacement === 'function') ? replacement() : replacement;
 57 		 });
 58 	};
 59 	
 60 	/**
 61 	 * Wrapper to call the supplant method on a given string, taking the
 62 	 * nsClasses object as the associative array containing the replacement
 63 	 * pairs
 64 	 *
 65 	 * @param {String} str
 66 	 * @return {String}
 67 	 */
 68 	function renderTemplate (str) {
 69 		return (typeof str === 'string') ? supplant(str, nsClasses) : str;
 70 	};

 72 	/**
 73 	 * Generates a selector string with this plugins's namespace prefixed the
 74 	 * each classname
 75 	 *
 76 	 * Usage:
 77 	 * 		nsSel('header,', 'main,', 'foooter ul')
 78 	 * 		will return
 79 	 * 		".aloha-myplugin-header, .aloha-myplugin-main, .aloha-mypluzgin-footer ul"
 80 	 *
 81 	 * @return {String}
 82 	 */
 83 	function nsSel () {
 84 		var strBldr = [], prx = ns;
 85 		$.each(arguments, function () { strBldr.push('.' + (this == '' ? prx : prx + '-' + this)); });
 86 		return $.trim(strBldr.join(' '));
 87 	};
 88 	
 89 	/**
 90 	 * Generates a string with this plugins's namespace prefixed the each
 91 	 * classname
 92 	 *
 93 	 * Usage:
 94 	 *		nsClass('header', 'innerheaderdiv')
 95 	 *		will return
 96 	 *		"aloha-myplugin-header aloha-myplugin-innerheaderdiv"
 97 	 *
 98 	 * @return {String}
 99 	 */
100 	function nsClass () {
101 		var strBldr = [], prx = ns;
102 		$.each(arguments, function () { strBldr.push(this == '' ? prx : prx + '-' + this); });
103 		return $.trim(strBldr.join(' '));
104 	};

106 	/**
107 	 * Coverts hexidecimal string #00ffcc into rgb array [0, 255, 127]
108 	 *
109 	 * @param {String} hex - Hexidecimal string representing color. In the form
110 	 *						 #ff3344 or #f34
111 	 * @return {Array} rgb representation of hexidecimal color
112 	 */
113 	function hex2rgb (hex) {
114 		var hex = hex.replace('#', '').split('');
115 		
116 		if (hex.length == 3) {
117 			hex[5] = hex[4] = hex[2];
118 			hex[3] = hex[2] = hex[1];
119 			hex[1] = hex[0];
120 		}
121 		
122 		var rgb = [];
123 		
124 		for (var i = 0; i < 3; ++i) {
125 			rgb[i] = parseInt('0x' + hex[i * 2] + hex[i * 2 + 1], 16);
126 		}
127 		
128 		return rgb;
129 	};

303 		/**
304 		 * Do a binary search through all citations for a given uid.
305 		 * The bit shifting may be a *bit* of an overkill, but with big lists
306 		 * it proves to be significantly more performant
307 		 *
308 		 * @param {String} uid - uid of citation
309 		 */
310 		getIndexOfCitation: function (uid) {
311 			var c   = this.citations;
312 			var max = c.length;
313 			var min = 0;
314 			var mid;
315 			var cuid;
316 			
317 			// Infinite loop guard for debugging.
318 			// So your tab/browser doesn't freeze up like a Christmas turkey ;-)
319 			// var __guard = 1000;
320 			
321 			while (min < max /* && --__guard */) {
322 				mid = (min + max) >> 1; // Math.floor(i) / 2 == i >> 1 == ~~(i / 2)
323 				if ((cuid = c[mid].uid) == uid) {
324 					return mid;
325 				} else if (cuid > uid) {
326 					max = mid;
327 				} else if (cuid < uid) {
328 					min = mid + 1;
329 				}
330 			}
331 			
332 			return -1;
333 		},

358 		addInlineQuote: function () {
359 			var classes = [nsClass('wrapper'), nsClass(++uid)].join(' ');
360 			var markup = $(supplant(
361 					'<q class="{classes}" data-cite-id="{uid}"></q>',
362 					{uid:uid, classes:classes}
363 				));
364 			var rangeObject = Aloha.Selection.rangeObject;
365 			var foundMarkup;
366 
367 			// now re enable the editable
368 			if (Aloha.activeEditable) {
369 				jQuery(Aloha.activeEditable.obj[0]).click();
370 			}
371 
372 			// check whether the markup is found in the range (at the start of the range)
373 			foundMarkup = rangeObject.findMarkup(function() {
374 				if (this.nodeName && markup.get(0) &&
375 					(typeof this.nodeName === 'string') &&
376 					(typeof markup.get(0).nodeName === 'string')) {
377 					return this.nodeName.toLowerCase() == markup.get(0).nodeName.toLowerCase();
378 				}
379 				
380 				return false;
381 			}, Aloha.activeEditable.obj);
382 
383 			if (foundMarkup) {
384 				// remove the markup
385 				if (rangeObject.isCollapsed()) {
386 					// when the range is collapsed, we remove exactly the one DOM element
387 					domUtils.removeFromDOM(foundMarkup, rangeObject, true);
388 				} else {
389 					// the range is not collapsed, so we remove the markup from the range
390 					domUtils.removeMarkup(rangeObject, markup, Aloha.activeEditable.obj);
391 				}
392 			} else {
393 				// when the range is collapsed, extend it to a word
394 				if (rangeObject.isCollapsed()) {
395 					domUtils.extendToWord(rangeObject);
396 				}
397 
398 				// add the markup
399 				domUtils.addMarkup(rangeObject, markup);
400 			}
401 			
402 			// select the modified range
403 			rangeObject.select();
404 			
405 			this.referenceContainer && this.addCiteToReferences(uid);
406 			
407 			if (this.sidebar) {
408 				this.sidebar.open();
409 			}
410 			//	.activatePanel(nsClass('sidebar-panel'), markup);
411 			
412 			return false;
413 		},

415 		/**
416 		 * Adds an item for the citation matching the given uid to the
417 		 * references list. If no OL list for references exist, we create one.
418 		 * This method will assume that this.referenceContainer is a jQuery
419 		 * object container into which the references list should be built.
420 		 *
421 		 * @param {String} uid - uid of citation
422 		 */
423 		addCiteToReferences: function (uid) {
424 			var index = this.getIndexOfCitation(uid);
425 			
426 			if (index == -1) {
427 				return;
428 			}
429 			
430 			var wrapper = $('.aloha-editable-active ' + nsSel(uid));
431 			var note    = 'cite-note-' + uid;
432 			var ref     = 'cite-ref-'  + uid;
433 			
434 			wrapper.append(
435 				supplant(
436 					'<sup id="{ref}" contenteditable="false"><a href="#{note}">[{count}]</a></sup>',
437 					{
438 						ref   : ref,
439 						note  : note,
440 						count : index + 1
441 					}
442 				)
443 			);
444 			
445 			if (this.referenceContainer.find('ol.references').length == 0) {
446 				this.referenceContainer
447 					.append('<h2>References</h2>')
448 					.append('<ol class="references"></ol>');
449 			}
450 			
451 			this.referenceContainer.find('ol.references').append(
452 				supplant(
453 					'<li id="{note}"><a href="#{ref}">^</a>   <span></span></li>',
454 					{
455 						ref  : ref,
456 						note : note
457 					}
458 				)
459 			);
460 		},

462 		/**
463 		 * Responsible for updating the citation reference in memory, and in
464 		 * the references list when a user adds or changes information for a
465 		 * given citation
466 		 *
467 		 * @param {String} uid
468 		 * @param {String} link
469 		 * @param {String} note
470 		 */
471 		addCiteDetails: function (uid, link, note) {
472 			this.citations[this.getIndexOfCitation(uid)] = {
473 				uid  : uid,
474 				link : link,
475 476 				note : note
477 			};
478 			
479 			if (link) {
480 				// Update link attribute
481 				var el = $(nsSel(uid)).attr('cite', link);
482 				
483 				// Highlight animation for happy user
484 				var round = Math.round;
485 				var from  = hex2rgb('#fdff9a');
486 				var to    = hex2rgb('#fdff9a');
487 				
488 				from.push(1);
489 				to.push(0);
490 				
491 				var diff = [ to[0] - from[0],
492 							 to[1] - from[1],
493 							 to[2] - from[2],
494 							 to[3] - from[3] ];
495 				
496 				el.css({
497 					__tick: 0,
498 					'background-color': 'rgba(' + from.join(',') + ')',
499 					'box-shadow': '0 0 20px rgba(' + from.join(',') + ')'
500 				})
501 				if ( !animating ) {
502 					animating = true;
503 					el.animate({__tick: 1}, {
504 						duration: 500,
505 						easing: 'linear',
506 						step: function (val, fx) {
507 							var rgba = [ round(from[0] + diff[0] * val),
508 							             round(from[1] + diff[1] * val),
509 							             round(from[2] + diff[2] * val),
510 							             from[3] + diff[3] * val   ];
511 							
512 							$(this).css({
513 								'background-color': 'rgba(' + rgba.join(',') + ')',
514 								'box-shadow': '0 0 ' + (20 * (1 - val)) + 'px rgba(' + from.join(',') + ')'
515 							});
516 						},
517 						complete: function() {
518 							animating = false;
519 						}
520 					});
521 				}
522 			}
523 			
524 			// Update information in references list for this citation
525 			if (this.referenceContainer) {
526 				$('li#cite-note-' + uid + ' span').html(
527 					supplant(
528 						link ? '<a class="external" target="_blank" href="{url}">{url}</a>' : '',
529 						{url:link}
530 					) + (note ? '. ' + note : '')
531 				)
532 			}
533 		},
534 		
535 		toString: function () {
536 			return 'aloha-citiation-plugin';
537 		}
