 48 	/**
 49 	 * Marks all cells of the specified column or columns as selected
 50 	 *
 51 	 * @return void
 52 	 */
 53 	TableSelection.prototype.selectColumns = function ( columnsToSelect ) {
 54 		this.unselectCells();
 55 
 56 		var rows = this.table.getRows();
 57 		// first row is the selection row (dump it, it's not needed)
 58 		rows.shift();
 59 		
 60 		var grid = Utils.makeGrid(rows);
 61 		for (var j = 0; j < columnsToSelect.length; j++) {
 62 			// check if this column is already selected.
 63 			if ( -1 !== $.inArray(columnsToSelect[j], this.selectedColumnIdxs) ) {
 64 				continue;
 65  66 			}
 67 			this.selectedColumnIdxs.push( columnsToSelect[j] );
 68 			for (var i = 0; i < grid.length; i++) {
 69 				var cellInfo = grid[i][columnsToSelect[j]];
 70 				if ( Utils.containsDomCell(cellInfo) ) {
 71 					$(cellInfo.cell).addClass(this.table.get('classCellSelected'));
 72 					this.selectedCells.push( cellInfo.cell );
 73 				}
 74 			}
 75 		}
 76 
 77 		this.selectionType = 'column';
 78 	};

 80 	/**
 81 	 * Marks all cells of the specified row or rows as selected
 82 	 *
 83 	 * @return void
 84 	 */
 85 	TableSelection.prototype.selectRows = function ( rowsToSelect ) {
 86 		this.unselectCells();
 87 
 88 		var rows = this.table.getRows();
 89 		
 90  	    rowsToSelect.sort( function ( a, b ) { return a - b; } );
 91 
 92 		for (var i = 0; i < rowsToSelect.length; i++) {
 93 			if ( rows[ rowsToSelect[i] ] ) {
 94 				// check if this row is already selected.
 95 	        	for ( var z = 0; z < this.selectedRowIdxs.length; z++ ) {
 96 	        		if ( rowsToSelect[i] == this.selectedRowIdxs[z] ) {
 97 	        			return;
 98 	        		}
 99 	        	}
100 				this.selectedRowIdxs.push( rowsToSelect[i] );
101 				// to not select first cell, which is a control cell
102 			    for ( var j = 1; j < rows[ rowsToSelect[i] ].cells.length; j++ ) {  
103 					this.selectedCells.push( rows[ rowsToSelect[i] ].cells[j] );
104 					// TODO make proper cell selection method
105 					$( rows[ rowsToSelect[i] ].cells[j] ).addClass( this.table.get('classCellSelected') );
106 			    }
107 			}
108 		}
109 		
110 	    this.selectionType = 'row';
111 	};

113 	TableSelection.prototype.selectAll = function () {
114 		var rowIndices = $.map( this.table.getRows(), function ( item, i ) {
115 			return i;
116 		});
117 
118 		//getRows() returns all rows, even the header row which we must not select
119 		rowIndices.shift();
120 
121 		this.selectRows( rowIndices );

149 	/**
150 	 * This method return true if all sellected cells are TH cells.
151 	 *
152 	 * @return boolean
153 	 */
154 	TableSelection.prototype.isHeader = function ( ) {
155 		
156         if ( this.selectedCells.length == 0 ) {
157         	return false;
158         }
159         
160         // take 1 column to detect if the header button is pressd
161 		for (var i = 0; i < this.selectedCells.length; i++) {
162 			if ( !this.selectedCells[i] || this.selectedCells[i].nodeName.toLowerCase() != 'th' ) {
163 				return false;
164 			}
165 		}
166 		return true;
167 	}

169 	/**
170 	 * This method removes the "selected" class from all selected cells
171 	 *
172 	 * @return void
173 	 */
174 	TableSelection.prototype.unselectCells = function(){
175 		var rows;
176 
177 		//don't unselect cells if cellSelectionMode is active
178 		if ( this.cellSelectionMode ) {
179     		return;
180 		}
181 
182 		if (this.selectedCells.length > 0) {
183 			
184 			rows = this.table.getRows();
185 			
186 			for (var i = 0; i < rows.length; i++) {
187 			    for ( var j = 1; j < rows[i].cells.length; j++ ) {  
188 					// TODO make proper cell selection method
189 					$( rows[i].cells[j] ).removeClass( this.table.get('classCellSelected') );
190 			    }
191 			}
192 
193 			this.selectedCells = new Array();
194 			this.selectedColumnIdxs = new Array();
195 			this.selectedRowIdxs = new Array();
196 
197 			//we keep 'cell' as the default selection type instead of
198 			//unsetting the selectionType to avoid an edge-case where a
199 			//click into a cell doesn't trigger a call to
200 			//TableCell.editableFocs (which would set the 'cell'
201 			//selection type) which would result in the FloatingMenu
202 			//losing the table scope.
203 			this.selectionType = 'cell';
204 
205 			this._notifyCellsUnselected();
206 		}
207 	};

209 	/**
210 	 * Returns the index of a given cell, in selectedCells
211 	 * returns -1 if the given cell is not in selectedCells 
212 	 * @params cell
213 	 *          DOMElement
214 	 *
215 	 * @return integer 
216 	 */
217 	TableSelection.prototype.selectionIndex = function(cell){
218 		for(var i = 0; i < this.selectedCells.length; i++){
219 			if(this.selectedCells[i] === cell){
220 				return i; 
221 			} 
222 		}
223 		return -1;
224 	};

227 	/**
228 	 * Given a contour creates a object representing a rectangle.
229 	 * This function only gives a useful return value if the given
230 	 * contour rectangular.
231 	 *
232 	 * @param {object} contour
233 	 *        a rectangular contour
234 	 * @return {object}
235 	 *        an object with the properties top, right, bottom, left, 
236 	 *        representing the rectangular contour.
237 	 */
238 	function getRectFromContour( contour ) {
239 		return {
240 			'top'   : contour.top[0],
241 			'right' : contour.right[0] + 1,
242 			'bottom': contour.bottom[0] + 1,
243 			'left'  : contour.left[0]
244 		};
245 	}

247 	/**
248 	 * Given a grid and contour, determines whether the contour is
249 	 * rectangular, and each cell in the rectangle is selected.
250 	 *
251 	 * @param {array} grid
252 	 *        a two-dimensional array representing a grid see Utils.makeGrid
253 	 * @param {object} contour
254 	 *        an object reprensenting a contour see Utils.makeContour
255 	 * @param {function} isSelected
256 	 *        a function that determines whether a cell in the given grid
257 	 *        is selected for merging.
258 	 * @return {boolean}
259 	 *        true if all cells inside the contour are selected and can
260 	 *        be merged.
261 	 */
262 	function isMergeable(grid, contour, isSelected) {
263 		var mergeable = true;
264 		if (   -1 !== Utils.indexOfAnyBut( contour.top   , contour.top[0]    )
265 			|| -1 !== Utils.indexOfAnyBut( contour.right , contour.right[0]  )
266 			|| -1 !== Utils.indexOfAnyBut( contour.bottom, contour.bottom[0] )
267 			|| -1 !== Utils.indexOfAnyBut( contour.left  , contour.left[0]   ) ) {
268 			// the outside of the selected area is jagged (not a rectangle)
269 			mergeable = false;
270 		} else {
271 			// the outside of the selected area is a rectangle, but we
272 			// must also ensore that there are no holes in the selection
273 			var rect = getRectFromContour( contour )
274 			Utils.walkGridInsideRect( grid, rect, function ( cellInfo ) {
275 				if ( ! isSelected( cellInfo ) ) {
276 					mergeable = false;
277 					return false;
278 				}
279 			});
280 		}
281 		return mergeable;
282 	}

284 	/**
285 	 * This method merges all selected cells
286 	 *
287 	 * @return void
288 	 */
289 	TableSelection.prototype.mergeCells = function(){
290 
291 		var selectedCells = this.selectedCells;
292 		if ( 0 === selectedCells.length ) {
293 			return;
294 		}
295 
296 		var isSelected = function ( cellInfo ) {
297 			return -1 != $.inArray( cellInfo.cell, selectedCells );
298 		};
299 
300 		var grid = Utils.makeGrid( this.table.getRows() );
301 		var contour = Utils.makeContour( grid, isSelected );
302 
303 		if ( ! isMergeable( grid, contour, isSelected ) ) {
304 			Aloha.showMessage(new Aloha.Message({
305 				title : i18n.t('Table'),
306 				text : i18n.t('table.mergeCells.notRectangular'),
307 				type : Aloha.Message.Type.ALERT
308 			}));
309 			return;
310 		}
311 
312 		var selectedRect = getRectFromContour( contour );
313 		var $firstCell = $( grid[ selectedRect.top ][ selectedRect.left ].cell );
314 		var $firstContainer = $( TableCell.getContainer( $firstCell.get( 0 ) ) );
315 
316 		Utils.walkGridInsideRect( grid, selectedRect, function ( cellInfo, x, y ) {
317 			if (   x - cellInfo.spannedX === selectedRect.left
318 				&& y - cellInfo.spannedY === selectedRect.top ) {
319 				return;
320 			}
321 			var cell = cellInfo.cell;
322 			var contents = $( TableCell.getContainer( cell ) ).contents();
323 			// only append the delimiting space if there is some non-whitespace
324 			for ( var i = 0; i < contents.length; i++ ) {
325 				if (   "string" !== typeof contents[i]
326 				    || "" !== $.trim( contents[i] ) ) {
327 					$firstContainer.append( " " );
328 					$firstContainer.append( contents );
329 					break;
330 				}
331 			}
332 			$( cell ).remove();
333 		});
334 
335 		$firstCell.attr({ 'rowspan': selectedRect.bottom - selectedRect.top,
336 						  'colspan': selectedRect.right  - selectedRect.left });
337 
338 		//select the merged cell
339 		this.selectedCells = [ $firstCell.get( 0 ) ];
340 
341 		//reset flags
342 		this.cellSelectionMode = false; 
343 		this.baseCellPosition = null;
344 		this.lastSelectionRange = null; 
345 		this.selectionType = 'cell';
346 	};

348 	/**
349 	 * This method splits all selected cells (if they are already have row or column spans)
350 	 *
351 	 * @return void
352 	 */
353 	TableSelection.prototype.splitCells = function(){
354 		var selection = this;
355 
356 		// split the selected cells or currently active cell
357 		var cells_to_split = this.selectedCells;
358 		if (cells_to_split.length > 0) {
359 
360 			$(cells_to_split).each(function(){
361 				Utils.splitCell(this, function () {
362 					return selection.table.newActiveCell().obj;
363 				});
364 			});
365 
366 			//reset flags
367 			this.cellSelectionMode = false; 
368 			this.baseCellPosition = null;
369 			this.lastSelectionRange = null; 
370 			this.selectionType = 'cell';
371 		}
372 	};
373 
374 	return TableSelection;
