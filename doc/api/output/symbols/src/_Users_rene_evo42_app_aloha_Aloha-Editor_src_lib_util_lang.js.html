 21 /**
 22  * Takes over all properties from the 'properties' object to the target object.
 23  * If a property in 'target' with the same name as a property in 'properties' is already defined it is overridden.
 24  *
 25  * Example:
 26  *
 27  * var o1 = {a : 1, b : 'hello'};
 28  * var o2 = {a : 3, c : 'world'};
 29  *
 30  * GENTICS.Utils.applyProperties(o1, o2);
 31  *
 32  * Will result in an o1 object like this:
 33  *
 34  * {a : 3, b: 'hello', c: 'world'}
 35  *
 36  * @static
 37  * @return void
 38  */
 39 GENTICS.Utils.applyProperties = function (target, properties) {
 40 	var name;
 41 	for (name in properties) {
 42 		if (properties.hasOwnProperty(name)) {
 43 			target[name] = properties[name];
 44 		}
 45 	}
 46 };

 48 /**
 49  * Generate a unique hexadecimal string with 4 charachters
 50  * @return {string}
 51  */
 52 GENTICS.Utils.uniqeString4 = function () {
 53    return (((1+Math.random())*0x10000)|0).toString(16).substring(1);
 54 };

 56 /**
 57  * Generate a unique value represented as a 32 character hexadecimal string,
 58  * such as 21EC2020-3AEA-1069-A2DD-08002B30309D
 59  * @return {string}
 60  */
 61 GENTICS.Utils.guid = function () {
 62 	var S4 = GENTICS.Utils.uniqeString4;
 63 	return (S4()+S4()+'-'+S4()+'-'+S4()+'-'+S4()+'-'+S4()+S4()+S4());
 64 };
