  8 define(
  9 ['aloha/core', 'aloha/jquery', 'aloha/command', 'aloha/selection', 'util/dom', 'aloha/contenthandlermanager', 'aloha/console'],
 10 function(Aloha, jQuery, command, selection, dom, ContentHandlerManager, console) {
 11 	"use strict";
 12 
 13 	// Exported commands
 14 	command.register( 'inserthtml', {
 15 		action: function(value, range) {
 16 			var 
 17 				$editable = jQuery(dom.getEditingHostOf(range.startContainer)),
 18 				cac = range.commonAncestorContainer,
 19 				i,
 20 				selectedRange,
 21 				domNodes = [];
 22 			
 23 			/**
 24 			 * Paste the given object into the current selection.
 25 			 * If inserting fails (because the object is not allowed to be inserted), unwrap the contents and try with that.
 26 			 * @param object object to be pasted
 27 			 */
 28 			function pasteElement(object) {
 29 				var $object = jQuery(object),
 30 					contents;
 31 
 32 				// try to insert the element into the DOM with limit the editable host
 33 				// this fails when an element is not allowed to be inserted
 34 				if (!dom.insertIntoDOM($object, range, $editable, false)) {
 35 					
 36 					// if that is not possible, we unwrap the content and insert every child element
 37 					 contents = $object.contents();
 38 
 39 					// when a block level element was unwrapped, we at least insert a break
 40 					if (dom.isBlockLevelElement(object) || dom.isListElement(object)) {
 41 						pasteElement(jQuery('<br/>').get(0));
 42 					}
 43 
 44 					// and now all children (starting from the back)
 45 					for ( i = contents.length - 1; i >= 0; --i) {
 46 						pasteElement(contents[i]);
 47 					}
 48 				}
 49 			};
 50 
 51 			// apply content handler to cleanup inserted data
 52 			//if (typeof Aloha.settings.contentHandler.insertHtml === 'undefined') {
 53 			// just use all registerd content handler or specity Aloha.defaults.contentHandler.insertHtml manually?
 54 			//	Aloha.settings.contentHandler.insertHtml = Aloha.defaults.contentHandler.insertHtml;
 55 			//}
 56 			value = ContentHandlerManager.handleContent( value, { contenthandler: Aloha.settings.contentHandler.insertHtml } );
 57 
 58 			// allowed values are string or jQuery objects
 59 			// add value to a container div
 60 			if ( typeof value === 'string' ){
 61  62 				value = jQuery( '<div>' + value + '</div>' );
 63 			} else if ( value instanceof jQuery ) {
 64 				value = jQuery( '<div>' ).append(value);
 65 			} else {
 66 				throw "INVALID_VALUE_ERR";
 67 			}
 68 			
 69 			// get contents of container div
 70 			domNodes = value.contents();
 71 			
 72 			// check if range starts an ends in same editable host
 73 //			if ( !(dom.inSameEditingHost(range.startContainer, range.endContainer)) ) {
 74 //				throw "INVALID_RANGE_ERR";
 75 //			}
 76 			
 77 			// delete currently selected contents
 78 			dom.removeRange(range);
 79 			
 80 			for ( i = domNodes.length - 1; i >= 0; --i) {
 81 				// insert the elements
 82 				pasteElement(domNodes[i]);
 83 			}
 84 
 85 			// Call collapse() on the context object's Selection,
 86 			// with last child's parent as the first argument and one plus its index as the second.
 87 			if (domNodes.length > 0) {
 88 				range = dom.setCursorAfter(domNodes.get(domNodes.length - 1));
 89 			} else {
 90 				// if nothing was pasted, just reselect the old range
 91 				range.select();
 92 			}
 93 
 94 			dom.doCleanup({merge:true, removeempty: true}, range, cac);
 95 			//In some cases selecting the range does not work properly 
 96 			//e.g. when pasting from word in an h2 after the first character in IE
 97 			//in these cases we should fail gracefully.
 98 			//TODO check why the selection is failing
 99 			try {
100 				range.select();
101 			} catch (e) {
102 				console.warn('Error:',e);
103 			}
104 
105 		}
106 	});
107 
108 });
