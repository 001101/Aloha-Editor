1 /**
  2  * In order to choose the correct internationalization for among the
  3  * grid.locale.{lang} files, we will manually require each of these
  4  * grid.locale.* files (where * is the language code of the locale). These
  5  * files have each been slightly modified to no longer extend Aloha.jQuery with
  6  * a "jgrid" object, but rather a jgrid_* object. So instead of the jQuery
  7  * object having a single "jgrid" object, it will have several objects with
  8  * names like "jgrid_en", and "jgrid_de".
  9  * In order to determine which of these i18n objects should be used with the
 10  * browser, we read a "jgrid.locale" key from the i18n.js file in browser/nls,
 11  * an use this value to choose which of the jgrid_* objects, Aloha.jQuery.jgrid
 12  * should point to
 13  */
 14 define( [
 15 	'aloha/jquery',
 16 	'i18n!browser/nls/i18n',
 17 	'jquery-plugin!browser/../vendor/grid.locale.en',
 18 	'jquery-plugin!browser/../vendor/grid.locale.de'
 19 ], function ( jQuery, i18n ) {
 20 	var locale = i18n[ 'jgrid.locale' ] || 'en';
 21 	if ( typeof jQuery.jgrid == 'undefined' ) {
 22 		jQuery.jgrid = {};
 23 	}
 24 	jQuery.extend( jQuery.jgrid, jQuery[ 'jgrid_' + locale ] );
 25 	jQuery.jgrid_en = jQuery.jgrid_de = void 0;
 26 } );
