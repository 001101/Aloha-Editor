 13 define(
 14 [ 'aloha', 'aloha/jquery' ],
 15 function ( Aloha, jQuery ) {
 16 	'use strict'
 17 	
 18 	/**
 19 	 * Internal data as array with following format:
 20 	 *
 21 	 * [
 22 	 *   { name: 'Aloha Editor - The HTML5 Editor', url:'http://aloha-editor.com', type:'website' },
 23 	 *   { name: 'Aloha Logo', url:'http://www.aloha-editor.com/images/aloha-editor-logo.png', type:'image'  }
 24 	 * ];
 25 	 *
 26 	 * @private
 27 	 */
 28 	var urlset = [
 29 		{ name: 'Aloha Test', url: '#alohatest', type: 'website' },
 30 		{ name: 'Test One', url: '#test1', type: 'website' },
 31 		{ name: 'Test Two', url: '#test2', type: 'website' },
 32 		{ name: 'Test Three', url: '#test3', type: 'website' },
 33 		{ name: 'Test Four', url: '#test4', type: 'image' }
 34 	];
 35 	
 36 	new ( Aloha.AbstractRepository.extend( {
 37 		
 38 		_constructor: function () {
 39 			this._super( 'slowlinklist' );
 40 		},
 41 		
 42 		/**
 43 		 * Internal folder structure
 44 		 * @hide
 45 		 */
 46 		folder: [],
 47 		
 48 		/**
 49 		 * initalize LinkList, parse all links, build folder structure and add
 50 		 * additional properties to the items
 51 		 */
 52 		init: function () {
 53 			// Add ECMA262-5 Array method filter if not supported natively.
 54 			// But we will be very conservative and add to this single array
 55 			// object so that we do not tamper with the native Array prototype
 56 			// object
 57 			if ( !( 'filter' in Array.prototype ) ) {
 58 				urlset.filter = function ( filter, that /*opt*/ ) {
 59 					var other = [],
 60 					    v,
 61 					    i = 0,
 62 					    n = this.length;
 63 					
 64 					for ( ; i < n; i++ ) {
 65 						if ( i in this && filter.call( that, v = this[ i ], i, this ) ) {
 66 							other.push( v );
 67 						}
 68 					}
 69 					
 70 					return other;
 71 				};
 72 			}
 73 			
 74 			var l = urlset.length;
 75 			
 76 			// generate folder structure
 77 		    for ( var i = 0; i < l; ++i ) {
 78 		    	var e = urlset[ i ];
 79 		    	e.repositoryId = this.repositoryId;
 80 		    	e.id = e.id ? e.id : e.url;
 81 				
 82 		    	var u = e.uri = this.parseUri( e.url ), 
 83 					// add hostname as root folder
 84 		    	    path = this.addFolder( '', u.host ),
 85 				    pathparts = u.path.split( '/' );
 86 				
 87 		    	for ( var j = 0; j < pathparts.length; j++ ) {
 88 		    		if ( pathparts[ j ] &&
 89 		    			 // It's a file because it has an extension.
 90 		    			 // Could improve this one :)
 91  92 		    			 pathparts[ j ].lastIndexOf( '.' ) < 0 ) {
 93 			    		path = this.addFolder( path, pathparts[ j ] );
 94 		    		}
 95 		    	}
 96 				
 97 		    	e.parentId = path;
 98 		    	
 99 				urlset[ i ] = new Aloha.RepositoryDocument( e );
100 		    }
101 			
102 		    this.repositoryName = 'Linklist';
103 		},
104 		
105 		/**
106 		 * @param {String} path
107 		 * @param {String} name
108 		 * @return {String}
109 		 */
110 		addFolder: function ( path, name ) {
111 			var type = path ? 'folder' : 'hostname',
112 			    p = path ? path + '/' + name : name;
113 			
114 			if ( name && !this.folder[ p ] ) {
115 				this.folder[ p ] = new Aloha.RepositoryFolder( {
116 					id: p,
117 					name: name || p,
118 					parentId: path,
119 					type: 'host',
120 					repositoryId: this.repositoryId
121 				} );
122 			}
123 			
124 			return p;
125 		},
126 		
127 		/**
128 		 * Searches a repository for object items matching query if
129 		 * objectTypeFilter. If none is found it returns null.
130 		 *
131 		 * @param {Object} p
132 		 * @param {Function} callback
133 		 */
134 		query: function ( p, callback ) {
135 			// Not supported; filter, orderBy, maxItems, skipcount, renditionFilter
136 			var r = new RegExp( p.queryString, 'i' );
137 			
138 			var d = urlset.filter( function ( e, i, a ) {
139 				return (
140 					( !p.queryString || e.name.match( r ) || e.url.match( r ) ) &&
141 					( !p.objectTypeFilter || ( !p.objectTypeFilter.length ) || jQuery.inArray( e.type, p.objectTypeFilter ) > -1 ) &&
142 					true //( !p.inFolderId || p.inFolderId == e.parentId )
143 				);
144 			} );
145 			
146 			setTimeout( function () {
147 				callback.call( this, d );
148 			}, 2000 );
149 		},
150 		
151 		/**
152 		 * returns the folder structure as parsed at init
153 		 *
154 		 * @param {Object} p
155 		 * @param {Function} callback
156 		 */
157 		getChildren: function ( p, callback ) {
158 			var d = [],
159 			    e;
160 			
161 			for ( e in this.folder ) {
162 				var l = this.folder[ e ].parentId;
163 				if ( typeof this.folder[ e ] != 'function' && ( // extjs prevention
164 165 					this.folder[ e ].parentId == p.inFolderId || // all subfolders
166 					( !this.folder[ e ].parentId && p.inFolderId == this.repositoryId ) // the hostname
167 				) ) {
168 					d.push( this.folder[ e ] );
169 				}
170 			}
171 			
172 			setTimeout( function () {
173 				callback.call( this, d );
174 			}, 2000 );
175 		},
176 		
177 		//parseUri 1.2.2
178 		//(c) Steven Levithan <stevenlevithan.com>
179 		//MIT License
180 		//http://blog.stevenlevithan.com/archives/parseuri
181 		parseUri: function(str) {
182 			var	o = {
183 					strictMode: false,
184 					key: [ "source","protocol","authority","userInfo","user","password","host","port","relative","path","directory","file","query","anchor"],
185 					q: {
186 						name: "queryKey",
187 						parser: /(?:^|&)([^&=]*)=?([^&]*)/g
188 					},
189 					parser: {
190 						strict: /^(?:([^:\/?#]+):)?(?:\/\/((?:(([^:@]*)(?::([^:@]*))?)?@)?([^:\/?#]*)(?::(\d*))?))?((((?:[^?#\/]*\/)*)([^?#]*))(?:\?([^#]*))?(?:#(.*))?)/,
191 						loose: /^(?:(?![^:@]+:[^:@\/]*@)([^:\/?#.]+):)?(?:\/\/)?((?:(([^:@]*)(?::([^:@]*))?)?@)?([^:\/?#]*)(?::(\d*))?)(((\/(?:[^?#](?![^?#\/]*\.[^?#\/.]+(?:[?#]|$)))*\/?)?([^?#\/]*))(?:\?([^#]*))?(?:#(.*))?)/
192 					}
193 				},
194 				m = o.parser[o.strictMode ? "strict" : "loose"].exec(str),
195 				uri = {},
196 				i = 14;
197 			
198 			while (i--) uri[o.key[i]] = m[i] || "";
199 			
200 			uri[o.q.name] = {};
201 			uri[o.key[12]].replace(o.q.parser, function ($0, $1, $2) {
202 				if ($1) uri[o.q.name][$1] = $2;
203 			});
204 			
205 			return uri;
206 		},
207 		
208 		/**
209 		 * Get the repositoryItem with given id
210 		 * Callback: {GENTICS.Aloha.Repository.Object} item with given id
211 		 * @param itemId {String} id of the repository item to fetch
212 		 * @param callback {function} callback function
213 		 */
214 		getObjectById: function ( itemId, callback ) {
215 			var i = 0,
216 			    l = urlset.length,
217 			    d = [];
218 			
219 			for ( ; i < l; i++ ) {
220 				if ( urlset[ i ].id == itemId ) {
221 					d.push( urlset[ i ] );
222 				}
223 			}
224 			
225 			callback.call( this, d );
226 		}
227 		
228 	} ) )();
229 
230 } );
