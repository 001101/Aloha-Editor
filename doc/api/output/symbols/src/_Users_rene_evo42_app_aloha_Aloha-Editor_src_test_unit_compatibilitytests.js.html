  7 define( [], function() {
  8 	"use strict";
  9 
 10 	// Start the tests, when Aloha is ready
 11 	Aloha.bind('aloha-ready', function() {
 12 		
 13 		// for IE, it is necessary to have this setTimeout. Otherwise, the
 14 		// editable would not be recognized as having contentEditable turned on
 15 		// right after editable.aloha
 16 		setTimeout(function() {
 17 
 18 			// Test if jQuery is present
 19 			test( 'jQuery present before .noConflict()', function() {
 20 				equals( beforeNoConflict.fn.jquery, '1.5.1', 'Legacy jQuery version is correct' );
 21 			});
 22 			
 23 			// Test if jQuery has been removed
 24 			test( 'jQuery removed from window', function() {
 25 				equals( typeof afterNoConflict, 'undefined', 'Legacy jQuery version is correct' );
 26 			});
 27 			
 28 			// Test if legacy jQuery version is correct
 29 			test( 'jQuery compatibility test', function() {
 30 				equals( window.jQuery.fn.jquery, '1.2.1', 'Legacy jQuery version is correct' );
 31 			});
 32 		
 33 			// Test if legacy jQuery version is correct
 34 			test( 'Aloha.jQuery test', function() {
 35 				equals( Aloha.jQuery.fn.jquery, '1.5.1', 'Loaded jQuery version is correct' );
 36 			});
 37 		
 38 			// Test if legacy jQuery version is correct
 39 			test( 'Aloha.require("aloha/jquery") test', function() {
 40 				equals( Aloha.require("aloha/jquery").fn.jquery, '1.5.1', 'Required jQuery version is correct' );
 41 			});
 42 			
 43 			// Test if the jquery plugins necessary for aloha core are attached to the Aloha jQuery object
 44 			test( 'jquery core plugin test', function() {
 45 				equals( typeof Aloha.jQuery.store, 'function', 'Check whether the jQuery plugin "store" was attached to Aloha jQuery' );
 46 				equals( typeof Aloha.jQuery.toJSON, 'function', 'Check whether the jQuery plugin "json" was attached to Aloha jQuery' );
 47 			});
 48 		
 49 			// Test if third party jquery plugins, loaded with the 'jquery-plugin' loader are attached to the Aloha jQuery object
 50 			test( '3rd party jquery plugin test', function() {
 51 				// plugintest2 tries to load a jquery plugin with require in 'aloha' context. 
 52 				equals( typeof Aloha.jQuery().alohaTest, 'function', 'Check whether the jQuery plugin "alohaTest" was attached to Aloha jQuery' );
 53 			});
 54 		}, 1);
 55 	});
 56 });
