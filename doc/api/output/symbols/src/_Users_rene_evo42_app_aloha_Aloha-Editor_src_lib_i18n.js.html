  9 /**
 10  * This plugin handles i18n! prefixed modules. It does the following:
 11  *
 12  * 1) A regular module can have a dependency on an i18n bundle, but the regular
 13  * module does not want to specify what locale to load. So it just specifies
 14  * the top-level bundle, like "i18n!nls/colors".
 15  *
 16  * This plugin will load the i18n bundle at nls/colors, see that it is a root/master
 17  * bundle since it does not have a locale in its name. It will then try to find
 18  * the best match locale available in that master bundle, then request all the
 19  * locale pieces for that best match locale. For instance, if the locale is "en-us",
 20  * then the plugin will ask for the "en-us", "en" and "root" bundles to be loaded
 21  * (but only if they are specified on the master bundle).
 22  *
 23  * Once all the bundles for the locale pieces load, then it mixes in all those
 24  * locale pieces into each other, then finally sets the context.defined value
 25  * for the nls/colors bundle to be that mixed in locale.
 26  *
 27  * 2) A regular module specifies a specific locale to load. For instance,
 28  * i18n!nls/fr-fr/colors. In this case, the plugin needs to load the master bundle
 29  * first, at nls/colors, then figure out what the best match locale is for fr-fr,
 30  * since maybe only fr or just root is defined for that locale. Once that best
 31  * fit is found, all of its locale pieces need to have their bundles loaded.
 32  *
 33  * Once all the bundles for the locale pieces load, then it mixes in all those
 34  * locale pieces into each other, then finally sets the context.defined value
 35  * for the nls/fr-fr/colors bundle to be that mixed in locale.
 36  */
 37 (function () {
 38     //regexp for reconstructing the master bundle name from parts of the regexp match
 39     //nlsRegExp.exec("foo/bar/baz/nls/en-ca/foo") gives:
 40     //["foo/bar/baz/nls/en-ca/foo", "foo/bar/baz/nls/", "/", "/", "en-ca", "foo"]
 41     //nlsRegExp.exec("foo/bar/baz/nls/foo") gives:
 42     //["foo/bar/baz/nls/foo", "foo/bar/baz/nls/", "/", "/", "foo", ""]
 43     //so, if match[5] is blank, it means this is the top bundle definition.
 44     var nlsRegExp = /(^.*(^|\/)nls(\/|$))([^\/]*)\/?([^\/]*)/;
 45 
 46     //Helper function to avoid repeating code. Lots of arguments in the
 47     //desire to stay functional and support RequireJS contexts without having
 48     //to know about the RequireJS contexts.
 49     function addPart(locale, master, needed, toLoad, prefix, suffix) {
 50         if (master[locale]) {
 51             needed.push(locale);
 52             if (master[locale] === true || master[locale] === 1) {
 53                 toLoad.push(prefix + locale + '/' + suffix);
 54             }
 55         }
 56     }
 57 
 58     function addIfExists(req, locale, toLoad, prefix, suffix) {
 59         var fullName = prefix + locale + '/' + suffix;
 60         if (require._fileExists(req.nameToUrl(fullName, null))) {
 61             toLoad.push(fullName);
 62         }
 63     }
 64 
 65     define({
 66         version: '0.24.0',
 67         /**
 68          * Called when a dependency needs to be loaded.
 69          */
 70         load: function (name, req, onLoad, config) {
 71             config = config || {};
 72 
 73             var masterName,
 74                 match = nlsRegExp.exec(name),
 75                 prefix = match[1],
 76                 locale = match[4],
 77                 suffix = match[5],
 78                 parts = locale.split("-"),
 79                 toLoad = [],
 80                 value = {},
 81                 i, part, current = "";
 82 
 83             //If match[5] is blank, it means this is the top bundle definition,
 84             //so it does not have to be handled. Locale-specific requests
 85             //will have a match[4] value but no match[5]
 86             if (match[5]) {
 87                 //locale-specific bundle
 88                 prefix = match[1];
 89                 masterName = prefix + suffix;
 90             } else {
 91                 //Top-level bundle.
 92                 masterName = name;
 93                 suffix = match[4];
 94                 locale = config.locale || (config.locale =
 95                         typeof navigator === "undefined" ? "root" :
 96                         (navigator.language ||
 97                          navigator.userLanguage || "root").toLowerCase());
 98                 parts = locale.split("-");
 99             }
100 
101             if (config.isBuild) {
102                 //Check for existence of all locale possible files and
103                 //require them if exist.
104                 toLoad.push(masterName);
105                 addIfExists(req, "root", toLoad, prefix, suffix);
106                 for (i = 0; (part = parts[i]); i++) {
107                     current += (current ? "-" : "") + part;
108                     addIfExists(req, current, toLoad, prefix, suffix);
109                 }
110                 req(toLoad);
111                 onLoad();
112             } else {
113                 //First, fetch the master bundle, it knows what locales are available.
114                 req([masterName], function (master) {
115                     //Figure out the best fit
116                     var needed = [];
117 
118                     //Always allow for root, then do the rest of the locale parts.
119                     addPart("root", master, needed, toLoad, prefix, suffix);
120                     for (i = 0; (part = parts[i]); i++) {
121                         current += (current ? "-" : "") + part;
122                         addPart(current, master, needed, toLoad, prefix, suffix);
123                     }
124 
125                     //Load all the parts missing.
126                     req(toLoad, function () {
127                         var i, partBundle;
128                         for (i = needed.length - 1; i > -1 && (part = needed[i]); i--) {
129                             partBundle = master[part];
130                             if (partBundle === true || partBundle === 1) {
131                                 partBundle = req(prefix + part + '/' + suffix);
132                             }
133                             require.mixin(value, partBundle);
134                         }
135 						
136 						// MODIFICATION FROM ALOHA START: add a t() function
137 						value.t = function(key) {
138 							if (this[key]) {
139 								return this[key];
140 							} else {
141 								return key;
142 							}
143 						}
144 						// END OF ALOHA MODIFICATION
145 
146 						//All done, notify the loader.
147                         onLoad(value);
148                     });
149                 });
150             }
151         }
152     });
153 }());
