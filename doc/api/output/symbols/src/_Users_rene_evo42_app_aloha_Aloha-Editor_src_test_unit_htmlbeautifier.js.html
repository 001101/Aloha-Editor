 14   Usage:
 15     style_html(html_source);
 16 
 17     style_html(html_source, options);
 18 
 19   The options are:
 20     indent_size (default 4)          — indentation size,
 21     indent_char (default space)      — character to indent with,
 22     max_char (default 70)            -  maximum amount of characters per line,
 23     brace_style (default "collapse") - "collapse" | "expand" | "end-expand"
 24             put braces on the same line as control statements (default), or put braces on own line (Allman / ANSI style), or just put end braces on own line.
 25     unformatted (default ['a'])      - list of tags, that shouldn't be reformatted
 26 
 27     e.g.
 28 
 29     style_html(html_source, {
 30       'indent_size': 2,
 31       'indent_char': ' ',
 32       'max_char': 78,
 33       'brace_style': 'expand',
 34       'unformatted': ['a', 'sub', 'sup', 'b', 'i', 'u']
 35     });

 38 function style_html(html_source, options) {
 39 //Wrapper function to invoke all the necessary constructors and deal with the output.
 40 
 41   var multi_parser,
 42       indent_size,
 43       indent_character,
 44       max_char,
 45       brace_style;
 46 
 47   options = options || {};
 48   indent_size = options.indent_size || 4;
 49   indent_character = options.indent_char || ' ';
 50   brace_style = options.brace_style || 'collapse';
 51   max_char = options.max_char || '70';
 52   unformatted = options.unformatted || ['a'];
 53 
 54   function Parser() {
 55 
 56     this.pos = 0; //Parser position
 57     this.token = '';
 58     this.current_mode = 'CONTENT'; //reflects the current Parser mode: TAG/CONTENT
 59     this.tags = { //An object to hold tags, their position, and their parent-tags, initiated with default values
 60       parent: 'parent1',
 61       parentcount: 1,
 62       parent1: ''
 63     };
 64     this.tag_type = '';
 65     this.token_text = this.last_token = this.last_text = this.token_type = '';
 66 
 67     this.Utils = { //Uilities made available to the various functions
 68       whitespace: "\n\r\t ".split(''),
 69       single_token: 'br,input,link,meta,!doctype,basefont,base,area,hr,wbr,param,img,isindex,?xml,embed'.split(','), //all the single tags for HTML
 70       extra_liners: 'head,body,/html'.split(','), //for tags that need a line of whitespace before them
 71       in_array: function (what, arr) {
 72         for (var i=0; i<arr.length; i++) {
 73           if (what === arr[i]) {
 74             return true;
 75           }
 76         }
 77         return false;
 78       }
 79     }
 80 
 81     this.get_content = function () { //function to capture regular content between tags
 82 
 83       var input_char = '';
 84       var content = [];
 85       var space = false; //if a space is needed
 86       while (this.input.charAt(this.pos) !== '<') {
 87         if (this.pos >= this.input.length) {
 88           return content.length?content.join(''):['', 'TK_EOF'];
 89         }
 90 
 91         input_char = this.input.charAt(this.pos);
 92         this.pos++;
 93         this.line_char_count++;
 94 
 95         if (this.Utils.in_array(input_char, this.Utils.whitespace)) {
 96           if (content.length) {
 97             space = true;
 98           }
 99           this.line_char_count--;
100           continue; //don't want to insert unnecessary space
101         }
102         else if (space) {
103           if (this.line_char_count >= this.max_char) { //insert a line when the max_char is reached
104             content.push('\n');
105             for (var i=0; i<this.indent_level; i++) {
106               content.push(this.indent_string);
107             }
108             this.line_char_count = 0;
109           }
110           else{
111             content.push(' ');
112             this.line_char_count++;
113           }
114           space = false;
115         }
116         content.push(input_char); //letter at-a-time (or string) inserted to an array
117       }
118       return content.length?content.join(''):'';
119     }
120 
121     this.get_script = function () { //get the full content of a script to pass to js_beautify
122 
123       var input_char = '';
124       var content = [];
125       var reg_match = new RegExp('\<\/script' + '\>', 'igm');
126       reg_match.lastIndex = this.pos;
127       var reg_array = reg_match.exec(this.input);
128       var end_script = reg_array?reg_array.index:this.input.length; //absolute end of script
129       while(this.pos < end_script) { //get everything in between the script tags
130         if (this.pos >= this.input.length) {
131           return content.length?content.join(''):['', 'TK_EOF'];
132         }
133 
134         input_char = this.input.charAt(this.pos);
135         this.pos++;
136 
137         content.push(input_char);
138       }
139       return content.length?content.join(''):''; //we might not have any content at all
140     }
141 
142     this.record_tag = function (tag){ //function to record a tag and its parent in this.tags Object
143       if (this.tags[tag + 'count']) { //check for the existence of this tag type
144         this.tags[tag + 'count']++;
145         this.tags[tag + this.tags[tag + 'count']] = this.indent_level; //and record the present indent level
146       }
147       else { //otherwise initialize this tag type
148         this.tags[tag + 'count'] = 1;
149         this.tags[tag + this.tags[tag + 'count']] = this.indent_level; //and record the present indent level
150       }
151       this.tags[tag + this.tags[tag + 'count'] + 'parent'] = this.tags.parent; //set the parent (i.e. in the case of a div this.tags.div1parent)
152       this.tags.parent = tag + this.tags[tag + 'count']; //and make this the current parent (i.e. in the case of a div 'div1')
153     }
154 
155     this.retrieve_tag = function (tag) { //function to retrieve the opening tag to the corresponding closer
156       if (this.tags[tag + 'count']) { //if the openener is not in the Object we ignore it
157         var temp_parent = this.tags.parent; //check to see if it's a closable tag.
158         while (temp_parent) { //till we reach '' (the initial value);
159           if (tag + this.tags[tag + 'count'] === temp_parent) { //if this is it use it
160             break;
161           }
162           temp_parent = this.tags[temp_parent + 'parent']; //otherwise keep on climbing up the DOM Tree
163         }
164         if (temp_parent) { //if we caught something
165           this.indent_level = this.tags[tag + this.tags[tag + 'count']]; //set the indent_level accordingly
166           this.tags.parent = this.tags[temp_parent + 'parent']; //and set the current parent
167         }
168         delete this.tags[tag + this.tags[tag + 'count'] + 'parent']; //delete the closed tags parent reference...
169         delete this.tags[tag + this.tags[tag + 'count']]; //...and the tag itself
170         if (this.tags[tag + 'count'] == 1) {
171           delete this.tags[tag + 'count'];
172         }
173         else {
174           this.tags[tag + 'count']--;
175         }
176       }
177     }
178 
179     this.get_tag = function () { //function to get a full tag and parse its type
180       var input_char = '';
181       var content = [];
182       var space = false;
183 
184       do {
185         if (this.pos >= this.input.length) {
186           return content.length?content.join(''):['', 'TK_EOF'];
187         }
188 
189         input_char = this.input.charAt(this.pos);
190         this.pos++;
191         this.line_char_count++;
192 
193         if (this.Utils.in_array(input_char, this.Utils.whitespace)) { //don't want to insert unnecessary space
194           space = true;
195           this.line_char_count--;
196           continue;
197         }
198 
199         if (input_char === "'" || input_char === '"') {
200           if (!content[1] || content[1] !== '!') { //if we're in a comment strings don't get treated specially
201             input_char += this.get_unformatted(input_char);
202             space = true;
203           }
204         }
205 
206         if (input_char === '=') { //no space before =
207           space = false;
208         }
209 
210         if (content.length && content[content.length-1] !== '=' && input_char !== '>'
211             && space) { //no space after = or before >
212           if (this.line_char_count >= this.max_char) {
213             this.print_newline(false, content);
214             this.line_char_count = 0;
215           }
216           else {
217             content.push(' ');
218             this.line_char_count++;
219           }
220           space = false;
221         }
222         content.push(input_char); //inserts character at-a-time (or string)
223       } while (input_char !== '>');
224 
225       var tag_complete = content.join('');
226       var tag_index;
227       if (tag_complete.indexOf(' ') != -1) { //if there's whitespace, thats where the tag name ends
228         tag_index = tag_complete.indexOf(' ');
229       }
230       else { //otherwise go with the tag ending
231         tag_index = tag_complete.indexOf('>');
232       }
233       var tag_check = tag_complete.substring(1, tag_index).toLowerCase();
234       if (tag_complete.charAt(tag_complete.length-2) === '/' ||
235           this.Utils.in_array(tag_check, this.Utils.single_token)) { //if this tag name is a single tag type (either in the list or has a closing /)
236         this.tag_type = 'SINGLE';
237       }
238       else if (tag_check === 'script') { //for later script handling
239         this.record_tag(tag_check);
240         this.tag_type = 'SCRIPT';
241       }
242       else if (tag_check === 'style') { //for future style handling (for now it justs uses get_content)
243         this.record_tag(tag_check);
244         this.tag_type = 'STYLE';
245       }
246       else if (this.Utils.in_array(tag_check, unformatted)) { // do not reformat the "unformatted" tags
247         var comment = this.get_unformatted('</'+tag_check+'>', tag_complete); //...delegate to get_unformatted function
248         content.push(comment);
249         this.tag_type = 'SINGLE';
250       }
251       else if (tag_check.charAt(0) === '!') { //peek for <!-- comment
252         if (tag_check.indexOf('[if') != -1) { //peek for <!--[if conditional comment
253           if (tag_complete.indexOf('!IE') != -1) { //this type needs a closing --> so...
254             var comment = this.get_unformatted('-->', tag_complete); //...delegate to get_unformatted
255             content.push(comment);
256           }
257           this.tag_type = 'START';
258         }
259         else if (tag_check.indexOf('[endif') != -1) {//peek for <!--[endif end conditional comment
260           this.tag_type = 'END';
261           this.unindent();
262         }
263         else if (tag_check.indexOf('[cdata[') != -1) { //if it's a <[cdata[ comment...
264           var comment = this.get_unformatted(']]>', tag_complete); //...delegate to get_unformatted function
265           content.push(comment);
266           this.tag_type = 'SINGLE'; //<![CDATA[ comments are treated like single tags
267         }
268         else {
269           var comment = this.get_unformatted('-->', tag_complete);
270           content.push(comment);
271           this.tag_type = 'SINGLE';
272         }
273       }
274       else {
275         if (tag_check.charAt(0) === '/') { //this tag is a double tag so check for tag-ending
276           this.retrieve_tag(tag_check.substring(1)); //remove it and all ancestors
277           this.tag_type = 'END';
278         }
279         else { //otherwise it's a start-tag
280           this.record_tag(tag_check); //push it on the tag stack
281           this.tag_type = 'START';
282         }
283         if (this.Utils.in_array(tag_check, this.Utils.extra_liners)) { //check if this double needs an extra line
284           this.print_newline(true, this.output);
285         }
286       }
287       return content.join(''); //returns fully formatted tag
288     }
289 
290     this.get_unformatted = function (delimiter, orig_tag) { //function to return unformatted content in its entirety
291 
292       if (orig_tag && orig_tag.indexOf(delimiter) != -1) {
293         return '';
294       }
295       var input_char = '';
296       var content = '';
297       var space = true;
298       do {
299 
300         if (this.pos >= this.input.length) {
301           return content;
302         }
303 
304         input_char = this.input.charAt(this.pos);
305         this.pos++
306 
307         if (this.Utils.in_array(input_char, this.Utils.whitespace)) {
308           if (!space) {
309             this.line_char_count--;
310             continue;
311           }
312           if (input_char === '\n' || input_char === '\r') {
313             content += '\n';
314             /*  Don't change tab indention for unformatted blocks.  If using code for html editing, this will greatly affect <pre> tags if they are specified in the 'unformatted array'
315             for (var i=0; i<this.indent_level; i++) {
316               content += this.indent_string;
317             }
318             space = false; //...and make sure other indentation is erased
319             */
320             this.line_char_count = 0;
321             continue;
322           }
323         }
324         content += input_char;
325         this.line_char_count++;
326         space = true;
327 
328 
329       } while (content.indexOf(delimiter) == -1);
330       return content;
331     }
332 
333     this.get_token = function () { //initial handler for token-retrieval
334       var token;
335 
336       if (this.last_token === 'TK_TAG_SCRIPT') { //check if we need to format javascript
337         var temp_token = this.get_script();
338         if (typeof temp_token !== 'string') {
339           return temp_token;
340         }
341         token = js_beautify(temp_token.replace(/^[\r\n]+/, ''), {
342           'indent_size': this.indent_size,
343           'indent_char': this.indent_character,
344           'brace_style': this.brace_style
345         }); //call the JS Beautifier
346         return [token, 'TK_CONTENT'];
347       }
348       if (this.current_mode === 'CONTENT') {
349         token = this.get_content();
350         if (typeof token !== 'string') {
351           return token;
352         }
353         else {
354           return [token, 'TK_CONTENT'];
355         }
356       }
357 
358       if (this.current_mode === 'TAG') {
359         token = this.get_tag();
360         if (typeof token !== 'string') {
361           return token;
362         }
363         else {
364           var tag_name_type = 'TK_TAG_' + this.tag_type;
365           return [token, tag_name_type];
366         }
367       }
368     }
369 
370     this.printer = function (js_source, indent_character, indent_size, max_char, brace_style) { //handles input/output and some other printing functions
371 
372       this.input = js_source || ''; //gets the input for the Parser
373       this.output = [];
374       this.indent_character = indent_character;
375       this.indent_string = '';
376       this.indent_size = indent_size;
377       this.brace_style = brace_style;
378       this.indent_level = 0;
379       this.max_char = max_char;
380       this.line_char_count = 0; //count to see if max_char was exceeded
381 
382       for (var i=0; i<this.indent_size; i++) {
383         this.indent_string += this.indent_character;
384       }
385 
386       this.print_newline = function (ignore, arr) {
387         this.line_char_count = 0;
388         if (!arr || !arr.length) {
389           return;
390         }
391         if (!ignore) { //we might want the extra line
392           while (this.Utils.in_array(arr[arr.length-1], this.Utils.whitespace)) {
393             arr.pop();
394           }
395         }
396         arr.push('\n');
397         for (var i=0; i<this.indent_level; i++) {
398           arr.push(this.indent_string);
399         }
400       }
401 
402       this.print_token = function (text) {
403         this.output.push(text);
404       }
405 
406       this.indent = function () {
407         this.indent_level++;
408       }
409 
410       this.unindent = function () {
411         if (this.indent_level > 0) {
412           this.indent_level--;
413         }
414       }
415     }
416     return this;
417   }
418 
419   /*_____________________--------------------_____________________*/
420 
421   multi_parser = new Parser(); //wrapping functions Parser
422   multi_parser.printer(html_source, indent_character, indent_size, max_char, brace_style); //initialize starting values
423 
424   while (true) {
425       var t = multi_parser.get_token();
426       multi_parser.token_text = t[0];
427       multi_parser.token_type = t[1];
428 
429     if (multi_parser.token_type === 'TK_EOF') {
430       break;
431     }
432 
433     switch (multi_parser.token_type) {
434       case 'TK_TAG_START':
435       case 'TK_TAG_STYLE':
436         multi_parser.print_newline(false, multi_parser.output);
437         multi_parser.print_token(multi_parser.token_text);
438         multi_parser.indent();
439         multi_parser.current_mode = 'CONTENT';
440         break;
441       case 'TK_TAG_SCRIPT':
442         multi_parser.print_newline(false, multi_parser.output);
443         multi_parser.print_token(multi_parser.token_text);
444         multi_parser.current_mode = 'CONTENT';
445         break;
446       case 'TK_TAG_END':
447         multi_parser.print_newline(true, multi_parser.output);
448         multi_parser.print_token(multi_parser.token_text);
449         multi_parser.current_mode = 'CONTENT';
450         break;
451       case 'TK_TAG_SINGLE':
452         multi_parser.print_newline(false, multi_parser.output);
453         multi_parser.print_token(multi_parser.token_text);
454         multi_parser.current_mode = 'CONTENT';
455         break;
456       case 'TK_CONTENT':
457         if (multi_parser.token_text !== '') {
458           multi_parser.print_newline(false, multi_parser.output);
459           multi_parser.print_token(multi_parser.token_text);
460         }
461         multi_parser.current_mode = 'TAG';
462         break;
463     }
464     multi_parser.last_token = multi_parser.token_type;
465     multi_parser.last_text = multi_parser.token_text;
466   }
467   return multi_parser.output.join('');
468 }
