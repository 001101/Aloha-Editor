 40 	 * Initialize the plugin and set initialize flag on true

 42 	Aloha.LinkChecker.init = function () {
 43 
 44 		this.proxyUrl = null;
 45 
 46 		if ( typeof Aloha.LinkChecker.settings.proxyUrl !== 'undefined' ) {
 47 			this.proxyUrl = Aloha.LinkChecker.settings.proxyUrl;
 48 		}
 49 
 50 		// initialize the timer
 51 		this.timer = {};
 52 
 53 		// initialize the running requests
 54 		this.xhr = {};
 55 
 56 		// remember reference to this class for callback
 57 		var that = this;
 58 
 59 		// mark active Editable with a css class
 60 		Aloha.EventRegistry.subscribe(
 61 				Aloha,
 62 				"editableActivated",
 63 				function (jEvent, aEvent) {
 64 					// find all link tags
 65 					aEvent.editable.obj.find('a').each(function() {
 66 						that.checkLink(this, jQuery(this).attr('href'), 0);
 67 					});
 68 				}
 69 		);
 70 
 71 		// remove active Editable ccs class
 72 		Aloha.EventRegistry.subscribe(
 73 				Aloha,
 74 				"editableDeactivated",
 75 				function (jEvent, aEvent) {
 76 					// remove link marks
 77 					that.makeClean(aEvent.editable.obj);
 78 				}
 79 		);
 80 
 81 		// remove active Editable ccs class
 82 		Aloha.EventRegistry.subscribe(
 83 				Aloha,
 84 				"hrefChanged",
 85 				function (jEvent, aEvent) {
 86 					that.checkLink(aEvent.obj, 'hrefChanged');
 87 				}
 88 		);
 89 
 90  91 	};

 93 	Aloha.LinkChecker.checkLink = function (obj, scope, delay, timeout) {
 94 		var that = this,
 95 			url, cleanUrl;
 96 
 97 		// extract url from link object
 98 		url = jQuery(obj).attr('href');
 99 		cleanUrl = url;
100 
101 		// i probably an internal link
102 		if ( typeof url == 'string' && !/^http/.test( url.toLowerCase() ) ) {
103 			this.makeCleanLink(obj);
104 			return;
105 		}
106 
107 		if ( this.proxyUrl ) {
108 			url = this.proxyUrl + url;
109 		}
110 
111 		// abort already running ajax requests for the scope
112 		if (this.xhr[scope]) {
113 			this.xhr[scope].abort();
114 			this.xhr[scope] = undefined;
115 		}
116 
117 		this.timer[scope] = this.urlExists(
118 			url,
119 			// success
120 			function(xhr) {
121 				that.makeCleanLink(obj);
122 			},
123 			//failure
124 			function(xhr) {
125 				var e, o;
126 
127 				if ( obj ) {
128 					if ( jQuery.inArray(xhr.status, that.errorCodes) >= 0 ) {
129 						e = xhr.status;
130 					} else {
131 						e = '0';
132 					}
133 					o = jQuery(obj);
134 					// when the link has a title and was not yet marked as being invalid, we store the title in 'data-title'
135 					if ( o.attr('title') && !o.attr('data-invalid') ) {
136 						o.attr('data-title', o.attr('title'));
137 					}
138 					// now we mark the link as being invalid
139 					o.attr('data-invalid', 'true');
140 					// and we set an error message to the title
141 					o.attr('title', cleanUrl+'. '+that.i18n('error.'+e));
142 					// set the link class
143 					if ( jQuery.inArray(xhr.status, that.warningCodes) >= 0 ) {
144 						o.addClass('aloha-link-warn');
145 					} else {
146 						o.addClass('aloha-link-error');
147 					}
148 				}
149 			},
150 			scope,
151 			timeout,
152 			delay
153 		);
154 	};

156 	Aloha.LinkChecker.urlExists = function (url, successFunc, failureFunc, scope, timeout, delay) {
157 		var that = this, newTimer;
158 
159 		// abort timer for that request
160 		clearTimeout(this.timer[scope]);
161 
162 		delay = (delay != null && typeof delay !== 'undefined' ) ? delay : 700;
163 
164 		// start timer for delayed request
165 		newTimer = setTimeout( function() {
166 
167 			// start request
168 			that.xhr[scope] = jQuery.ajax({
169 				url: url,
170 				timeout: timeout ? 10000 : timeout,
171 				type: 'HEAD',
172 				complete: function(xhr) {
173 					// abort timer for that request
174 					clearTimeout(newTimer);
175 					try {
176 						// if response HTTP status 200 link is ok
177 						// this implementation does NOT cover redirects!
178 						if (xhr.status < 400) {
179 							successFunc.call(this, xhr);
180 						} else {
181 							failureFunc.call(this, xhr);
182 						}
183 					} catch(e) {
184 						failureFunc.call(this, {'status':0});
185 					}
186 				}
187 			});
188 
189 		}, delay);
190 
191 		return newTimer;
192 	};

194 	Aloha.LinkChecker.makeCleanLink = function (obj) {
195 		if ( obj ) {
196 			var o = jQuery(obj);
197 			// restore the original title (if one existed)
198 			if ( o.attr('data-title') ) {
199 				o.attr('title', o.attr('data-title'));
200 			} else {
201 				// otherwise remove the title
202 				o.removeAttr('title');
203 			}
204 			// remove the temporary data
205 			o.removeAttr('data-title');
206 			o.removeAttr('data-invalid');
207 			// remove the classes
208 			o.removeClass('aloha-link-error');
209 			o.removeClass('aloha-link-warn');
210 		}
211 	};
212 
213 	Aloha.LinkChecker.makeClean = function (editable) {
214 		var that = this;
215 		// find all link tags
216 		editable.find('a').each(function() {
217 			that.makeCleanLink(this);
218 		});
219 	};

221 	Aloha.LinkChecker.urlencode = function (str) {
222 		// URL-encodes string
223 		//
224 		// version: 1008.1718
225 		// discuss at: http://phpjs.org/functions/urlencode
226 		// +   original by: Philip Peterson
227 		// +   improved by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
228 		// +	  input by: AJ
229 		// +   improved by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
230 		// +   improved by: Brett Zamir (http://brett-zamir.me)
231 		// +   bugfixed by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
232 		// +	  input by: travc
233 		// +	  input by: Brett Zamir (http://brett-zamir.me)
234 		// +   bugfixed by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
235 		// +   improved by: Lars Fischer
236 		// +	  input by: Ratheous
237 		// +	  reimplemented by: Brett Zamir (http://brett-zamir.me)
238 		// +   bugfixed by: Joris
239 		// +	  reimplemented by: Brett Zamir (http://brett-zamir.me)
240 		// %		  note 1: This reflects PHP 5.3/6.0+ behavior
241 		// %		note 2: Please be aware that this function expects to encode into UTF-8 encoded strings, as found on
242 		// %		note 2: pages served as UTF-8
243 		// *	 example 1: urlencode('Kevin van Zonneveld!');
244 		// *	 returns 1: 'Kevin+van+Zonneveld%21'
245 		// *	 example 2: urlencode('http://kevin.vanzonneveld.net/');
246 		// *	 returns 2: 'http%3A%2F%2Fkevin.vanzonneveld.net%2F'
247 		// *	 example 3: urlencode('http://www.google.nl/search?q=php.js&ie=utf-8&oe=utf-8&aq=t&rls=com.ubuntu:en-US:unofficial&client=firefox-a');
248 		// *	 returns 3: 'http%3A%2F%2Fwww.google.nl%2Fsearch%3Fq%3Dphp.js%26ie%3Dutf-8%26oe%3Dutf-8%26aq%3Dt%26rls%3Dcom.ubuntu%3Aen-US%3Aunofficial%26client%3Dfirefox-a'
249 		str = (str+'').toString();
250 
251 		// Tilde should be allowed unescaped in future versions of PHP (as reflected below), but if you want to reflect current
252 		// PHP behavior, you would need to add ".replace(/~/g, '%7E');" to the following.
253 		return encodeURIComponent(str).replace(/!/g, '%21').replace(/'/g, '%27').replace(/\(/g, '%28').
254 																		replace(/\)/g, '%29').replace(/\*/g, '%2A').replace(/%20/g, '+');
255 	};
