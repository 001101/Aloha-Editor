  8 define( [
  9 	'aloha',
 10 	'aloha/jquery',
 11 	'aloha/plugin',
 12 	'aloha/floatingmenu',
 13 	'i18n!wai-lang/nls/i18n',
 14 	'i18n!aloha/nls/i18n',
 15 	'wai-lang/languages',
 16 	'css!wai-lang/css/wai-lang.css'
 17 ], function( Aloha, jQuery, Plugin, FloatingMenu, i18n, i18nCore ) {
 18 	'use strict';
 19 
 20 	var WAI_LANG_CLASS = 'aloha-wai-lang',
 21 	    GENTICS = window.GENTICS,
 22 	    addMarkupToSelectionButton,
 23 	    langField;
 24 
 25 	return Plugin.create( 'wai-lang', {
 26 
 27 		/**
 28 		 * Configure the available languages (i18n) for this plugin
 29 		 */
 30 		languages: [ 'en', 'de' ],
 31 
 32 		/**
 33 		 * Default configuration allows spans everywhere
 34 		 */
 35 		config: [ 'span' ],
 36 
 37 		/**
 38 		 * the defined object types to be used for this instance
 39 		 */
 40 		objectTypeFilter: [ 'language' ],
 41 
 42 		/**
 43 		 * Initialize the plugin:
 44 		 * Initializes UI components, and binds their event listeners.
 45 		 */
 46 		init: function() {
 47 			if ( this.settings.objectTypeFilter ) {
 48 				this.objectTypeFilter = this.settings.objectTypeFilter;
 49 			}
 50 
 51 			this.createButtons();
 52 			this.subscribeEvents();
 53 			this.bindInteractions();
 54 		},
 55 
 56 		/**
 57 		 * Subscribe for events
 58 		 */
 59 		subscribeEvents: function() {
 60 			var that = this;
 61 
 62 			// add the event handler for selection change
 63 			Aloha.bind( 'aloha-selection-changed', function( event, rangeObject ) {
 64 				var config,
 65 				    foundMarkup;
 66 
 67 				if ( Aloha.activeEditable ) {
 68 					// show/hide the button according to the configuration
 69 					config = that.getEditableConfig( Aloha.activeEditable.obj );
 70 					if ( jQuery.inArray( 'span', config ) !== -1 ) {
 71 						addMarkupToSelectionButton.setPressed( false );
 72 					} else {
 73 						addMarkupToSelectionButton.setPressed( true );
 74 
 75 						// leave if a is not allowed
 76 						return;
 77 					}
 78 
 79 					foundMarkup = that.findLangMarkup( rangeObject );
 80 					if ( foundMarkup ) {
 81 						addMarkupToSelectionButton.setPressed( true );
 82 						FloatingMenu.setScope( 'wai-lang' );
 83 						langField.setTargetObject( foundMarkup, 'lang' );
 84 					} else {
 85 						langField.setTargetObject( null );
 86 					}
 87 				}
 88 			} );
 89 		},
 90 
 91 		/**
 92 		 * Initialize the buttons:
 93 		 * Places the Wai-Lang UI buttons into the floating menu.
 94 		 */
 95 		createButtons: function() {
 96 			var that = this;
 97 
 98 			// Button for adding a language markup to the current selection
 99 			addMarkupToSelectionButton = new Aloha.ui.Button( {
100 				'name'      : 'wailang',
101 				'iconClass' : 'aloha-button aloha-button-wai-lang',
102 				'size'      : 'small',
103 				'onclick'   : function() {
104 					that.addRemoveMarkupToSelection();
105 				},
106 				'tooltip'   : i18n.t( 'button.add-wai-lang.tooltip' ),
107 				'toggle'    : true
108 			} );
109 
110 			FloatingMenu.addButton(
111 				'Aloha.continuoustext',
112 				addMarkupToSelectionButton,
113 				i18nCore.t( 'floatingmenu.tab.format' ),
114 				1
115 			);
116 
117 			// Add the new scope for the wai languages plugin tab
118 			FloatingMenu.createScope( 'wai-lang', 'Aloha.continuoustext' );
119 
120 			langField = new Aloha.ui.AttributeField( {
121 				'name'       : 'wailangfield',
122 				'width'      : 320,
123 				'valueField' : 'id',
124 				'minChars'   : 1
125 			} );
126 
127 			langField.setTemplate(
128 				'<div class="img-item">' +
129 					'<img class="typeahead-image" src="{url}" />' +
130 					'<div class="label-item">{name}</div>' +
131 				'</div>'
132 			);
133 
134 			langField.setObjectTypeFilter( this.objectTypeFilter );
135 
136 			// add the input field for links
137 			FloatingMenu.addButton(
138 				'wai-lang',
139 				langField,
140 				i18n.t( 'floatingmenu.tab.wai-lang' ),
141 				1
142 			);
143 
144 			var removeButton = new Aloha.ui.Button( {
145 				'name'      : 'removewailang',
146 				'iconClass' : 'aloha-button aloha-button-wai-lang-remove',
147 				'size'      : 'small',
148 				'onclick'   : function() {
149 					that.removeLangMarkup();
150 				},
151 				'tooltip'   : i18n.t( 'button.add-wai-lang-remove.tooltip' ),
152 				'toggle'    : false
153 			} );
154 
155 			FloatingMenu.addButton(
156 				'wai-lang',
157 				removeButton,
158 				i18n.t( 'floatingmenu.tab.wai-lang' ),
159 				1
160 			);
161 		},
162 
163 		/**
164 		 * Given a range object, finds and returns a Wai-Lang wrapper DOM
165 		 * element within the range.
166 		 *
167 		 * @param {GENTICS.Utils.RangeObject} range
168 		 * @return {?DOMObject} the dom object found, or false if nothing found
169 		 */
170 		findLangMarkup: function( range ) {
171 			range = range || Aloha.Selection.getRangeObject();
172 			if ( Aloha.activeEditable ) {
173 				return range.findMarkup( function() {
174 					return jQuery( this ).hasClass( WAI_LANG_CLASS )
175 						|| jQuery( this ).is( '[lang]' );
176 			    }, Aloha.activeEditable.obj );
177 			}
178 			return null;
179 		},
180 
181 		/**
182 		 * Check whether the range is within a span that contains a lang
183 		 * attribute.
184 		 *
185 		 * @param {GENTICS.Utils.RangeObject} range range where to insert the
186 		 *									  object (at start or end)
187 		 * @return {?DOMObject} the dom object found, or false if nothing found
188 		 * @hide
189 		 */
190 		findLanguageMarkup: function( range ) {
191 			range = range || Aloha.Selection.getRangeObject();
192 			if ( Aloha.activeEditable ) {
193 				return range.findMarkup( function() {
194 					return this.nodeName === 'SPAN';
195 				}, Aloha.activeEditable.obj );
196 			}
197 			return null;
198 		},
199 
200 		/**
201 		 *
202 		 */
203 		removeLangMarkup: function() {
204 			var range = Aloha.Selection.getRangeObject(),
205 			    foundMarkup = this.findLangMarkup( range );
206 
207 		    if ( foundMarkup ) {
208 		        // remove the abbr
209 		        GENTICS.Utils.Dom.removeFromDOM( foundMarkup, range, true );
210 
211 		        // select the (possibly modified) range
212 		        range.select();
213 				FloatingMenu.setScope( 'Aloha.continousText' );
214 				langField.setTargetObject( null );
215 				FloatingMenu.doLayout();
216 		    }
217 		},
218 
219 		/**
220 		 * Parse a all editables for elements that have a lang attribute and
221 		 * bind an onclick event
222 		 */
223 		bindInteractions: function() {
224 			var that = this;
225 
226 			// on blur check if lang is empty, if so remove the <a> tag
227 			langField.addListener( 'blur', function( obj, event ) {
228 				// @todo check for a valid value -- now it's also possible to insert abcd; but that's not valid
229 				if ( !this.getValue() ) {
230 					that.removeMarkup();
231 				}
232 			} );
233 
234 			Aloha.ready( function() {
235 				that.handleExistingSpans();
236 			} );
237 		},
238 
239 		/**
240 		 * Find all existing spans and register hotkey hotkeys and make
241 		 * annotations of languages visible.
242 		 */
243 		handleExistingSpans: function() {
244 			var that = this;
245 
246 			// Add the Link shortcut to all editables
247 			jQuery.each( Aloha.editables, function( key, editable ) {
248 				// Hotkey for adding new language annotations: CTRL+I
249 				editable.obj.keydown( that.handleKeyDown );
250 			} );
251 
252 			jQuery.each( Aloha.editables, function( key, editable ) {
253 				// Find all spans with lang attributes and add some css and
254 				// event handlers
255 				editable.obj.find( 'span[lang]' ).each( function( i ) {
256 					that.makeVisible( this );
257 				} );
258 			} );
259 		},
260 
261 		/**
262 		 * @param {Event} e
263 		 * @return {?Boolean}
264 		 */
265 		handleKeyDown: function( e ) {
266 			if ( e.metaKey && e.which === 73 ) {
267 				if ( this.findLangMarkup() ) {
268 					FloatingMenu.activateTabOfButton( 'wailangfield' );
269 					langField.focus();
270 				} else {
271 					this.addMarkupToSelection();
272 				}
273 
274 				// Prevent from further handling.
275 				// on a MAC Safari, cursor would jump to location bar.
276 				// We have to use ESC and then META+I instead.
277 				return false;
278 			}
279 		},
280 
281 		/**
282 		 * Make the given element visible by adding some styles to it.
283 		 */
284 		makeVisible: function( element ) {
285 			// Make existing spans with language attribute visible
286 			// Flags can be added via the metaview plugin
287 			jQuery( element ).css(
288 				'background-image',
289 				'url(' + Aloha.getPluginUrl( 'wai-lang' ) + '/img/flags/' +
290 					jQuery( element ).attr( 'lang' ) + '.png)'
291 			);
292 			jQuery( element ).addClass( WAI_LANG_CLASS );
293 		},
294 
295 		/**
296 		 * Format the current selection or if collapsed the current word as
297 		 * element that should be annotated.
298 		 */
299 		formatLanguageSpan: function() {
300 			if ( Aloha.activeEditable ) {
301 				var range = Aloha.Selection.getRangeObject();
302 				if ( this.findLanguageMarkup( range ) ) {
303 					this.removeMarkup();
304 				} else {
305 					this.insertMarkup();
306 				}
307 			}
308 		},
309 
310 		/**
311 		 * Toggles markup around selection.
312 		 */
313 		addRemoveMarkupToSelection: function() {
314 			if ( addMarkupToSelectionButton.pressed ) {
315 				this.removeLangMarkup();
316 			} else {
317 				this.addMarkupToSelection( false );
318 			}
319 		},
320 
321 		/**
322 		 * Retrieves the current selected range, and wraps it with wai-lang
323 		 * content.
324 		 */
325 		addMarkupToSelection: function() {
326 			var range = Aloha.Selection.getRangeObject();
327 
328 			// Do not add markup to an area that already contains a markup
329 			if ( this.findLangMarkup( range ) ) {
330 				return;
331 			}
332 
333 			FloatingMenu.activateTabOfButton( 'wailangfield' );
334             FloatingMenu.setScope( 'wai-lang' );
335 
336 			if ( range.isCollapsed() ) {
337 				GENTICS.Utils.Dom.extendToWord( range );
338 			}
339 
340 			if ( !range.isCollapsed() ) {
341 				GENTICS.Utils.Dom.addMarkup( range,
342 					jQuery( '<span class="' + WAI_LANG_CLASS + '"></span>' ),
343 					false );
344 			}
345 
346 			range.select();
347 			// langField.focus();
348 		},
349 
350 		/**
351 		 * Remove an a tag.
352 		 */
353 		removeMarkup: function() {
354 			var foundMarkup = this.findLangMarkup(),
355 			    range;
356 
357 			if ( foundMarkup ) {
358 				range = Aloha.Selection.getRangeObject();
359 				GENTICS.Utils.Dom.removeFromDOM( foundMarkup, range, true );
360 
361 				// select the (possibly modified) range
362 				range.select();
363 			}
364 		},
365 
366 		/**
367 		 * Make the given jQuery object (representing an editable) clean for saving
368 		 * Find all elements with lang attributes and remove the attribute.
369 		 * @param {jQuery} obj jQuery object to make clean
370 		 */
371 		makeClean: function( obj ) {
372 			obj.find( 'span[lang]' ).each( function() {
373 				jQuery( this ).removeClass( WAI_LANG_CLASS );
374 			} );
375 		}
376 
377 	} );
378 } );
