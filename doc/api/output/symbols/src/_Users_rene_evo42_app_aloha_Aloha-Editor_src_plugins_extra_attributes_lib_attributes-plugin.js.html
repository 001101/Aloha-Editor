  7 define(
  8 ['aloha','aloha/plugin', 'aloha/floatingmenu', 'i18n!attributes/nls/i18n', 'i18n!aloha/nls/i18n', 'css!attributes/css/attributes.css'],
  9 function(Aloha, Plugin, FloatingMenu, i18n, i18nCore) {
 10 	"use strict";
 11 
 12 	var
 13 		jQuery = Aloha.jQuery,
 14 		$ = jQuery,
 15 		GENTICS = window.GENTICS,
 16 		Aloha = window.Aloha;
 17 	
 18     return Plugin.create('attributes', {
 19 		_constructor: function(){
 20 			this._super('attributes');
 21 		},
 22 		
 23 		// namespace prefix for this plugin
 24     // Pseudo-namespace prefix
 25 		ns : 'aloha-attributes',
 26 		uid  : 'attributes',
 27 		// namespaced classnames
 28 		nsClasses : {},
 29     
 30     
 31 		supplant : function(str, obj) {
 32 			return str.replace(/\{([a-z0-9\-\_]+)\}/ig, function (str, p1, offset, s) {
 33 				var replacement = obj[p1] || str;
 34 				return (typeof replacement == 'function') ? replacement() : replacement;
 35 			});
 36 		},
 37 		
 38 		/**
 39 		 * Wrapper to all the supplant method on a given string, taking the
 40 		 * nsClasses object as the associative array containing the replacement
 41 		 * pairs
 42 		 *
 43 		 * @param {String} str
 44 		 * @return {String}
 45 		 */
 46 		renderTemplate : function(str) {
 47 			return (typeof str === 'string') ? this.supplant(str, this.nsClasses) : str;
 48 		},
 49 		
 50 		/**
 51 		 * Generates a selector string with this component's namepsace prefixed the
 52 		 * each classname
 53 		 *
 54 		 * Usage:
 55 		 *		nsSel('header,', 'main,', 'foooter ul')
 56 		 *		will return
 57 		 *		".aloha-myplugin-header, .aloha-myplugin-main, .aloha-mypluzgin-footer ul"
 58 		 *
 59 		 * @return {String}
 60 		 */
 61 		nsSel : function() {
 62 			var strBldr = [], prx = this.ns;
 63 			$.each(arguments, function () { strBldr.push('.' + (this == '' ? prx : prx + '-' + this)); });
 64  65 			return strBldr.join(' ').trim();
 66 		},
 67 		
 68  69 		/**
 70 		 * Generates s string with this component's namepsace prefixed the each
 71 		 * classname
 72 		 *
 73 		 * Usage:
 74 		 *		nsClass('header', 'innerheaderdiv')
 75 		 *		will return
 76 		 *		"aloha-myplugin-header aloha-myplugin-innerheaderdiv"
 77 		 *
 78 		 * @return {String}
 79 		 */
 80 		nsClass : function (){
 81 			var strBldr = [], prx = this.ns;
 82 			$.each(arguments, function () { strBldr.push(this == '' ? prx : prx + '-' + this); });
 83 			return strBldr.join(' ').trim();
 84 		},
 85 		
 86 		config: ['true'],
 87 		
 88 		//activeOn: 'a,span,div,p,q,blockquote,h1,h2,h3,h4,h5,h6,em,i,b',
 89 		
 90 		activeOn : function(effective) {
 91 			if (typeof this.settings.disabled === 'boolean' && this.settings.disabled) {
 92 				return false;
 93 			}
 94 			if (typeof effective != 'undefined' && effective != null) {
 95 				return true;
 96 			}
 97 			return false;
 98 		},
 99 				
100 		/**
101 		 * Initialize the plugin
102 		 */
103 		init: function () {
104 			var that = this;
105 			this.nsClasses = {
106 				newattributename	: this.nsClass('newattributename'),
107 				newattributebutton	: this.nsClass('newattributebutton'),
108 				newattributewert	: this.nsClass('newattributewert'),
109 				container	: this.nsClass('container'),
110 				attribcontainer	: this.nsClass('attribcontainer'),
111 				newattribute	: this.nsClass('newattribute'),
112 				item	: this.nsClass('item'),
113 				element	: this.nsClass('element'),
114 				iteminput	: this.nsClass('iteminput')
115 			};
116 			if ( typeof this.settings.activeOn !== 'undefined') {
117 				this.activeOn = this.settings.activeOn;
118 			}
119 			Aloha.ready( function (ev, sidebars) { 
120 				that.initSidebar(Aloha.Sidebar.right); 
121 			});
122 		},
123 				
124 		getSidebarContent: function() {
125 			return this.renderTemplate(
126 					'<div class="{container}">\
127 						<h2 id="{element}">Element:</h2>\
128 						<h2>Vorhandene Attribute</h2>\
129 						<div class="{attribcontainer}">\
130 							attribcontainer\
131 						</div>\
132 						\
133 134 						<div class="{newattribute}">\
135 						<h2>Neues Attribut</h2>\
136 						<label for="{newattributename}">Name:</label><input type="text" id="{newattributename}"/>\
137 						<label for="{newattributewert}">Wert:</label><input type="text" id="{newattributewert}"/>\
138 						<button id="{newattributebutton}">Hinzufügen</button>\
139 						</div>\
140 						\
141 					</div>'
142 				);
143 		},
144 		
145 		updateSidebarWithAttributes: function() {
146 			var that = this;
147 			var el = this.effective[0];
148 			var $container = this.content.find(this.nsSel('attribcontainer'));
149 			$container.html('');
150 			for (var attr, i=0, attrs=el.attributes, l=attrs.length; i<l; i++){
151 				attr = attrs.item(i)
152 				var item = jQuery(this.renderTemplate('<div class="{item}"><label for="{iteminput}'+attr.nodeName+'">'+attr.nodeName+'</label><input id="{iteminput}'+attr.nodeName+'" class="{iteminput}" data-attrname="'+attr.nodeName+'" type="text" value="'+attr.nodeValue+'"/></div>'));
153 				$container.append(item);
154 			}
155 			$container.find(this.nsSel('iteminput')).blur(function(){
156 				var value = jQuery(this).val();
157 				var name = jQuery(this).attr('data-attrname');
158 				
159 				if (typeof value == 'undefined' || value == '') {
160 					jQuery(this).parents(that.nsSel('item')).remove();
161 					pl.correchtHeight();
162 				} else {
163 					jQuery(el).attr(name,value);
164 				}
165 			});
166 			var elemheader = this.content.find('#' + this.nsClass('element'));
167 			elemheader.html("Element: " + el.tagName);
168 		},
169 		
170 		correctHeight: function() {
171 			this.sidebar.correctHeight();
172 		},
173 		
174 		initSidebar: function(sidebar) {
175 			var pl = this;
176 			pl.sidebar = sidebar;
177 			var sidebarcontent = this.getSidebarContent();
178 			sidebar.addPanel({
179                     
180                     id         : pl.nsClass('sidebar-panel'),
181                     title     : 'Attribute',
182                     content     : '',
183                     expanded : true,
184                     activeOn : function(ef){return pl.activeOn(ef);},
185                     
186                     onInit     : function () {
187                         var that = this;
188                         pl.content = this.setContent(sidebarcontent).content;
189                         
190                         pl.content.find('#'+pl.nsClass('newattributebutton')).click(function () {
191                             var name = jQuery('#'+pl.nsClass('newattributename')).val();
192 							var wert = jQuery('#'+pl.nsClass('newattributewert')).val();
193 							jQuery('#'+pl.nsClass('newattributename')).val('');
194 							jQuery('#'+pl.nsClass('newattributewert')).val('');
195 							jQuery(pl.effective).attr(name, wert);
196 							pl.updateSidebarWithAttributes();
197 							pl.correchtHeight();
198                         });
199 						/*
200 						content.find(nsSel('reset-button')).click(function () {
201                             var content = that.content;
202                             pl.processH(that.effective);
203 							jQuery(that.effective).removeClass('aloha-customized');
204 							that.content.find(nsSel('input')).val(that.effective.attr('id'));
205                         });*/
206                     },
207                     
208                     onActivate: function (effective) {
209 						var that = this;
210 						that.effective = effective;
211 						//DO STUFF HERE
212 						pl.effective = effective;
213 						pl.updateSidebarWithAttributes();
214 						pl.correctHeight();
215                     }
216                     
217                 });
218 		}
219 	});
220 });
