  8 define(
  9 ['aloha', 'aloha/jquery', 'aloha/plugin', 'undo/vendor/undo', 'undo/vendor/diff_match_patch_uncompressed'],
 10 function( Aloha, jQuery, Plugin) {
 11 	"use strict";
 12 	var
 13 	    dmp = new diff_match_patch,
 14 	    resetFlag = false;
 15 
 16 	function reversePatch(patch) {
 17 		var reversed = dmp.patch_deepCopy(patch);
 18 		for (var i = 0; i < reversed.length; i++) {
 19 			for (var j = 0; j < reversed[i].diffs.length; j++) {
 20 				reversed[i].diffs[j][0] = -(reversed[i].diffs[j][0]);
 21 			}
 22 		}
 23 		return reversed;
 24 	}
 25 
 26 	/**
 27 	 * register the plugin with unique name
 28      */
 29 	return Plugin.create('undo', {
 30 		/**
 31 		 * Initialize the plugin and set initialize flag on true
 32 		 */
 33 		init: function () {
 34 			var stack = new Undo.Stack(),
 35 			    EditCommand = Undo.Command.extend({
 36 					constructor: function(editable, patch) {
 37 						this.editable = editable;
 38 						this.patch = patch;
 39 					},
 40 					execute: function() {
 41 						//command object is created after execution.
 42 					},
 43 					undo: function() {
 44 						this.phase(reversePatch(this.patch));
 45 					},
 46 					redo: function() {
 47 						this.phase(this.patch);
 48 					},
 49 					phase: function(patch) {
 50 						var contents = this.editable.getContents(),
 51 						    applied = dmp.patch_apply(patch, contents),
 52 						    newValue = applied[0],
 53 						    didNotApply = applied[1];
 54 						if (didNotApply.length) {
 55 							//error
 56 						}
 57 						this.reset(newValue);
 58 					},
 59 					reset: function(val) {
 60 						//we have to trigger a smartContentChange event
 61 						//after doing an undo or redo, but we mustn't
 62 						//push new commands on the stack, because there
 63 						//are no new commands, just the old commands on
 64 						//the stack that are undone or redone.
 65 						resetFlag = true;
 66 
 67 						var reactivate = null;
 68 						if (Aloha.getActiveEditable() === this.editable) {
 69 							Aloha.deactivateEditable();
 70 							reactivate = this.editable;
 71 						}
 72 
 73 						this.editable.obj.html(val);
 74 
 75 						if (null !== reactivate) {
 76 							reactivate.activate();
 77 						}
 78 
 79 						//TODO: this is a call to an internal
 80 						//function. There should be an API to generate
 81 						//new smartContentChangeEvents.
 82 						this.editable.smartContentChange({type : 'blur'});
 83 
 84 						resetFlag = false;
 85 					}
 86 				});
 87 
 88 			stack.changed = function() {
 89 				// update UI
 90 			};
 91 
 92 			jQuery(document).keydown(function(event) {
 93 				if (!event.metaKey || event.keyCode != 90) {
 94 					return;
 95 				}
 96 				event.preventDefault();
 97 
 98 				//Before doing an undo, bring the smartContentChange
 99 				//event up to date.
100 				if ( null !== Aloha.getActiveEditable() ) {
101 					Aloha.getActiveEditable().smartContentChange({type : 'blur'});
102 				}
103 
104 				if (event.shiftKey) {
105 					stack.canRedo() && stack.redo();
106 				} else {
107 					stack.canUndo() && stack.undo();
108 				}
109 			});
110 
111 			Aloha.bind('aloha-smart-content-changed', function(jevent, aevent) {
112 				if (resetFlag) {
113 					return;
114 				}
115 				var oldValue = aevent.snapshotContent,
116 				newValue = aevent.editable.getContents(),
117 				patch = dmp.patch_make(oldValue, newValue);
118 				// only push an EditCommand if something actually changed.
119 				if (0 !== patch.length) {
120 					stack.execute( new EditCommand( aevent.editable, patch ) );
121 				}
122 			});
123 		},
124 
125 		/**
126 		 * toString method
127 		 * @return string
128 		 */
129 		toString: function () {
130 			return 'undo';
131 		}
132 
133 	});
134 });
