150 	var AbstractTwoColumnBlock = block.AbstractBlock.extend({
151 		containerDefinitions: {
152 			left: {
153 				selector: '.column-left'
154 			},
155 			right: {
156 				selector: '.column-right'
157 			}
158 		},
159 
160 		contents: null,
161 
162 		_initialized: false,
163 
164 		_registerAsBlockified: function() {
165 			// do NOT register myself as blockified yet, as we have to wait for our children to be blockified.
166 		},
167 
168 		whenAllChildrenBlockified: function(next) {
169 			var that = this, numberOfNotYetBlockifiedElements = 0, $containersWithDomBlocks = {};
170 
171 			var template = jQuery('<div />').html(this.$innerElement.html());
172 
173 			var findBlocksForDomElements = function() {
174 				var containersWithBlocks = {};
175 				jQuery.each($containersWithDomBlocks, function(containerName, $domBlocks) {
176 					containersWithBlocks[containerName] = jQuery.map($domBlocks, function($domBlock) {
177 						return BlockManager.getBlock($domBlock);
178 					})
179 				});
180 				return containersWithBlocks;
181 			}
182 
183 			var decrementAndCheckNumberOfBlockifiedElements = function() {
184 				numberOfNotYetBlockifiedElements--;
185 				if (numberOfNotYetBlockifiedElements <= 0) {
186 					next(findBlocksForDomElements());
187 				}
188 			}
189 			jQuery.each(this.containerDefinitions, function(name, config) {
190 				template.find(config.selector).empty();
191 
192 				var $container = that.element.find(config.selector).first();
193 				var $elements = $container.children(); // TODO: block container only contains other blocks as children, nothing else!
194 				$elements.each(function() {
195 					var $element = jQuery(this);
196 					if (!BlockManager.getBlock($element)) {
197 						// Not blockified yet
198 						numberOfNotYetBlockifiedElements++;
199 						$element.bind('block-initialized', decrementAndCheckNumberOfBlockifiedElements);
200 					}
201 				});
202 				$containersWithDomBlocks[name] = $elements;
203 			});
204 
205 			if (!this.attr('template')) {
206 				this.attr('template', template.html());
207 			}
208 
209 			if (numberOfNotYetBlockifiedElements <= 0) {
210 				next(findBlocksForDomElements());
211 			}
212 		},
213 		init: function() {
214 			var that = this;
215 			this.whenAllChildrenBlockified(function(containersWithBlocks) {
216 
217 				// Serialize child containers / blocks
218 				jQuery.each(containersWithBlocks, function(containerName, blockList) {
219 					var serializedBlocks = [];
220 
221 					jQuery.each(blockList, function() {
222 						var block = this;
223 						serializedBlocks.push(block.serialize());
224 						// TODO for later -- if we have stuff between blocks - block.element.children().remove() might be helpful.
225 					});
226 					that.attr(containerName, JSON.stringify(serializedBlocks)); // TODO: use cross-browser version of JSON.stringify.
227 				});
228 
229 				that._initialized = true;
230 				that.element.trigger('block-initialized');
231 			});
232 		},
233 
234 		render: function(innerElement) {
235 			var that = this;
236 			if (!this._initialized) {
237 				// If block is not yet initialized, we will defer rendering internally.
238 				this.element.bind('block-initialized', function() {
239 					that.element.unbind('block-initialized');
240 					that.render(innerElement);
241 				});
242 
243 				return;
244 			}
245 			innerElement.html(this.attr('template'));
246 			jQuery.each(this.containerDefinitions, function(name, config) {
247 				var serializedBlocks = JSON.parse(that.attr(name)); // TODO: use cross-browser version of JSON.parse.
248 				jQuery.each(serializedBlocks, function() {
249 					var serializedBlock = this;
250 					// TODO: copy/paste code from BlockPasteHandler
251 					var newBlock = jQuery('<' + serializedBlock.tag + '/>');
252 					newBlock.attr('class', serializedBlock.classes);
253 
254 					jQuery.each(serializedBlock.attributes, function(k, v) {
255 						if (k === 'about') {
256 							newBlock.attr('about', v);
257 						} else {
258 							newBlock.attr('data-' + k, v);
259 						}
260 					});
261 
262 					innerElement.find(config.selector).append(newBlock);
263 					BlockManager._blockify(newBlock);
264 				});
265 			});
266 			this._currentlyRendering = false;
267 		}
268 	});
