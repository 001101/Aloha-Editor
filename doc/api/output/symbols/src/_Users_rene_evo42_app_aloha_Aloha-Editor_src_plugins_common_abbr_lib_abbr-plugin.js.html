  8 define( [
  9 	'aloha',
 10 	'aloha/jquery',
 11 	'aloha/plugin',
 12 	'aloha/floatingmenu',
 13 	'i18n!abbr/nls/i18n',
 14 	'i18n!aloha/nls/i18n'
 15 ], function ( Aloha, jQuery, Plugin, FloatingMenu, i18n, i18nCore ) {
 16 	"use strict";
 17 	
 18 	var GENTICS = window.GENTICS;
 19 
 20 	/**
 21 	 * register the plugin with unique name
 22 	 */
 23 	return Plugin.create( 'abbr', {
 24 		/**
 25 		 * Configure the available languages
 26 		 */
 27 		languages: [ 'en', 'de' ],
 28 
 29 		/**
 30 		 * default button configuration
 31 		 */
 32 		config: [ 'abbr' ],
 33 
 34 		/**
 35 		 * Initialize the plugin and set initialize flag on true
 36 		 */
 37 		init: function () {
 38 			this.createButtons();
 39 		    this.subscribeEvents();
 40 		    this.bindInteractions();
 41 		},
 42 
 43 		/**
 44 		 * Initialize the buttons
 45 		 */
 46 		createButtons: function () {
 47 		    var me = this;
 48 
 49 		    // format Abbr Button
 50 		    // this button behaves like a formatting button like (bold, italics, etc)
 51 		    this.formatAbbrButton = new Aloha.ui.Button( {
 52 		    	'name' : 'abbr',
 53 		        'iconClass' : 'aloha-button aloha-button-abbr',
 54 		        'size' : 'small',
 55 		        'onclick' : function () { me.formatAbbr(); },
 56 		        'tooltip' : i18n.t( 'button.abbr.tooltip' ),
 57 		        'toggle' : true
 58 		    } );
 59 		    FloatingMenu.addButton(
 60 		        'Aloha.continuoustext',
 61 		        this.formatAbbrButton,
 62 		        i18nCore.t( 'floatingmenu.tab.format' ),
 63 		        1
 64 		    );
 65 
 66 		    // insert Abbr
 67 		    // always inserts a new abbr
 68 		    this.insertAbbrButton = new Aloha.ui.Button( {
 69 		    	'name' : 'insertAbbr',
 70 		    	'iconClass' : 'aloha-button aloha-button-abbr',
 71 		        'size' : 'small',
 72 		        'onclick' : function () { me.insertAbbr( false ); },
 73 		        'tooltip' : i18n.t( 'button.addabbr.tooltip' ),
 74 		        'toggle' : false
 75 		    } );
 76 			FloatingMenu.addButton(
 77 		        'Aloha.continuoustext',
 78 		        this.insertAbbrButton,
 79 		        i18nCore.t( 'floatingmenu.tab.insert' ),
 80 		        1
 81 		    );
 82 
 83 		    // add the new scope for abbr
 84 		    FloatingMenu.createScope( 'abbr', 'Aloha.continuoustext' );
 85 
 86 		    this.abbrField = new Aloha.ui.AttributeField( {
 87 		    	'width': 320,
 88 		    	'name': 'abbrText'
 89 		    } );
 90 		    // add the input field for abbr
 91 		    FloatingMenu.addButton(
 92 		        'abbr',
 93 		        this.abbrField,
 94 		        i18n.t( 'floatingmenu.tab.abbr' ),
 95 		        1
 96 		    );
 97 		},
 98 
 99 		/**
100 		 * Parse a all editables for abbreviations
101 		 * Add the abbr shortcut to all edtiables
102 		 */
103 		bindInteractions: function () {
104 			var me = this;
105 			
106 		    // on blur check if abbr title is empty. If so remove the a tag
107 		    this.abbrField.addListener( 'blur', function ( obj, event ) {
108 		        if ( this.getValue() == '' ) {
109 		            me.removeAbbr();
110 		        }
111 		    } );
112 
113 		    // add to all editables the abbr shortcut
114 		    for ( var i = 0; i < Aloha.editables.length; i++ ) {
115 		        // CTRL+G
116 		        Aloha.editables[ i ].obj.keydown( function ( e ) {
117 		    		if ( e.metaKey && e.which == 71 ) {
118 				        if ( me.findAbbrMarkup() ) {
119 				        	FloatingMenu.activateTabOfButton( 'abbrText' );
120 				            me.abbrField.focus();
121 				        } else {
122 				        	me.insertAbbr();
123 				        }
124 						
125 				        // prevent from further handling
126 			            // on a MAC Safari cursor would jump to location bar. Use ESC then META+L
127 				        e.stopPropagation();
128 				        e.preventDefault();
129 						
130 			            return false;
131 		    		}
132 		        } );
133 		    }
134 		},
135 
136 		/**
137 		 * Subscribe for events
138 		 */
139 		subscribeEvents: function () {
140 			var me = this;
141 
142 		    // add the event handler for selection change
143 			Aloha.bind( 'aloha-selection-changed', function ( event, rangeObject ) {
144 		        if ( Aloha.activeEditable ) {
145 		        	// show/hide the button according to the configuration
146 		        	var config = me.getEditableConfig( Aloha.activeEditable.obj );
147 
148 		        	if ( jQuery.inArray( 'abbr', config ) != -1 ) {
149 		        		me.formatAbbrButton.show();
150 		        		me.insertAbbrButton.show();
151 		        	} else {
152 		        		me.formatAbbrButton.hide();
153 		        		me.insertAbbrButton.hide();
154 			        	// TODO this should not be necessary here!
155 			        	// FloatingMenu.doLayout();
156 		        		// leave if a is not allowed
157 		        		return;
158 		        	}
159 
160 		//        if ( !Aloha.Selection.mayInsertTag('abbr') ) {
161 		//        	me.insertAbbrButton.hide();
162 		//        }
163 
164 		        	var foundMarkup = me.findAbbrMarkup( rangeObject );
165 		        	if ( foundMarkup ) {
166 		        		// abbr found
167 		        		me.insertAbbrButton.hide();
168 		        		me.formatAbbrButton.setPressed( true );
169 		        		FloatingMenu.setScope( 'abbr' );
170 		        		me.abbrField.setTargetObject( foundMarkup, 'title' );
171 		        	} else {
172 		        		// no abbr found
173 		        		me.formatAbbrButton.setPressed( false );
174 		        		me.abbrField.setTargetObject( null );
175 		        	}
176 		        	// TODO this should not be necessary here!
177 		        	// FloatingMenu.doLayout();
178 		        }
179 		    });
180 		},
181 
182 		/**
183 		 * Check whether inside a abbr tag
184 		 * @param {GENTICS.Utils.RangeObject} range range where to insert the object (at start or end)
185 		 * @return markup
186 		 * @hide
187 		 */
188 		findAbbrMarkup: function ( range ) {
189 			if ( typeof range == 'undefined' ) {
190 		        var range = Aloha.Selection.getRangeObject();
191 		    }
192 			
193 			if ( Aloha.activeEditable ) {
194 			    return range.findMarkup( function() {
195 			        return this.nodeName.toLowerCase() == 'abbr';
196 			    }, Aloha.activeEditable.obj );
197 			} else {
198 				return null;
199 			}
200 		},
201 
202 		/**
203 		 * Format the current selection or if collapsed the current word as abbr.
204 		 * If inside a abbr tag the abbr is removed.
205 		 */
206 		formatAbbr: function () {
207 			var range = Aloha.Selection.getRangeObject();
208 
209 		    if ( Aloha.activeEditable ) {
210 		        if ( this.findAbbrMarkup( range ) ) {
211 		            this.removeAbbr();
212 		        } else {
213 		            this.insertAbbr();
214 		        }
215 		    }
216 		},
217 
218 		/**
219 		 * Insert a new abbr at the current selection. When the selection is collapsed,
220 		 * the abbr will have a default abbr text, otherwise the selected text will be
221 		 * the abbr text.
222 		 */
223 		insertAbbr: function ( extendToWord ) {
224 		    // current selection or cursor position
225 		    var range = Aloha.Selection.getRangeObject();
226 
227 		    // do not insert a abbr in a abbr
228 		    if ( this.findAbbrMarkup( range ) ) {
229 		        return;
230 		    }
231 
232 		    // activate floating menu tab
233 		    FloatingMenu.activateTabOfButton('abbrText');
234 
235 		    // if selection is collapsed then extend to the word.
236 		    if ( range.isCollapsed() && extendToWord != false ) {
237 		        GENTICS.Utils.Dom.extendToWord( range );
238 239 		    }
240 			
241 		    if ( range.isCollapsed() ) {
242 		        // insert a abbr with text here
243 		        var abbrText = i18n.t( 'newabbr.defaulttext' );
244 		        var newAbbr = jQuery( '<abbr title="">' + abbrText + '</abbr>' );
245 		        GENTICS.Utils.Dom.insertIntoDOM( newAbbr, range, jQuery( Aloha.activeEditable.obj ) );
246 		        range.startContainer = range.endContainer = newAbbr.contents().get( 0 );
247 		        range.startOffset = 0;
248 		        range.endOffset = abbrText.length;
249 		    } else {
250 		        var newAbbr = jQuery( '<abbr title=""></abbr>' );
251 		        GENTICS.Utils.Dom.addMarkup( range, newAbbr, false );
252 		    }
253 			
254 		    range.select();
255 			
256 		    this.abbrField.focus();
257 		//	this.abbrChange();
258 		},
259 
260 		/**
261 		 * Remove an a tag.
262 		 */
263 		removeAbbr: function () {
264 		    var range = Aloha.Selection.getRangeObject();
265 		    var foundMarkup = this.findAbbrMarkup();
266 		    if ( foundMarkup ) {
267 		        // remove the abbr
268 		        GENTICS.Utils.Dom.removeFromDOM( foundMarkup, range, true );
269 		        // select the (possibly modified) range
270 		        range.select();
271 		    }
272 		},
273 
274 		/**
275 		 * Make the given jQuery object (representing an editable) clean for saving
276 		 * Find all abbrs and remove editing objects
277 		 * @param obj jQuery object to make clean
278 		 * @return void
279 		 */
280 		makeClean: function ( obj ) {
281 			// nothing to do...
282 		},
283 
284 		/**
285 		* toString method
286 		* @return string
287 		*/
288 		toString: function () {
289 			return 'abbr';
290 		}
291 
292 	} );
293 	
294 } );
