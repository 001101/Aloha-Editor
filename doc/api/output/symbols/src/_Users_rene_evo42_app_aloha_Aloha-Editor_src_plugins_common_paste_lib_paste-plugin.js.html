  8 /**
  9  * Paste Plugin
 10  * ------------
 11  * The paste plugin intercepts all browser paste events that target aloha-
 12  * editables, and redirects the events into a hidden div. Once pasting is done
 13  * into this div, its contents will be processed by registered content handlers
 14  * before being copied into the active editable, at the current range.
 15  */
 16 
 17 define(
 18 [ 'aloha/core', 'aloha/plugin', 'aloha/jquery', 'aloha/command',
 19   'aloha/console' ],
 20 function ( Aloha, Plugin, jQuery, Commands, console ) {
 21 	'use strict';
 22 	
 23 	var GENTICS = window.GENTICS,
 24 	    $window = jQuery( window ),
 25 	    pasteRange = null,
 26 	    pasteEditable = null;
 27 	
 28 	// We need to hide the editable div. We'll use clip:rect for chrome and IE,
 29 	// and width/height for FF
 30 	var $pasteDiv = jQuery( '<div id="pasteContainer" ' +
 31 			'style="position:absolute; clip:rect(0px, 0px, 0px, 0px); ' +
 32 			'width: 1px; height: 1px;"></div>' ).contentEditable( true );
 33 	
 34 	/**
 35 	 * Redirects the paste event into the hidden pasteDiv
 36 	 */
 37 	function redirectPaste () {
 38 		// store the current range
 39 		pasteRange = Aloha.getSelection().getRangeAt( 0 );
 40 		pasteEditable = Aloha.activeEditable;
 41 		
 42 		// store the current scroll position
 43 		$pasteDiv.css( {
 44 			top: $window.scrollTop(),
 45 			left: $window.scrollLeft() - 200
 46 		} );
 47 		
 48 		// empty the pasteDiv
 49 		$pasteDiv.contents().remove();
 50 		
 51 		if ( pasteEditable ) {
 52 			// TODO test in IE!
 53 			pasteEditable.obj.blur();
 54 		}
 55 		
 56 		GENTICS.Utils.Dom.setCursorInto( $pasteDiv.get( 0 ) );
 57 		
 58 		$pasteDiv.focus();
 59 	};
 60 
 61 	/**
 62 	 * Gets the pasted content and inserts them into the current active
 63 	 * editable
 64 	 */
 65 	function getPastedContent () {
 66 		var that = this,
 67 		    pasteDivContents;
 68 		
 69 		// insert the content into the editable at the current range
 70 		if ( pasteRange && pasteEditable ) {
 71 			// activate and focus the editable
 72 			// @todo test in IE
 73 			//pasteEditable.activate();
 74 			pasteEditable.obj.click();
 75 			
 76 			pasteDivContents = $pasteDiv.html();
 77 			
 78 			// We need to remove an insidious nbsp that IE inserted into our
 79 			// paste div, otherwise it will result in an empty paragraph being
 80 			// created right before the pasted content, if the pasted content
 81 			// is a paragraph
 82 			if ( jQuery.browser.msie &&
 83 					/^ /.test( pasteDivContents ) ) {
 84 				pasteDivContents = pasteDivContents.substring( 6 );
 85 			}
 86 			
 87 			pasteDivContents = jQuery.trim( pasteDivContents );
 88 			
 89 			// Detects a situation where we are about to paste into a selection
 90 			// that looks like this: <p> [</p>...
 91 			// The nbsp inside the <p> node was placed there to make the empty
 92 			// paragraph visible in HTML5 conformant rendering engines, like
 93 			// WebKit. Without the white space, such browsers would correctly
 94 			// render an empty <p> as invisible.
 95 			// Note that we do not "prop up" otherwise empty paragraph nodes
 96 			// using a <br />, as WebKit does, because IE does display empty
 97 			// paragraphs which are content-editable and so a <br /> results in
 98 			// 2 lines instead of 1 being shown inside the paragraph.
 99 			// If we detect this situation, we remove the white space so that
100 			// when we paste a new paragraph into the paragraph, it is not be
101 			// split, leaving an empty paragraph on top of the pasted content
102 			// 
103 			// We use "/^(\s|%A0)$/.test( escape(" instead of
104 			// "/^(\s| )$/.test( escape(" because it seems that IE
105 			// transforms non-breaking spaces into atomic tokens
106 			var startContainer = pasteRange.startContainer;
107 			if ( startContainer.nodeType == 3 &&
108 					startContainer.parentNode.nodeName == 'P' &&
109 						startContainer.parentNode.childNodes.length == 1 &&
110 							/^(\s|%A0)$/.test( escape( startContainer.data ) ) ) {
111 				startContainer.data = '';
112 				pasteRange.startOffset = 0;
113 				
114 				// In case ... <p> []</p>
115 				if ( pasteRange.endContainer == startContainer ) {
116 					pasteRange.endOffset = 0;
117 				}
118 			}
119 			
120 			if ( Aloha.queryCommandSupported( 'insertHTML' ) ) {
121 				Aloha.execCommand( 'insertHTML', false, pasteDivContents,
122 					pasteRange );
123 			} else {
124 				console.error( 'Common.Paste', 'Command "insertHTML" not ' +
125 					'available. Enable the plugin "common/commands".' );
126 			}
127 		}
128 		
129 		pasteRange = void 0;
130 		pasteEditable = void 0;
131 		
132 		$pasteDiv.contents().remove();
133 	};
134 
135 	// Public Methods
136 	return Plugin.create( 'paste', {
137 		
138 		settings: {},
139 		
140 		init: function () {
141 			var that = this;
142 			
143 			jQuery( 'body' ).append( $pasteDiv );
144 			
145 			// subscribe to the event editableCreated to redirect paste events
146 			// into our hidden pasteDiv
147 			// TODO: move to paste command
148 			// http://support.mozilla.com/en-US/kb/Granting%20JavaScript%20access%20to%20the%20clipboard
149 			// https://code.google.com/p/zeroclipboard/
150 			Aloha.bind( 'aloha-editable-created', function ( event, editable ) {
151 				// browser-dependent events
152 				if ( jQuery.browser.msie ) {
153 					// We will only us the ugly beforepaste hack if we shall
154 					// not access the clipboard
155 					if ( that.settings.noclipboardaccess ) {
156 						editable.obj.bind( 'beforepaste', function ( event ) {
157 							redirectPaste();
158 							event.stopPropagation();
159 						} );
160 					} else {
161 						// uses the execCommand for IE
162 						editable.obj.bind( 'paste', function ( event ) {
163 							redirectPaste();
164 							
165 							var range = document.selection.createRange();
166 							range.execCommand( 'paste' );
167 							
168 							getPastedContent();
169 							// We manually unset the metaKey property because
170 							// the smartContentChange method will not process
171 							// this event if the metaKey property is set
172 							event.metaKey = void 0;
173 							
174 							Aloha.activeEditable.smartContentChange( event );
175 							event.stopPropagation();
176 							
177 							return false;
178 						} );
179 					}
180 				} else {
181 					editable.obj.bind( 'paste', function ( event ) {
182 						redirectPaste();
183 						// We need to accomodate a small amount execution
184 						// window to ensure that pasted content has actually
185 						// been inserted
186 						window.setTimeout( function () {
187 							getPastedContent();
188 							Aloha.activeEditable.smartContentChange( event );
189 							event.stopPropagation();
190 						}, 10 );
191 					} );
192 				}
193 			} );
194 
195 			// bind a handler to the paste event of the pasteDiv to get the
196 			// pasted content (but do this only once, not for every editable)
197 			if ( jQuery.browser.msie && that.settings.noclipboardaccess ) {
198 				$pasteDiv.bind( 'paste', function ( event ) {
199 					window.setTimeout( function () {
200 						getPastedContent();
201 						Aloha.activeEditable.smartContentChange( event );
202 						event.stopPropagation();
203 					}, 10 );
204 				} );
205 			}
206 		},
207 
208 		/**
209 		 * Register the given paste handler
210 		 * @deprecated
211 		 * @param pasteHandler paste handler to be registered
212 		 */
213 		register: function ( pasteHandler ) {
214 			console.deprecated( 'Plugins.Paste', 'register() for ' +
215 				'pasteHandler is deprecated. Use the ContentHandler Plugin ' +
216 				'instead.' );
217 		}
218 		
219 	} );
220 	
221 } );
