  7 define(
  8 [ 'aloha/core', 'aloha/jquery', 'aloha/contenthandlermanager', 'aloha/console', 'vendor/sanitize' ],
  9 function( Aloha, jQuery, ContentHandlerManager, console ) {
 10 	"use strict";
 11 	
 12 	var sanitize;
 13 	
 14 	// needed minimum sanitize configuration for Aloha itselft
 15 	/*Aloha.defaults.supports = jQuery.merge(Aloha.defaults.supports, {
 16 			elements: [ 'br', 'div', 'p', 'span' ]
 17 	});*/
 18 	
 19 	// predefined set of sanitize options if no dynamic or custom config is used
 20 	if( !Aloha.defaults.sanitize ) {
 21 		Aloha.defaults.sanitize = {}
 22 	}
 23 
 24 	// very restricted sanitize config
 25 	Aloha.defaults.sanitize.restricted = {
 26 		elements: [ 'b', 'em', 'i', 'strong', 'u', 'p', 'span', 'div', 'br' ]
 27 	}
 28 
 29 	// sanitize  config allowing a bit more (no tables)
 30 	Aloha.defaults.sanitize.basic = {
 31 		elements: [
 32 			'a', 'abbr', 'b', 'blockquote', 'br', 'cite', 'code', 'dd', 'dl', 'dt', 'em',
 33 			'i', 'li', 'ol', 'p', 'pre', 'q', 'small', 'strike', 'strong', 'sub',
 34 			'sup', 'u', 'ul' ],
 35 
 36 		attributes: {
 37 			'a' : ['href'],
 38 			'blockquote' : ['cite'],
 39 			'q' : ['cite'],
 40 			'abbr': ['title']
 41 		},
 42 
 43 		//add_attributes: {
 44 			//  'a': {'rel': 'nofollow'}
 45 		//},
 46 
 47 		protocols: {
 48 			'a' : {'href': ['ftp', 'http', 'https', 'mailto', '__relative__']},
 49 			'blockquote' : {'cite': ['http', 'https', '__relative__']},
 50 			'q' : {'cite': ['http', 'https', '__relative__']}
 51 		}
 52 	}
 53 
 54 	// relaxed sanitize config allows also tables
 55 	Aloha.defaults.sanitize.relaxed = {
 56 		elements: [
 57 			'a', 'abbr', 'b', 'blockquote', 'br', 'caption', 'cite', 'code', 'col',
 58 			'colgroup', 'dd', 'dl', 'dt', 'em', 'h1', 'h2', 'h3', 'h4', 'h5', 'h6',
 59 			'i', 'img', 'li', 'ol', 'p', 'pre', 'q', 'small', 'strike', 'strong',
 60 			'sub', 'sup', 'table', 'tbody', 'td', 'tfoot', 'th', 'thead', 'tr', 'u',
 61 			'ul', 'span', 'hr', 'object', 'div'
 62 		],
 63 
 64 		attributes: {
 65 			'a': ['href', 'title', 'id', 'data-gentics-aloha-repository', 'data-gentics-aloha-object-id'],
 66 			'div': [ 'id', 'class'],
 67 			'abbr': ['title'],
 68 			'blockquote': ['cite'],
 69 			'br': ['class'],
 70 			'col': ['span', 'width'],
 71 			'colgroup': ['span', 'width'],
 72 			'img': ['align', 'alt', 'height', 'src', 'title', 'width', 'class'],
 73 			'ol': ['start', 'type'],
 74 			'q': ['cite'],
 75 			'table': ['summary', 'width'],
 76 			'td': ['abbr', 'axis', 'colspan', 'rowspan', 'width'],
 77 			'th': ['abbr', 'axis', 'colspan', 'rowspan', 'scope', 'width'],
 78 			'ul': ['type']
 79 		},
 80 
 81 		protocols: {
 82 			'a': {'href': ['ftp', 'http', 'https', 'mailto', '__relative__']},
 83 			'blockquote': {'cite': ['http', 'https', '__relative__']},
 84 			'img': {'src' : ['http', 'https', '__relative__']},
 85 			'q': {'cite': ['http', 'https', '__relative__']}
 86 		}
 87 	}
 88 
 89 	function initSanitize () {
 90 		var 
 91 			filter = [ 'restricted', 'basic', 'relaxed' ],
 92 			config = Aloha.defaults.supports; // @TODO: needs to be implemented into all plugins
 93 		
 94 		// @TODO think about Aloha.settings.contentHandler.sanitize name/options
 95 		if ( Aloha.settings.contentHandler.sanitize && jQuery.inArray(Aloha.settings.contentHandler.sanitize, filter) > -1 ) {
 96 			config = eval('Aloha.defaults.sanitize.' + Aloha.settings.contentHandler.sanitize);
 97 		} else {
 98 			// use relaxed filter by default
 99 			config = Aloha.defaults.sanitize.relaxed;
100 		}
101 		
102 		// @TODO move to Aloha.settings.contentHandler.sanitize.allows ?
103 		if ( Aloha.settings.contentHandler.allows ) {
104 			config = Aloha.settings.contentHandler.allows;
105 		}
106 
107 		// add a filter to stop cleaning elements with contentEditable "false"
108 		config.filters = [function( elem ) {
109 			return elem.contentEditable != "false";
110 		}];
111 
112 		sanitize = new Sanitize( config );
113 	}
114 
115 	var SanitizeContentHandler = ContentHandlerManager.createHandler({
116 		/**
117 		 * Handle the content from eg. paste action and sanitize the html
118 		 * @param content
119 		 */
120 		handleContent: function( content )  {
121 			if ( typeof sanitize === 'undefined' ) {
122 			   initSanitize();
123 			}
124 
125 			if ( typeof content === 'string' ){
126 				content = jQuery( '<div>' + content + '</div>' ).get(0);
127 			} else if ( content instanceof jQuery ) {
128 				content = jQuery( '<div>' ).append(content).get(0);
129 			}
130 
131 			return jQuery('<div>').append(sanitize.clean_node(content)).html();
132 		}
133 	});
134 	
135 	return SanitizeContentHandler;
136 });
