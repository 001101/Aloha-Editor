  8 define([ 'aloha/jquery', 'block/blockmanager', 'aloha/sidebar', 'block/editormanager'],
  9 	function (jQuery, BlockManager, Sidebar, EditorManager) {
 10 	"use strict";
 11 
 12 	/**
 13 	 * @name block.sidebarattributeeditor
 14 	 * @class Sidebar attribute editor singleton
 15 	 */
 16 	return new (Class.extend(
 17 	/** @lends block.sidebarattributeeditor */
 18 	{
 19 
 20 		_sidebar: null,
 21 
 22 		/**
 23 		 * Initialize the sidebar attribute editor and bind events
 24 		 */
 25 		init: function() {
 26 			var that = this;
 27 			
 28 			//Obsolete: that._initSidebar();
 29 			this._sidebar = Sidebar.right.show();
 30 			
 31 			BlockManager.bind('block-selection-change', this._onBlockSelectionChange, this);
 32 		},
 33 		
 34 		/* Obsolete:
 35 			We no longer need to initialize our own sidebar.
 36 			We will use those that are provided to use by Aloha
 37 				
 38 		_initSidebar: function() {
 39 			return;
 40 			this._sidebar = new Sidebar({
 41 				position: 'right',
 42 				width: 250,
 43 				isOpen: true,
 44 				panels: []
 45 			});
 46 		},
 47 		 */
 48 
 49 		/**
 50 		 * @param {Array} selectedBlocks
 51 		 */
 52 		_onBlockSelectionChange: function(selectedBlocks) {
 53 			var that = this;
 54 			if (!this._sidebar) {
 55 				return;
 56 			}
 57 			// TODO: Clearing the whole sidebar might not be what we want; instead we might only want
 58 			// to clear certain panels.
 59 			that._sidebar.container.find('.aloha-sidebar-panels').children().remove();
 60 			that._sidebar.panels = {};
 61 
 62 			jQuery.each(selectedBlocks, function() {
 63 				var schema = this.getSchema(),
 64 					block = this,
 65 					editors = [];
 66 
 67 				if (!schema) {
 68 					// If no schema returned, we do not want to add panels.
 69 					return;
 70 				}
 71 				that._sidebar.addPanel({
 72 					title: block.getTitle(),
 73 					expanded: true,
 74 					onInit: function() {
 75 						var $form = jQuery('<form />');
 76 						$form.submit(function() {
 77 							// Disable form submission
 78 							return false;
 79 						});
 80 						jQuery.each(schema, function(attributeName, definition) {
 81 							var editor = EditorManager.createEditor(definition);
 82 
 83 							// Editor -> Block binding
 84 							editor.bind('change', function(value) {
 85 								block.attr(attributeName, value);
 86 							});
 87 
 88 							// Block -> Editor binding
 89 							block.bind('change', function() {
 90 								editor.setValue(block.attr(attributeName));
 91 							})
 92 
 93 							$form.append(editor.render());
 94 
 95 							// Set initial value Block -> Editor
 96 							editor.setValue(block.attr(attributeName));
 97 
 98 							editors.push(editor);
 99 						});
100 						this.setContent($form);
101 					},
102 
103 					deactivate: function() {
104 						// On deactivating the panel, we need to tell each editor to deactivate itself,
105 						// so it can throw another change event.
106 						jQuery.each(editors, function(index, editor) {
107 							editor._deactivate();
108 						});
109 
110 						// This code is from the superclass
111 						this.isActive = false;
112 						this.content.parent('li').hide();
113 						this.effectiveElement = null;
114 					}
115 				});
116 			});
117 		}
118 	}))();
119 });
