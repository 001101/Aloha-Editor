127 		/**
128 		 * Apply the given markup to the given range. This will either
129 		 * add or remove the markup (depending on whether the markup is
130 		 * currently active at the start of the range)
131 		 * @param editable editable as jQuery object
132 		 * @param rangeObject range object
133 		 * @param markup as jQuery object
134 		 * @param nesting true when nesting of the markup is allowed, false if not
135 		 */
136 		applyMarkup : function (editable, rangeObject, markup, nesting) {
137 			var markupIsApplied = false;
138 			rangeObject.clearCaches();
139 			rangeObject.updateMarkupEffectiveAtStart();
140 
141 			for (var i = 0; i < rangeObject.markupEffectiveAtStart.length; i++) {
142 				var effectiveMarkup = rangeObject.markupEffectiveAtStart[ i ];
143 				if (Aloha.Selection.standardTextLevelSemanticsComparator(effectiveMarkup, markup)) {
144 					markupIsApplied = true;
145 				}
146 			}
147 
148 			if (markupIsApplied) {
149 				GENTICS.Utils.Dom.removeMarkup(rangeObject, markup, editable);
150 			} else {
151 				GENTICS.Utils.Dom.addMarkup(rangeObject, markup, nesting);
152 			}
153 			rangeObject.correctRange();
154 			rangeObject.select();
155 			rangeObject.clearCaches();
156 			rangeObject.updateMarkupEffectiveAtStart();
157 		},

231 		/**
232 		 * Parses selection markers in the innerHTML of the given element and 
233 		 * creates a range object whose start and end determined from these
234 		 * selection markers.
235 		 *
236 		 * @param {editable} - DOM Node element containing two selection
237 		 *					   markers. Exactly one of these two markers should
238 		 *					   be a start marker string (ie: "[", "{", or
239 		 *					   "data-start"), and one should be an end marker
240 		 *					   string (ie: "]", "}", "data-end").
241 		 *					   
242 		 * @param {Object: range} Range object
243 		 */
244 		addRange: function ( editable ) {
245 			function nextNode(node) {
246 				if (node.hasChildNodes()) {
247 					return node.firstChild;
248 				}
249 				return nextNodeDescendants(node);
250 			};
251 			
252 			function nextNodeDescendants(node) {
253 				while (node && !node.nextSibling) {
254 					node = node.parentNode;
255 				}
256 				if (!node) {
257 					return null;
258 				}
259 				return node.nextSibling;
260 			};
261 			
262 			function getNodeIndex(node) {
263 				var ret = 0;
264 				while (node.previousSibling) {
265 					ret++;
266 267 					node = node.previousSibling;
268 				}
269 				return ret;
270 			};
271 			
272 			var html = editable.html();
273 			// A variety of checks to avoid simple errors.  Not foolproof, of course.
274 			var re = /\{|\[|data-start/g;
275 			var markers = [];
276 			var marker;
277 			while (marker = re.exec(html)) {
278 				markers.push(marker);
279 			}
280 			if (markers.length != 1) {
281 				throw "Need exactly one start marker ([ or { or data-start), found " + markers.length;
282 			}
283 
284 			var re = /\}|\]|data-end/g;
285 			var markers = [];
286 287 			var marker;
288 			while (marker = re.exec(html)) {
289 				markers.push(marker);
290 			}
291 			if (markers.length != 1) {
292 				throw "Need exactly one end marker (] or } or data-end), found " + markers.length;
293 			}
294 
295 			var node = editable[0];
296 
297 			var startNode, startOffset, endNode, endOffset;
298 
299 			// For braces that don't lie inside text nodes, we can't just set
300 			// innerHTML, because that might disturb the DOM.  For instance, if the
301 			// brace is right before a <tr>, it could get moved outside the table
302 			// entirely, which messes everything up pretty badly.  So we instead
303 			// allow using data attributes: data-start and data-end on the start and
304 			// end nodes, with a numeric value indicating the offset.  This format
305 			// doesn't allow the parent div to be a start or end node, but in that case
306 			// you can always use the curly braces.
307       // Lakshan: changed from querySelector to use jQuery selecotrs since former
308       // doesn't work on IE7
309 			if (jQuery(node).find("[data-start]")[0]) {
310 				startNode = jQuery( node ).find("[data-start]")[0];
311 				startOffset = startNode.getAttribute("data-start");
312 				startNode.removeAttribute("data-start");
313 			}
314 			if (jQuery( node ).find("[data-end]")[0]) {
315 				endNode = jQuery( node ).find("[data-end]")[0];
316 				endOffset = endNode.getAttribute("data-end");
317 				endNode.removeAttribute("data-end");
318 			}
319 
320 			var cur = node;
321 			while (true) {
322 				if (!cur || (cur != node && !($_.compareDocumentPosition(cur, node) & $_.Node.DOCUMENT_POSITION_CONTAINS))) {
323 					break;
324 				}
325 
326 				if (cur.nodeType != $_.Node.TEXT_NODE) {
327 					cur = nextNode(cur);
328 					continue;
329 				}
330 
331 				var data = cur.data.replace(/[\]\}]/g, "");
332 				var startIdx = data.indexOf("[");
333 
334 				data = cur.data.replace(/[\[\{]/g, "");
335 				var endIdx = data.indexOf("]");
336 
337 				cur.data = cur.data.replace(/[\[\]]/g, "");
338 
339 				if (startIdx != -1) {
340 					startNode = cur;
341 					startOffset = startIdx;
342 				}
343 
344 				if (endIdx != -1) {
345 					endNode = cur;
346 					endOffset = endIdx;
347 				}
348 
349 				// These are only legal as the first or last
350 				data = cur.data.replace(/\}/g, "");
351 				var elStartIdx = data.indexOf("{");
352 
353 				data = cur.data.replace(/\{/g, "");
354 				var elEndIdx = data.indexOf("}");
355 
356 				if (elStartIdx == 0) {
357 					startNode = cur.parentNode;
358 					startOffset = getNodeIndex(cur);
359 				} else if (elStartIdx != -1) {
360 					startNode = cur.parentNode;
361 					startOffset = getNodeIndex(cur) + 1;
362 				}
363 				if (elEndIdx == 0) {
364 					endNode = cur.parentNode;
365 					endOffset = getNodeIndex(cur);
366 				} else if (elEndIdx != -1) {
367 					endNode = cur.parentNode;
368 					endOffset = getNodeIndex(cur) + 1;
369 				}
370 				
371 				cur.data = cur.data.replace(/[{}]/g, "");
372 				if (!cur.data.length) {
373 					if (cur == startNode || cur == endNode) {
374 						throw "You put a square bracket where there was no text node . . .";
375 					}
376 					var oldCur = cur;
377 					cur = nextNode(cur);
378 					oldCur.parentNode.removeChild(oldCur);
379 				} else {
380 					cur = nextNode(cur);
381 				}
382 			}
383 			
384 			var range = Aloha.createRange();
385 			range.setStart( startNode, startOffset );
386 			range.setEnd( endNode, endOffset );
387 			
388 			return range;
389 		},

391 		addBrackets: function (range) {
392 			var marker;
393 			//@{
394 			// Handle the collapsed case specially, to avoid confusingly getting the
395 			// markers backwards in some cases
396 			if (range.endContainer.nodeType == $_.Node.TEXT_NODE
397 				|| range.endContainer.nodeType == $_.Node.COMMENT_NODE) {
398 					if (range.collapsed) {
399 						marker = '[]'
400 					} else {
401 						marker = ']'
402 					}
403 					range.endContainer.insertData(range.endOffset, marker);
404 				} else {
405 					if (range.collapsed) {
406 						marker = '{}'
407 					} else {
408 						marker = '}'
409 					}
410 					if (range.endOffset != range.endContainer.childNodes.length
411 					&& range.endContainer.childNodes[range.endOffset].nodeType == $_.Node.TEXT_NODE) {
412 						range.endContainer.childNodes[range.endOffset].insertData(0, marker);
413 					} else if (range.endOffset != 0
414 					&& range.endContainer.childNodes[range.endOffset - 1].nodeType == $_.Node.TEXT_NODE) {
415 						range.endContainer.childNodes[range.endOffset - 1].appendData( marker );
416 					} else {
417 						range.endContainer.insertBefore(document.createTextNode( marker ),
418 							range.endContainer.childNodes.length == range.endOffset
419 							? null
420 							: range.endContainer.childNodes[range.endOffset]);
421 					}
422 				}
423 				if (range.collapsed) {
424 					return;
425 				}
426 				if (range.startContainer.nodeType == $_.Node.TEXT_NODE
427 				|| range.startContainer.nodeType == $_.Node.COMMENT_NODE) {
428 					range.startContainer.insertData(range.startOffset, "[");
429 				} else {
430 					marker = '{';
431 					if (range.startOffset != range.startContainer.childNodes.length
432 					&& range.startContainer.childNodes[range.startOffset].nodeType == $_.Node.TEXT_NODE) {
433 						range.startContainer.childNodes[range.startOffset].insertData(0, marker);
434 					} else if (range.startOffset != 0
435 					&& range.startContainer.childNodes[range.startOffset - 1].nodeType == $_.Node.TEXT_NODE) {
436 						range.startContainer.childNodes[range.startOffset - 1].appendData(marker);
437 					} else {
438 						// Seems to serialize as I'd want even for tables . . . IE doesn't
439 						// allow undefined to be passed as the second argument (it throws
440 						// an exception), so we have to explicitly check the number of
441 						// children and pass null.
442 						range.startContainer.insertBefore(document.createTextNode(marker),
443 							range.startContainer.childNodes.length == range.startOffset
444 							? null
445 							: range.startContainer.childNodes[range.startOffset]);
446 					}
447 				}
448 449 			}
450 	});

453 	/**
454 	 * Create a jQuery plugin to extract the HTML of a given jQuery object
455 	 */
456 	jQuery.fn.extractHTML = function( attributes ) {
457 		attributes = typeof attributes === 'undefined' ? ['class', 'id'] : attributes;
458 		var fullResult = [];
459 		if ( typeof attr !== 'undefined' ) {
460 			attributes.push(attr);
461 		}
462 		
463 		jQuery.each(this, function() {
464 			var that = jQuery(this);
465 			var result = {};
466 			result.nodeName = that[0].nodeName.toLowerCase();
467 			fullResult.push(result);
468 			if (that[0].nodeType == 3) {
469 				result.text = that.text();
470 			} else if (that[0].nodeType == 1) {
471 				jQuery.each(attributes, function(index, attr) {
472 					result[attr] = that.attr(attr);
473 				});
474 				var contents = that.contents();
475 				if (contents.length) {
476 					result.contents = contents.extractHTML(attributes);
477 				}
478 			}
479 		});
480 
481 		return fullResult;
482 	};
