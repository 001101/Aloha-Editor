  6 
  7 define(
  8 ['testutils'],
  9 function( TestUtils ) {
 10 	"use strict";
 11 	
 12 	var aQuery = Aloha.jQuery;
 13 
 14 	var browser, browserversion;
 15 	if (aQuery.browser.msie) {
 16 		browser = "msie";
 17 	} else if (aQuery.browser.webkit) {
 18 		browser = "webkit";
 19 	} else if (aQuery.browser.opera) {
 20 		browser = "opera";
 21 	} else if (aQuery.browser.mozilla) {
 22 		browser = "mozilla";
 23 	}
 24 	browserversion = browser + aQuery.browser.version;
 25 
 26 	// Test whether Aloha is properly initialized
 27 	asyncTest( 'Aloha Startup Test', function() {
 28 		var timeout = setTimeout(function() {
 29 			ok(false, 'Aloha was not initialized within 60 seconds');
 30 			start();
 31 		}, 60000);
 32 		Aloha.ready( function() {
 33 			clearTimeout( timeout );
 34 			ok( true, 'Aloha Event was fired' );
 35 			start();
 36 		});
 37 	});
 38 
 39 	// All other tests are done when Aloha is ready
 40 	Aloha.ready( function() {
 41 
 42 		var 
 43 			editable = aQuery( '#edit' ),
 44 			converterResult = aQuery('<div>'),
 45 			converter = aQuery('<div>');
 46 
 47 		// we never want to see the floatingmenu here
 48 		var floatingMenu = Aloha.require('aloha/floatingmenu');
 49 		floatingMenu.doLayout = function() {
 50 			this.hide();
 51 		};
 52 
 53 		// aloha'fy the editable
 54 		editable.aloha();
 55 		
 56 		for ( var i = 0; i < tests.tests.length; i++ ) {
 57 
 58 			var	check = tests.tests[i],
 59 				excluded = false;
 60 
 61 			if ( typeof check === 'undefined' ) {
 62 				check = {};
 63 			}
 64 
 65 			if ( typeof check.exclude === 'undefined' ) {
 66 				check.exclude = false;
 67 			}
 68 
 69 			if ( typeof check.include === 'undefined' ) {
 70 				check.include = false;
 71 			}
 72 
 73 			if (check.exclude && typeof check.exclude === 'string') {
 74 				check.exclude = [check.exclude];
 75 			}
 76 			if (check.include && typeof check.include === 'string') {
 77 				check.include = [check.include];
 78 			}
 79 			// if exclude is set, check whether we need to exclude the test
 80 			if (check.exclude && (check.exclude.indexOf(browser) !== -1 || check.exclude.indexOf(browserversion) !== -1)) {
 81 				excluded = true;
 82 			}
 83 			// if include is set, check whether we shall include the test
 84 			if (check.include && check.include.indexOf(browser) === -1 && check.include.indexOf(browserversion) === -1) {
 85 				excluded = true;
 86 			}
 87 
 88 			check.value = ( typeof check.value !== 'undefined') ? check.value : tests.defaultValue;
 89 			check.attributes = ( typeof check.attributes !== 'undefined') ? check.attributes : tests.defaultAttributes;
 90 
 91 			converter.text(check.start);
 92 			var descStart = converter.html();
 93 
 94 			converter.text(check.value);
 95 			var descValue = converter.html();
 96 
 97 			converterResult.text(check.execResult);
 98 			var descResult = converterResult.html();
 99 
100 			var descName = '"' + descStart + '" → "' + descResult + '"';
101 			if (descValue && descValue !== descStart) {
102 				descName += ' ("' + descValue +'")';
103 			}
104 
105 			var name = check.name ||  descName;
106 
107 			module( tests.defaultCommand + ' ' + (i+1) + (excluded ? ' (EX)' : ''), {
108 				setup: function() {
109 					// fill the editable area with the start value
110 					editable.html(this.check.start);
111 					editable.focus();
112 				},
113 				teardown: function() {
114 					// goodbye
115 				}
116 			});
117 			
118 			if ( excluded ) {
119 				test( name, {check:check}, function() {
120 				});
121 			} else {
122 				test( name, {check:check}, function() {
123 					var 
124 					check = this.check,
125 					command = check.command || tests.defaultCommand,
126 					range = TestUtils.addRange( editable ),
127 					execResult,
128 					execToggle,
129 					result,
130 					r;
131 					
132 					r = Aloha.createRange();
133 					r.setStart( range.startContainer, range.startOffset) ;
134 					r.setEnd( range.endContainer, range.endOffset);
135 					Aloha.getSelection().removeAllRanges();
136 					Aloha.getSelection().addRange(r);
137 					
138 					// Start
139 					if ( typeof check.indetermStart !== 'undefined' ) {
140 						// query command indeterminacy and compare
141 						result = Aloha.queryCommandIndeterm( command );
142 						deepEqual( result, check.indetermStart, 'queryCommandIndeterm start' );
143 					}
144 					if ( typeof check.stateStart !== 'undefined' ) {
145 						// query command state and compare
146 						result = Aloha.queryCommandState( command );
147 						deepEqual( result, check.stateStart, 'queryCommanState start' );
148 					}
149 					if ( typeof check.valueStart !== 'undefined' ) {
150 						// query command value and compare
151 						result = Aloha.queryCommandValue( command );
152 						deepEqual( result, check.valueStart, 'queryCommandValue start' );
153 					}
154 					
155 					// ExecCommand
156 					if ( typeof check.execResult !== 'undefined' ) {
157 						// execute the command
158 						Aloha.execCommand( command, false, check.value );
159 						// place the marker at the selection and add brackets
160 						range = rangy.getSelection().getRangeAt(0);
161 						TestUtils.addBrackets(range);
162 						
163 						// clone the editable object
164 						result = Aloha.editables[0].obj.clone(false);
165 						// get the content as html
166 						var resultHtml = result.html();
167 						// set the content back (this will merge adjacent text nodes and remove empty text nodes)
168 						result.html(resultHtml);
169 						// get the contents of the editable
170 						result = result.contents();
171 						execResult = aQuery( '<div>' + check.execResult + '</div>' );
172 						
173 						// remove browser specific elements from expected results
174 						if (browser) {
175 							execResult.find('[data-test-exclude~="'+browser+'"]').remove();
176 						}
177 						execResult = execResult.contents();
178 						
179 						deepEqual( result.extractHTML( check.attributes ), execResult.extractHTML( check.attributes ), 'execCommand result' );
180 					}
181 					
182 					// Result
183 					if ( typeof check.indetermResult !== 'undefined' ) {
184 						// query command indeterminacy and compare
185 						result = Aloha.queryCommandIndeterm( command );
186 						deepEqual( result, check.indetermResult, 'queryCommandIndeterm result' );
187 					}
188 					if ( typeof check.stateResult !== 'undefined' ) {
189 						// query command state and compare
190 						result = Aloha.queryCommandState( command );
191 						deepEqual( result, check.stateResult, 'queryCommanState result' );
192 					}
193 					if ( typeof check.valueResult !== 'undefined' ) {
194 						// query command value and compare
195 						result = Aloha.queryCommandValue( command );
196 						deepEqual( result, check.valueResult, 'queryCommandValue result' );
197 					}
198 					
199 					// log expected result + result
200 					/* // not working now
201 					if ( typeof execResult === 'object' && typeof result === 'object' ) {
202 						window.console.log( execResult.html() + ' -- ' + result.html());
203 					}*/
204 					
205 					if ( check.execToggle ) {
206 						range = TestUtils.addRange( editable );
207 						
208 						r = Aloha.createRange();
209 						r.setStart( range.startContainer, range.startOffset) ;
210 						r.setEnd( range.endContainer, range.endOffset);
211 						Aloha.getSelection().removeAllRanges();
212 						Aloha.getSelection().addRange(r);
213 						
214 						// toggle ExecCommand
215 						if ( typeof check.execToggle !== 'undefined' ) {
216 							// execute the command
217 							result = Aloha.execCommand( command, false, check.value );
218 							// place the marker at the selection and add brackets
219 220 							range = rangy.getSelection().getRangeAt(0);
221 							TestUtils.addBrackets(range);
222 							result = Aloha.editables[0].getContents( true );			
223 							execToggle = aQuery( '<div>' + check.execToggle + '</div>' ).contents();
224 							deepEqual( result.extractHTML( check.attributes ), execToggle.extractHTML( check.attributes ), 'execCommand toggle result' );
225 						}
226 						
227 						// Toggle result
228 						if ( typeof check.indetermToggle !== 'undefined' ) {
229 							// query command indeterminacy and compare
230 							result = Aloha.queryCommandIndeterm( command );
231 							deepEqual( result, check.indetermToggle, 'queryCommandIndeterm toggle result' );
232 						}
233 						if ( typeof check.stateToggle !== 'undefined' ) {
234 							// query command state and compare
235 							result = Aloha.queryCommandState( command );
236 							deepEqual( result, check.stateToggle, 'queryCommanState toggle result' );
237 						}
238 						if ( typeof check.valueResult !== 'undefined' ) {
239 							// query command value and compare
240 							result = Aloha.queryCommandValue( command );
241 							deepEqual( result, check.valueToggle, 'queryCommandValue toggle result' );
242 						}
243 					}
244 					
245 				});
246 			}
247 		}
248 	});
249 });
