  8 define(
  9 ['aloha', 'aloha/plugin', 'aloha/jquery', 'aloha/floatingmenu', 'i18n!format/nls/i18n', 'i18n!aloha/nls/i18n', 'aloha/console',
 10  		'css!format/css/format.css'],
 11 function(Aloha, Plugin, jQuery, FloatingMenu, i18n, i18nCore) {
 12 	"use strict";
 13 	var
 14 		GENTICS = window.GENTICS;
 15 
 16 	/**
 17 	 * register the plugin with unique name
 18 	 */
 19 	return Plugin.create('format', {
 20 		/**
 21 		 * Configure the available languages
 22 		 */
 23 		languages: ['en', 'de', 'fr', 'eo', 'fi', 'ru', 'it', 'pl'],
 24 
 25 		/**
 26 		 * default button configuration
 27 		 */
 28 		config: [ 'strong', 'em', 'b', 'i','del','sub','sup', 'p', 'h1', 'h2', 'h3', 'h4', 'h5', 'h6', 'pre', 'removeFormat'],
 29 
 30 		/**
 31 		 * Initialize the plugin and set initialize flag on true
 32 		 */
 33 		init: function () {
 34 			// Prepare
 35 			var me = this;
 36 
 37 			this.initButtons();
 38 
 39 			// apply specific configuration if an editable has been activated
 40 			Aloha.bind('aloha-editable-activated',function (e, params) {
 41 				//debugger;
 42 				me.applyButtonConfig(params.editable.obj);
 43 			});
 44 
 45 			/*
 46 			Aloha.defaults.supports = jQuery.merge(Aloha.defaults.supports, {
 47 					elements: [ 'strong', 'em', 'b', 'i','del','sub','sup', 'p', 'h1', 'h2', 'h3', 'h4', 'h5', 'h6', 'pre' ]
 48 			});
 49 			*/
 50 
 51 		},
 52 
 53 		/**
 54 		 * applys a configuration specific for an editable
 55 		 * buttons not available in this configuration are hidden
 56 		 * @param {Object} id of the activated editable
 57 		 * @return void
 58 		 */
 59 		applyButtonConfig: function (obj) {
 60 
 61 			var config = this.getEditableConfig(obj),
 62 				button, i, len;
 63 
 64 			// now iterate all buttons and show/hide them according to the config
 65 			for ( button in this.buttons) {
 66 				if (jQuery.inArray(button, config) != -1) {
 67 					this.buttons[button].button.show();
 68 				} else {
 69 					this.buttons[button].button.hide();
 70 				}
 71 			}
 72 
 73 			// and the same for multisplit items
 74 			len = this.multiSplitItems.length;
 75 			for (i = 0; i < len; i++) {
 76 				if (jQuery.inArray(this.multiSplitItems[i].name, config) != -1) {
 77 					this.multiSplitButton.showItem(this.multiSplitItems[i].name);
 78 				} else {
 79 					this.multiSplitButton.hideItem(this.multiSplitItems[i].name);
 80 				}
 81 			}
 82 		},
 83 
 84 		/**
 85 		 * initialize the buttons and register them on floating menu
 86 		 * @param event event object
 87 		 * @param editable current editable object
 88 		 */
 89 		initButtons: function () {
 90 			var
 91 				scope = 'Aloha.continuoustext',
 92 				that = this;
 93 
 94 			// reset
 95 			this.buttons = {};
 96 
 97 			// collect the multisplit items here
 98 			this.multiSplitItems = [];
 99 			//this.multiSplitButton;
100 
101 			//iterate configuration array an push buttons to buttons array
102 			jQuery.each(this.config, function(j, button) {
103 				switch( button ) {
104 					// text level semantics:
105 					case 'em':
106 					case 'strong':
107 					case 'b':
108 					case 'i':
109 					case 'cite':
110 					case 'q':
111 					case 'code':
112 					case 'abbr':
113 					case 'del':
114 					case 'sub':
115 					case 'sup':
116 						that.buttons[button] = {'button' : new Aloha.ui.Button({
117 							'name' : button,
118 							'iconClass' : 'aloha-button aloha-button-' + button,
119 							'size' : 'small',
120 							'onclick' : function () {
121 								var
122 									markup = jQuery('<'+button+'></'+button+'>'),
123 									rangeObject = Aloha.Selection.rangeObject,
124 									foundMarkup;
125 
126 								// check whether the markup is found in the range (at the start of the range)
127 								foundMarkup = rangeObject.findMarkup(function() {
128 									return this.nodeName.toLowerCase() == markup.get(0).nodeName.toLowerCase();
129 								}, Aloha.activeEditable.obj);
130 
131 								if (foundMarkup) {
132 									// remove the markup
133 									if (rangeObject.isCollapsed()) {
134 										// when the range is collapsed, we remove exactly the one DOM element
135 										GENTICS.Utils.Dom.removeFromDOM(foundMarkup, rangeObject, true);
136 									} else {
137 										// the range is not collapsed, so we remove the markup from the range
138 										GENTICS.Utils.Dom.removeMarkup(rangeObject, markup, Aloha.activeEditable.obj);
139 									}
140 								} else {
141 									// when the range is collapsed, extend it to a word
142 									if (rangeObject.isCollapsed()) {
143 										GENTICS.Utils.Dom.extendToWord(rangeObject);
144 									}
145 
146 									// add the markup
147 									GENTICS.Utils.Dom.addMarkup(rangeObject, markup);
148 								}
149 								// select the modified range
150 								rangeObject.select();
151 								return false;
152 							},
153 							'tooltip' : i18n.t('button.' + button + '.tooltip'),
154 							'toggle' : true
155 						}), 'markup' : jQuery('<'+button+'></'+button+'>')};
156 
157 						FloatingMenu.addButton(
158 							scope,
159 							that.buttons[button].button,
160 							i18nCore.t('floatingmenu.tab.format'),
161 							1
162 						);
163 						break;
164 
165 					case 'p':
166 					case 'h1':
167 					case 'h2':
168 					case 'h3':
169 					case 'h4':
170 					case 'h5':
171 					case 'h6':
172 					case 'pre':
173 						that.multiSplitItems.push({
174 							'name' : button,
175 							'tooltip' : i18n.t('button.' + button + '.tooltip'),
176 							'iconClass' : 'aloha-button ' + i18n.t('aloha-button-' + button),
177 							'markup' : jQuery('<'+button+'></'+button+'>'),
178 							'click' : function() {
179 								Aloha.Selection.changeMarkupOnSelection(jQuery('<' + button + '></' + button + '>'));
180 							}
181 						});
182 						break;
183 
184 					// wide multisplit buttons
185 					case 'removeFormat':
186 						that.multiSplitItems.push({
187 							'name' : button,
188 							'text' : i18n.t('button.' + button + '.text'),
189 							'tooltip' : i18n.t('button.' + button + '.tooltip'),
190 							'iconClass' : 'aloha-button aloha-button-' + button,
191 							'wide' : true,
192 							'click' : function() {
193 								that.removeFormat();
194 							}
195 						});
196 						break;
197 					//no button defined
198 					default:
199 						Aloha.log('warn', this, 'Button "' + button + '" is not defined');
200 						break;
201 				}
202 			});
203 
204 			if (this.multiSplitItems.length > 0) {
205 				this.multiSplitButton = new Aloha.ui.MultiSplitButton({
206 					'name' : 'phrasing',
207 					'items' : this.multiSplitItems
208 				});
209 				FloatingMenu.addButton(
210 					scope,
211 					this.multiSplitButton,
212 					i18nCore.t('floatingmenu.tab.format'),
213 					3
214 				);
215 			}
216 
217 			// add the event handler for selection change
218 			Aloha.bind('aloha-selection-changed',function(event,rangeObject){
219 				// iterate over all buttons
220 				var
221 					statusWasSet = false, effectiveMarkup,
222 					foundMultiSplit, i, j, multiSplitItem;
223 
224 				jQuery.each(that.buttons, function(index, button) {
225 					statusWasSet = false;
226 					for ( i = 0; i < rangeObject.markupEffectiveAtStart.length; i++) {
227 						effectiveMarkup = rangeObject.markupEffectiveAtStart[ i ];
228 						if (Aloha.Selection.standardTextLevelSemanticsComparator(effectiveMarkup, button.markup)) {
229 							button.button.setPressed(true);
230 							statusWasSet = true;
231 						}
232 					}
233 					if (!statusWasSet) {
234 						button.button.setPressed(false);
235 					}
236 				});
237 
238 				if (that.multiSplitItems.length > 0) {
239 					foundMultiSplit = false;
240 
241 					// iterate over the markup elements
242 					for ( i = 0; i < rangeObject.markupEffectiveAtStart.length && !foundMultiSplit; i++) {
243 						effectiveMarkup = rangeObject.markupEffectiveAtStart[ i ];
244 
245 						for ( j = 0; j < that.multiSplitItems.length && !foundMultiSplit; j++) {
246 							multiSplitItem = that.multiSplitItems[j];
247 
248 							if (!multiSplitItem.markup) {
249 								continue;
250 							}
251 
252 							// now check whether one of the multiSplitItems fits to the effective markup
253 							if (Aloha.Selection.standardTextLevelSemanticsComparator(effectiveMarkup, multiSplitItem.markup)) {
254 								that.multiSplitButton.setActiveItem(multiSplitItem.name);
255 								foundMultiSplit = true;
256 							}
257 						}
258 					}
259 
260 					if (!foundMultiSplit) {
261 						that.multiSplitButton.setActiveItem(null);
262 					}
263 				}
264 			});
265 
266 		},
267 
268 		/**
269 		 * Removes all formatting from the current selection.
270 		 */
271 		removeFormat: function() {
272 273 			var formats = [ 'strong', 'em', 'b', 'i', 'cite', 'q', 'code', 'abbr', 'del', 'sub', 'sup'],
274 				rangeObject = Aloha.Selection.rangeObject,
275 				i;
276 			
277 			// formats to be removed by the removeFormat button may now be configured using Aloha.settings.plugins.format.removeFormats = ['b', 'strong', ...]
278 			if (this.settings.removeFormats) {
279 				formats = this.settings.removeFormats;
280 			}
281 
282 			if (rangeObject.isCollapsed()) {
283 				return;
284 			}
285 
286 			for (i = 0; i < formats.length; i++) {
287 				GENTICS.Utils.Dom.removeMarkup(rangeObject, jQuery('<' + formats[i] + '></' + formats[i] + '>'), Aloha.activeEditable.obj);
288 			}
289 
290 			// select the modified range
291 			rangeObject.select();
292 			// TODO: trigger event - removed Format
293 
294 		},
295 
296 		/**
297 		* toString method
298 		* @return string
299 		*/
300 		toString: function () {
301 			return 'format';
302 		}
303 	});
304 });
