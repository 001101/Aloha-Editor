  8  * Aloha List Enforcer
  9  * -------------------
 10  * Enforces a one top-level list per editable policy ;-)
 11  * This plugin will register editables and enforce lists in them. List enforced
 12  * editables will be permitted to contain, exactly one top-level element which
 13  * must be a (OL or a UL) list element.
 14  */
 15 
 16 define( [
 17 	'aloha',
 18 	'aloha/jquery',
 19 	'aloha/plugin',
 20 	'aloha/floatingmenu',
 21 	'aloha/console'
 22 ], function( Aloha, jQuery, Plugin, FloatingMenu, console ) {
 23 	'use strict';
 24 
 25 	/**
 26 	 * An internal array of all editables inwhich to enforce lists.
 27 	 *
 28 	 * @private
 29 	 */
 30 	var listEnforcedEditables = [];
 31 
 32 	/**
 33 	 * Given an editable which has been configured to enforce lists,
 34 	 * ensures that there is exactly one top-level list in the editable.
 35 	 * If there are no lists, one will be added, using the
 36 	 * placeHolderListString. If there is more than one list, they will be
 37 	 * merged into the first list.
 38 	 *
 39 	 * @private
 40 	 * @param {jQuery} $editable
 41 	 * @param {String} placeHolderListString
 42 	 */
 43 	function enforce ( $editable, placeHolderListString ) {
 44 		// Check if this editable is configured to enforce lists
 45 		if ( jQuery.inArray( $editable[ 0 ], listEnforcedEditables ) === -1 ) {
 46 			return;
 47 		}
 48 
 49 		// Remove all temporary <br>s in the editable, which we may have
 50 		// inserted when we activated this editable and found it empty. These
 51 		// <br>s are needed to make the otherwise emty <li> visible (in IE).
 52 		$editable.find( '.GENTICS_temporary' ).remove();
 53 
 54 		// Check for the presence of at least one non-empty list. We consider
 55 		// a list to be not empty if it has atleast one item whose contents are
 56 		// more than a single (propping) <br> tag.
 57 
 58 		var hasList = false;
 59 
 60 		$editable.find( 'li' ).each( function(){
 61 			// nb: jQuery text() method returns the text contents of the
 62 			// element without <br>s being rendered.
 63 			if ( jQuery.trim( jQuery( this ).text() ) !== '' ) {
 64 				hasList = true;
 65 				// Stop looking for lists as soon as we find our first
 66 				// non-empty list
 67 				return false;
 68 			}
 69 		} );
 70 
 71 		// If we found no non-empty list, then we add our empty dummy list that
 72 		// the user can work with.
 73 		if ( !hasList ) {
 74 			$editable.html( placeHolderListString );
 75 		}
 76 
 77 		// Concatinate all top-level lists into the first, before, thereby
 78 		// merging all top-level lists into one.
 79 		var $lists = $editable.find( '>ul,>ol' ),
 80 		    j = $lists.length,
 81 		    i;
 82 		if ( j > 1 ) {
 83 			var $firstList = jQuery( $lists[0] );
 84 			for ( i = 1; i < j; ++i ) {
 85 				$firstList.append( jQuery( $lists[ i ] ).find( '>li' ) );
 86 				jQuery( $lists[ i ] ).remove();
 87 			}
 88 		}
 89 
 90 		// Remove all top-level elements which are not lists
 91 		$editable.find( '>*:not(ul,ol)' ).remove();
 92 	};
 93 
 94 	return Plugin.create( 'listenforcer', {
 95 
 96 		languages: [ 'en', 'de' ],
 97 
 98 		_constructor: function() {
 99 			this._super( 'listenforcer' );
100 		},
101 
102 		/**
103 		 * Initializes the listenforcer plugin:
104 		 * We read the aloha configuration settings to determine which
105 		 * editables are to have list enforced in them.
106 		 * We bind handlers to 3 events (aloha-editable-activated,
107 		 * aloha-editable-deactivated, and aloha-smart-content-changed) on
108 		 * which we will process the current active editable and enfore lists
109 		 * in it.
110 		 */
111 		init: function() {
112 			var that = this,
113 			    elemsToEnforce = this.settings.editables || [],
114 				elemToEnforce,
115 				i,
116 				j = elemsToEnforce.length;
117 
118 			// Register all editables that are to enforce lists.
119 			// The following types of items can be used as jQuery selectors:
120 			// String, DOMElement, and jQuery
121 			for ( i = 0; i < j; i++ ) {
122 				elemToEnforce = elemsToEnforce[ i ];
123 				if ( typeof elemToEnforce === 'string' ||
124 						elemToEnforce.nodeName ||
125 							elemToEnforce instanceof jQuery ) {
126 					this.addEditableToEnforcementList( jQuery( elemToEnforce )[0] );
127 				} else {
128 					console.warn(
129 						'Aloha List Enforcer Plugin',
130 						'Object "' + elemToEnforce.toString() + '" can not ' +
131 						'be used as a jQuery selector with which to register' +
132 						' an editable to be list enforced.'
133 					);
134 				}
135 			}
136 
137 			Aloha.bind( 'aloha-editable-activated', function( $event, params ) {
138 				enforce( params.editable.obj,
139 					'<ul><li><br class="GENTICS_temporary" /></li></ul>' );
140 			} );
141 
142 			Aloha.bind( 'aloha-editable-deactivated', function( $event, params ) {
143 				enforce( params.editable.obj, '' );
144 			} );
145 
146 			Aloha.bind( 'aloha-smart-content-changed', function( $event, params ) {
147 				// window.console.log( 'Smart content changed event' );
148 				enforce( params.editable.obj,
149 					'<ul><li><br class="GENTICS_temporary" /></li></ul>' );
150 			} );
151 		},
152 
153 		/**
154 		 * Registers the given editable to be list-enforced.
155 		 *
156 		 * @param {DOMElement} editable
157 		 */
158 		addEditableToEnforcementList: function( editable ) {
159 			if ( editable ) {
160 				listEnforcedEditables.push( editable );
161 			}
162 		}
163 
164 	} );
165 } );
