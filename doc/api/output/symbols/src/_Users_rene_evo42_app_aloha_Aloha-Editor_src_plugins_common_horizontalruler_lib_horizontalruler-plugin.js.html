  7 define(
  8 ['aloha', 'aloha/jquery', 'aloha/plugin', 'aloha/floatingmenu', 'i18n!horizontalruler/nls/i18n', 'i18n!aloha/nls/i18n', 'css!horizontalruler/css/horizontalruler.css'],
  9 function(Aloha, jQuery, Plugin, FloatingMenu, i18n, i18nCore) {
 10 	"use strict";
 11 
 12 	var
 13 		GENTICS = window.GENTICS;
 14 
 15 	return Plugin.create('horizontalruler', {
 16 		_constructor: function(){
 17 			this._super('horizontalruler');
 18 		},
 19 		languages: ['en'],
 20 		init: function() {
 21 			var that = this;
 22 			var insertButton = new Aloha.ui.Button({
 23 				'name': 'hr',
 24 				'iconClass': 'aloha-button-horizontalruler',
 25 				'size': 'small',
 26 				'onclick': function(element, event) { that.insertHR(); },
 27 				'tooltip': i18n.t('button.addhr.tooltip'),
 28 				'toggle': false
 29 			});
 30 			FloatingMenu.addButton(
 31 				'Aloha.continuoustext',
 32 				insertButton,
 33 				i18nCore.t('floatingmenu.tab.insert'),
 34 				1
 35 			);
 36 			
 37 		},
 38 		insertHR: function(character) {
 39 			var self = this;
 40 			var range = Aloha.Selection.getRangeObject();
 41 			if(Aloha.activeEditable) {
 42 				var hr = jQuery('<hr>');
 43 				GENTICS.Utils.Dom.insertIntoDOM(
 44 					hr,
 45 					range,
 46 					jQuery(Aloha.activeEditable.obj),
 47 					true
 48 				);
 49 				range.select();
 50 			}
 51 		}
 52 	});
 53 
 54 });
