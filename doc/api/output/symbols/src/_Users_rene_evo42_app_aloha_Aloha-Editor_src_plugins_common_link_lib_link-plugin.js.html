249 			// add the event handler for selection change
250 			Aloha.bind( 'aloha-selection-changed', function ( event, rangeObject ) {
251 				var config,
252 				    foundMarkup;
253 				
254 				// Check if we need to ignore this selection changed event for
255 				// now and check whether the selection was placed within a
256 				// editable area.
257 				if ( !that.ignoreNextSelectionChangedEvent &&
258 						Aloha.Selection.isSelectionEditable() &&
259 							Aloha.activeEditable != null ) {
260 					// show/hide the button according to the configuration
261 					config = that.getEditableConfig( Aloha.activeEditable.obj );
262 					if ( jQuery.inArray( 'a', config ) != -1 ) {
263 						that.formatLinkButton.show();
264 						that.insertLinkButton.show();
265 					} else {
266 						that.formatLinkButton.hide();
267 						that.insertLinkButton.hide();
268 						// leave if a is not allowed
269 						return;
270 					}
271 					
272 					foundMarkup = that.findLinkMarkup( rangeObject );
273 					
274 					if ( foundMarkup ) {
275 						that.insertLinkButton.hide();
276 						that.formatLinkButton.setPressed( true );
277 						FloatingMenu.setScope( 'link' );
278 
279 						// remember the current tab selected by the user
280 						var currentTab = FloatingMenu.userActivatedTab;
281 
282 						// switch to the href tab (so that we make sure that the href field gets created)
283 						FloatingMenu.activateTabOfButton( 'href' );
284 						if ( currentTab ) {
285 							// switch back to the original tab
286 							FloatingMenu.userActivatedTab = currentTab;
287 						}
288 						// now we are ready to set the target object
289 						that.hrefField.setTargetObject( foundMarkup, 'href' );
290 						
291 						// if the selection-changed event was raised by the first click interaction on this page
292 						// the hrefField component might not be initialized. When the user switches to the link
293 						// tab to edit the link the field would be empty. We check for that situation and add a
294 						// special interval check to set the value once again
295 						if ( jQuery( '#' + that.hrefField.extButton.id ).length == 0 ) {
296 							// there must only be one update interval running at the same time
297 							if ( that.hrefUpdateInt !== null ) {
298 								clearInterval( that.hrefUpdateInt );
299 							}
300 							
301 							// register a timeout that will set the value as soon as the href field was initialized
302 							that.hrefUpdateInt = setInterval( function () {
303 								if ( jQuery( '#' + that.hrefField.extButton.id ).length > 0 ) { // the object was finally created
304 									that.hrefField.setTargetObject( foundMarkup, 'href' );
305 									clearInterval( that.hrefUpdateInt );
306 								}
307 							}, 200 );
308 						}
309 					} else {
310 						that.formatLinkButton.setPressed( false );
311 						that.hrefField.setTargetObject( null );
312 					}
313 				}
314 				
315 				that.ignoreNextSelectionChangedEvent = false;
316 			} );

432 		/**
433 		 * Parse a all editables for links and bind an onclick event
434 		 * Add the link short cut to all edtiables

436 		bindInteractions: function () {
437 			var that = this;
438 
439 			// update link object when src changes
440 			this.hrefField.addListener( 'keyup', function ( obj, event ) {
441 				// Now show all the ui-attributefield elements
442 				that.showComboList();
443 				
444 				// Handle ESC key press: We do a rough check to see if the user
445 				// has entered a link or searched for something
446 				if ( event.keyCode == 27 ) {
447 					var curval = that.hrefField.getQueryValue();
448 					if ( curval[ 0 ] == '/' || // local link
449 						 curval[ 0 ] == '#' || // inner document link
450 						 curval.match( /^.*\.([a-z]){2,4}$/i ) || // local file with extension
451 						 curval.match( /^htt.*/i ) // external link
452 					) {
453 						// could be a link better leave it as it is
454 					} else {
455 						// the user searched for something and aborted
456 						// restore original value
457 						that.hrefField.setValue( that.hrefField.getValue() );
458 459 						// or clean the field value
460 						// that.hrefField.setValue();
461 						that.hideComboList();
462 					}
463 				}
464 				
465 				that.hrefChange();
466 				
467 				// Handle the enter key. Terminate the link scope and show the final link.
468 				if ( event.keyCode == 13 ) {
469 					// Update the selection and place the cursor at the end of the link.
470 					var	range = Aloha.Selection.getRangeObject();
471 					
472 					// workaround to keep the found markup otherwise removelink won't work
473 //					var foundMarkup = that.findLinkMarkup( range );
474 //					console.dir(foundMarkup);
475 //					that.hrefField.setTargetObject(foundMarkup, 'href');
476 					
477 					// We have to ignore the next 2 onselectionchange events.
478 					// The first one we need to ignore is the one trigger when
479 					// we reposition the selection to right at the end of the
480 					// link.
481 					// Not sure what the next event is yet but we need to
482 					// ignore it as well, ignoring it prevents the value of
483 					// hrefField from being set to the old value.
484 					that.ignoreNextSelectionChangedEvent = true;
485 					range.startContainer = range.endContainer;
486 					range.startOffset = range.endOffset;
487 					range.select();
488 					that.ignoreNextSelectionChangedEvent = true;
489 					
490 					var hrefValue = jQuery( that.hrefField.extButton.el.dom ).attr( 'value' );
491 					
492 					if ( hrefValue == 'http://' || hrefValue == '' ) {
493 						that.removeLink( false );
494 					}
495 					
496 					setTimeout( function () {
497 						FloatingMenu.setScope( 'Aloha.continuoustext' );
498 					}, 100 );
499 					
500 					that.preventAutoSuggestionBoxFromExpanding();
501 				} else {
502 					// Check whether the value in the input field has changed
503 					// because if it has, then the ui-attribute object's store
504 					// needs to be cleared. The reason we need to do this
505 					// clearing is because once the auto-suggeset combo box is
506 					// shown and/or populated, the next enter keypress event
507 					// would be handled as if the user is selecting one of the
508 					// elements in the down down list.
509 					newValue = jQuery( that.hrefField.extButton.el.dom ).attr( 'value' );
510 					if ( oldValue != newValue ) {
511 						oldValue = newValue;
512 						// Drop local cache of suggestion items
513 						
514 						// Don't use this method because it will update the
515 						// loading message to say that no items were found ...
516 						// that.hrefField.extButton.store.removeAll();
517 						
518 						// ... instead we will manually delete the store data
519 						// ourselves
520 						var storeData = that.hrefField.extButton.store.data;
521 						storeData.items = [];
522 						storeData.key = [];
523 						storeData.length = 0;
524 						that.hrefField.extButton.store.lastQuery = null;
525 					}
526 				}
527 			} );
528

530 				.keydown( function ( e ) {
531 					Aloha.Log.debug( that, 'Meta key down.' );
532 					that.metaKey = e.metaKey;
533 					that.updateMousePointer();

544 		updateMousePointer: function () {

547 				jQuery( this.mouseOverLink ).removeClass( 'aloha-link-text' );

555 		/**
556 		 * Check whether inside a link tag
557 		 * @param {GENTICS.Utils.RangeObject} range range where to insert the
558 		 *			object (at start or end)
559 		 * @return markup
560 		 * @hide
561 		 */
562 		findLinkMarkup: function ( range ) {
563 			if ( typeof range == 'undefined' ) {
564 				range = Aloha.Selection.getRangeObject();
565 			}
566 			if ( Aloha.activeEditable ) {
567 				return range.findMarkup( function () {
568 					return this.nodeName.toLowerCase() == 'a';
569 				}, Aloha.activeEditable.obj );
570 			} else {
571 				return null;
572 			}

590 		 * Insert a new link at the current selection. When the selection is
591 		 * collapsed, the link will have a default link text, otherwise the
592 		 * selected text will be the link text.
593 		 */
594 		insertLink: function ( extendToWord ) {
595 			var that = this,
596 			    range = Aloha.Selection.getRangeObject(),
597 			    linkText,
598 			    newLink;
599 			
600 			// There are occasions where we do not get a valid range, in such
601 			// cases we should not try and add a link
602 			if ( !( range.startContainer && range.endContainer ) ) {
603 				return;
604 			}
605 			
606 			// do not nest a link inside a link
607 			if ( this.findLinkMarkup( range ) ) {
608 				return;
609 			}
610 			
611 			// activate floating menu tab
612 			FloatingMenu.activateTabOfButton( 'href' );
613 			
614 			// if selection is collapsed then extend to the word.
615 			if ( range.isCollapsed() && extendToWord !== false ) {
616 				GENTICS.Utils.Dom.extendToWord( range );
617 			}
618 			if ( range.isCollapsed() ) {
619 				// insert a link with text here
620 				linkText = i18n.t( 'newlink.defaulttext' );
621 				newLink = jQuery( '<a href="" class="aloha-new-link">' + linkText + '</a>' );
622 				GENTICS.Utils.Dom.insertIntoDOM( newLink, range, jQuery( Aloha.activeEditable.obj ) );
623 				range.startContainer = range.endContainer = newLink.contents().get( 0 );
624 				range.startOffset = 0;
625 				range.endOffset = linkText.length;
626 			} else {
627 				newLink = jQuery( '<a href="" class="aloha-new-link"></a>' );
628 				GENTICS.Utils.Dom.addMarkup( range, newLink, false );
629 			}
630 
631 			Aloha.activeEditable.obj.find( 'a.aloha-new-link' ).each( function ( i ) {
632 				that.addLinkEventHandlers( this );
633 				jQuery(this).removeClass( 'aloha-new-link' );
634 			} );
635 
636 			range.select();
637 
638 			// focus has to become before prefilling the attribute, otherwise
639 			// Chrome and Firefox will not focus the element correctly.
640 			this.hrefField.focus();
641 			// prefill and select the new href
642 			// We need this guard because there are time when the extButton's
643 			// el element has not yet available
644 			if ( this.hrefField.extButton.el ) {
645 				jQuery( this.hrefField.extButton.el.dom ).attr( 'value', 'http://' ).select();
646 			}
647 			this.hrefChange();
648 		},

650 		/**
651 		 * Remove an a tag and clear the current item from the hrefField
652 		 */
653 		removeLink: function ( terminateLinkScope ) {
654 			var	range = Aloha.Selection.getRangeObject(),
655 			    foundMarkup = this.findLinkMarkup();
656 			
657 			// clear the current item from the href field
658 			this.hrefField.setItem(null);
659 			if ( foundMarkup ) {
660 				// remove the link
661 				GENTICS.Utils.Dom.removeFromDOM( foundMarkup, range, true );
662 
663 				range.startContainer = range.endContainer;
664 				range.startOffset = range.endOffset;
665 
666 				// select the (possibly modified) range
667 				range.select();
668 				
669 				if ( typeof terminateLinkScope == 'undefined' ||
670 						terminateLinkScope === true ) {
671 					FloatingMenu.setScope( 'Aloha.continuoustext' );
672 				}
673 			}
674 		},

676 		/**
677 		 * Updates the link object depending on the src field
678 		 */
679 		hrefChange: function () {
680 			var that = this;
681 			
682 			// For now hard coded attribute handling with regex.
683 			// Avoid creating the target attribute, if it's unnecessary, so
684 			// that XSS scanners (AntiSamy) don't complain.
685 			if ( this.target != '' ) {
686 				this.hrefField.setAttribute(
687 					'target',
688 					this.target,
689 					this.targetregex,
690 					this.hrefField.getQueryValue()
691 				);
692 			}
693 			
694 			this.hrefField.setAttribute(
695 				'class',
696 				this.cssclass,
697 				this.cssclassregex,
698 				this.hrefField.getQueryValue()
699 			);
700 			
701 			Aloha.trigger( 'aloha-link-href-change', {
702 				 obj: that.hrefField.getTargetObject(),
703 				 href: that.hrefField.getQueryValue(),
704 				 item: that.hrefField.getItem()
705 			} );
706 			
707 			if ( typeof this.onHrefChange == 'function' ) {
708 				this.onHrefChange.call(
709 					this,
710 					this.hrefField.getTargetObject(),
711 					this.hrefField.getQueryValue(),
712 					this.hrefField.getItem()
713 				);
714 			}
715 		},
