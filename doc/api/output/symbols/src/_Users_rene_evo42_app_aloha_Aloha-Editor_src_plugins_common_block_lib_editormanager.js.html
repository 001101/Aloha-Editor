  8 define(
  9 ['aloha/registry'],
 10 function(Registry) {
 11 	"use strict";
 12 
 13 	/**
 14 	 * @name block.editormanager
 15 	 * @class Editor manager singleton
 16 	 */
 17 	return new (Registry.extend(
 18 	/** @lends block.editormanager */
 19 	{
 20 		/**
 21 		 * Create an editor from the given definition. Acts as a factory method
 22 		 * for editors.
 23 		 *
 24 		 * @param {Object} definition
 25 		 */
 26 		createEditor: function(definition) {
 27 			if (!this.has(definition.type)) {
 28 				throw 'Editor for type "' + definition.type + '" not found.';
 29 			}
 30 			var Editor = this.get(definition.type);
 31 			return new Editor(definition);
 32 		}
 33 	}))();
 34 });
