246 		/**
247 		 * Transform the current selection to/from a list
248 		 * @param ordered true when transforming to/from an ordered list, false for unordered lists
249 		 */
250 		transformList: function (ordered) {
251 			var domToTransform = this.getStartingDomObjectToTransform(),
252 				lastLi, i, jqNewLi, jqList, selectedSiblings, jqParentList,
253 				newPara, jqToTransform, nodeName;
254 
255 			// visible is set to true, but the button is not visible
256 			this.outdentListButton.show();
257 			this.indentListButton.show();
258 
259 			if (!domToTransform) {
260 				// wrap a paragraph around the selection
261 				Aloha.Selection.changeMarkupOnSelection(jQuery('<p></p>'));
262 				domToTransform = this.getStartingDomObjectToTransform();
263 
264 				if (!domToTransform) {
265 					if ( Aloha.Selection.rangeObject.startContainer.contentEditable ) {
266 						// create a new list with an empty item
267 						jqList = ordered ? jQuery('<ol></ol>') : jQuery('<ul></ul>');
268 						jqNewLi = jQuery('<li></li>');
269 						jqList.append(jqNewLi);
270 						
271 						var li = jqNewLi.get(0);
272 						var editable = Aloha.getActiveEditable().obj;
273 						//IE7 requires an (empty or non-empty) text node
274 						//inside the li for the selection to work.
275 						li.appendChild(document.createTextNode(""));
276 						/* if ( jQuery.browser.mozilla ) {
277 							li.appendChild(document.createElement("BR"));
278 						}*/
279 
280 						editable.append(jqList);
281 						editable.focus();
282 
283 						var range = Aloha.createRange();
284 						var selection = Aloha.getSelection();
285 						range.setStart( li.firstChild, 0 );
286 						range.setEnd( li.firstChild, 0 );
287 						selection.removeAllRanges();
288 						selection.addRange( range );
289 						Aloha.Selection.updateSelection();
290 						
291 						return;
292 					} else {
293 					Aloha.Log.error(this, 'Could not transform selection into a list');
294 					return;
295 				}
296 			}
297 			}
298 
299 			// check the dom object
300 			nodeName = domToTransform.nodeName.toLowerCase();
301 
302 			if (nodeName == 'ul' && !ordered) {
303 				// first check whether the list is nested into another list
304 				jqList = jQuery(domToTransform);
305 
306 				jqParentList = jqList.parent();
307 				if (jqParentList.length > 0
308 						&& GENTICS.Utils.Dom.isListElement(jqParentList.get(0))) {
309 					// when the list is nested into another, our list items will be
310 					// added to the list items of the outer list
311 
312 					// find the place where to put the children of the inner list
313 					if (jqParentList.get(0).nodeName.toLowerCase() === 'li') {
314 						// inner table is nested in a li (this conforms to the html5 spec)
315 						jqParentList.after(jqList.children());
316 						jqList.remove();
317 					} else {
318 						// inner table is nested in the outer list directly (this violates the html5 spec)
319 						jqList.children().unwrap();
320 					}
321 				} else {
322 					// we are in an unordered list and shall transform it to paragraphs
323 
324 					// transform all li into p
325 					jqToTransform = jQuery(domToTransform);
326 					jQuery.each(jqToTransform.children('li'), function(index, li) {
327 						newPara = Aloha.Markup.transformDomObject(li, 'p', Aloha.Selection.rangeObject);
328 						// if any lists are in the paragraph, move the to after the paragraph
329 						newPara.after(newPara.children('ol,ul'));
330 					});
331 
332 					// unwrap the li (remove the enclosing ul)
333 					jqToTransform.children().unwrap();
334 				}
335 			} else if (nodeName == 'ul' && ordered) {
336 				// we are in an unordered list and shall transform it to an ordered list
337 
338 				// transform the ul into an ol
339 				Aloha.Markup.transformDomObject(domToTransform, 'ol', Aloha.Selection.rangeObject);
340 
341 				// merge adjacent lists
342 				this.mergeAdjacentLists(jQuery(domToTransform));
343 			} else if (nodeName == 'ol' && !ordered) {
344 				// we are in an ordered list and shall transform it to an unordered list
345 
346 				// transform the ol into an ul
347 				Aloha.Markup.transformDomObject(domToTransform, 'ul', Aloha.Selection.rangeObject);
348 
349 				// merge adjacent lists
350 				this.mergeAdjacentLists(jQuery(domToTransform));
351 			} else if (nodeName == 'ol' && ordered) {
352 				// first check whether the list is nested into another list
353 				jqList = jQuery(domToTransform);
354 
355 				jqParentList = jqList.parent();
356 				if (jqParentList.length > 0
357 						&& GENTICS.Utils.Dom.isListElement(jqParentList.get(0))) {
358 					// when the list is nested into another, our list items will be
359 					// added to the list items of the outer list
360 
361 					// find the place where to put the children of the inner list
362 					if (jqParentList.get(0).nodeName.toLowerCase() === 'li') {
363 						// inner table is nested in a li (this conforms to the html5 spec)
364 						jqParentList.after(jqList.children());
365 						jqList.remove();
366 					} else {
367 						// inner table is nested in the outer list directly (this violates the html5 spec)
368 						jqList.children().unwrap();
369 					}
370 				} else {
371 					// we are in an unordered list and shall transform it to paragraphs
372 
373 					// transform all li into p
374 					jqToTransform = jQuery(domToTransform);
375 					jQuery.each(jqToTransform.children('li'), function(index, li) {
376 						newPara = Aloha.Markup.transformDomObject(li, 'p', Aloha.Selection.rangeObject);
377 						// if any lists are in the paragraph, move the to after the paragraph
378 						newPara.after(newPara.children('ol,ul'));
379 					});
380 
381 					// unwrap the li (remove the enclosing ul)
382 					jqToTransform.children().unwrap();
383 				}
384 			} else {
385 				// we are in something different from a list and shall transform it into a list
386 
387 				// get the also selected siblings of the dom object
388 				selectedSiblings = Aloha.Selection.rangeObject.getSelectedSiblings(domToTransform);
389 
390 				// create a new list
391 				jqList = ordered ? jQuery('<ol></ol>') : jQuery('<ul></ul>');
392 				// add a new list item
393 				jqNewLi = jQuery('<li></li>');
394 				// add the li into the list
395 				jqList.append(jqNewLi);
396 				// append the contents of the old dom element to the li
397 				jQuery(domToTransform).contents().appendTo(jqNewLi);
398 				// replace the old dom element with the new list
399 				jQuery(domToTransform).replaceWith(jqList);
400 
401 				// update the selection range
402 				if (Aloha.Selection.rangeObject.startContainer == domToTransform) {
403 					Aloha.Selection.rangeObject.startContainer = jqNewLi.get(0);
404 				}
405 				if (Aloha.Selection.rangeObject.endContainer == domToTransform) {
406 					Aloha.Selection.rangeObject.endContainer = jqNewLi.get(0);
407 				}
408 
409 				var lastAppendedLi = jqNewLi;
410 
411 				// now also transform all siblings
412 				if (selectedSiblings) {
413 					lastLi = false;
414 					for ( i = 0; i < selectedSiblings.length; ++i) {
415 						if (GENTICS.Utils.Dom.isBlockLevelElement(selectedSiblings[i])) {
416 							if (lastLi) {
417 								lastLi = false;
418 							}
419 
420 							// transform the block level element
421 							jqNewLi = Aloha.Markup.transformDomObject(selectedSiblings[i], 'li', Aloha.Selection.rangeObject);
422 							jqList.append(jqNewLi);
423 							lastAppendedLi = jqNewLi;
424 						} else {
425 							if (selectedSiblings[i].nodeType == 3
426 									&& jQuery.trim(selectedSiblings[i].data).length === 0) {
427 								continue;
428 							}
429 							if (!lastLi) {
430 								lastLi = jQuery('<li></li>');
431 								jqList.append(lastLi);
432 								lastAppendedLi = lastLi;
433 							}
434 							lastLi.append(selectedSiblings[i]);
435 						}
436 					}
437 				}
438 
439 				// merge adjacent lists
440 				this.mergeAdjacentLists(jqList);
441 
442 				//use rangy to change the selection to the contents of
443 				//the last li that was appended to the list
444 				var li = lastAppendedLi.get(0);
445 				if (GENTICS.Utils.Dom.isEmpty(li)) {
446 					var range = Aloha.createRange();
447 					var selection = Aloha.getSelection();
448 					//IE7 requires an (empty or non-empty) text node
449 					//inside the li for the selection to work.
450 					li.appendChild(document.createTextNode(""));
451 					range.selectNodeContents( li.lastChild );
452 					selection.removeAllRanges();
453 					selection.addRange( range );
454 					Aloha.Selection.updateSelection();
455 				}
456 			}
457 
458 			// refresh the selection
459 			this.refreshSelection();
460 		},

462 		/**
463 		 * Indent the selected list items by moving them into a new created, nested list
464 		 */
465 		indentList: function () {
466 			var listItem = this.getNearestSelectedListItem(),
467 				i, jqNewList, selectedSiblings, jqOldList, jqItemBefore;
468 
469 			if (listItem) {
470 				jqItemBefore = jQuery(listItem).prev('li');
471 
472 				// when we are in the first li of a list, there is no indenting
473 				if (jqItemBefore.length === 0) {
474 					// but we handled the TAB keystroke
475 					return false;
476 				}
477 				jqOldList = jQuery(listItem).parent();
478 
479 				// get the also selected siblings of the dom object
480 				selectedSiblings = Aloha.Selection.rangeObject.getSelectedSiblings(listItem);
481 
482 				// create the new list element by cloning the selected list element's parent
483 				jqNewList = jQuery(listItem).parent().clone(false).empty();
484 				jqNewList.append(listItem);
485 
486 				// we found a list item before the first selected one, so append the new list to it
487 				jqItemBefore.append(jqNewList);
488 
489 				// check for multiple selected items
490 				if (selectedSiblings) {
491 					for ( i = 0; i < selectedSiblings.length; ++i) {
492 						jqNewList.append(jQuery(selectedSiblings[i]));
493 					}
494 				}
495 
496 				// merge adjacent lists
497 				this.mergeAdjacentLists(jqNewList, true);
498 
499 				// refresh the selection
500 				this.refreshSelection();
501 
502 				return false;
503 			}
504 
505 			return true;
506 		},

508 		/**
509 		 * Outdent nested list items by moving them into the outer list
510 		 */
511 		outdentList: function () {
512 			var
513 				listItem = this.getNearestSelectedListItem(),
514 				i, jqNewPostList,
515 				jqListItem, jqList, jqParentList, wrappingLi,
516 				selectedSiblings, lastSelected;
517 
518 			if (listItem) {
519 				// check whether the list is nested into another list
520 				jqListItem = jQuery(listItem);
521 				jqList = jqListItem.parent();
522 
523 				// get the parent list
524 				jqParentList = jqList.parents('ul,ol');
525 
526 				// check whether the inner list is directly inserted into a li element
527 				wrappingLi = jqList.parent('li');
528 
529 				if (jqParentList.length > 0
530 						&& GENTICS.Utils.Dom.isListElement(jqParentList.get(0))) {
531 					// the list is nested into another list
532 
533 					// get the also selected siblings of the dom object
534 					selectedSiblings = Aloha.Selection.rangeObject.getSelectedSiblings(listItem);
535 
536 					// check for multiple selected items
537 					if (selectedSiblings && selectedSiblings.length > 0) {
538 						lastSelected = jQuery(selectedSiblings[selectedSiblings.length - 1]);
539 					} else {
540 						lastSelected = jqListItem;
541 					}
542 
543 					// check whether we found not selected li's after the selection
544 					if (lastSelected.nextAll('li').length > 0) {
545 						jqNewPostList = jqList.clone(false).empty();
546 						jqNewPostList.append(lastSelected.nextAll());
547 						lastSelected.append(jqNewPostList);
548 					}
549 
550 					// now move all selected li's into the higher list
551 					if (wrappingLi.length > 0) {
552 						wrappingLi.after(jqListItem);
553 					} else {
554 						jqList.before(jqListItem);
555 					}
556 
557 					// check for multiple selected items
558 					if (selectedSiblings && selectedSiblings.length > 0) {
559 						for ( i = selectedSiblings.length - 1; i >= 0; --i) {
560 							jqListItem.after(jQuery(selectedSiblings[i]));
561 						}
562 					}
563 
564 					// finally check whether there are elements left in the list
565 					if (jqList.contents('li').length === 0) {
566 						// list is completely empty, so remove it
567 						jqList.remove();
568 					}
569 
570 					// check whether the wrapping li is empty now
571 					if (wrappingLi.length > 0 && wrappingLi.contents().length === 0) {
572 						wrappingLi.remove();
573 					}
574 
575 					// refresh the selection
576 					this.refreshSelection();
577 				}
578 
579 				return false;
580 			}
581 
582 			return true;
583 		},

594 		/**
595 		 * Merge adjacent lists (of same type) into the first list
596 		 * @param jqList jQuery object of a list
597 		 * @param allTypes true if all types of lists may be merged, false if only same types may be merged
598 		 */
599 		mergeAdjacentLists: function (jqList, allTypes) {
600 			// first get the first previous sibling of same type
601 			var firstList = jqList.get(0), jqNextList;
602 
603 			while (
604 				firstList.previousSibling
605 				&& firstList.previousSibling.nodeType === 1
606 				&& this.isMergable(firstList.previousSibling, firstList, allTypes)
607 			) {
608 				firstList = firstList.previousSibling;
609 			}
610 
611 			jqList = jQuery(firstList);
612 			// now merge all adjacent lists into this one
613 			while (

	/**
746 	 * A key handler that should be run as a keyup handler for the
747 	 * backspace and del keys. keyup fires after the browser has already
748 	 * performed the delete - this handler will perform a cleanup if
749 	 * necessary.
750 	 *
751 	 * Will work around an IE bug which breaks nested lists in the
752 	 * following situation, where [] is the selection, if backspace is
753 	 * pressed (same goes for the del key if the selection is at the end
754 	 * of the li that occurs before the selection):
755 	 *
756 	 * <ul>
757 	 *  <li>one</li>
758 	 *  <li><ul><li>two</li></ul></li>
759 	 * </ul>
760 	 * <p>[]</p>
761 	 *
762 	 * The browser behaviour, if one would presses backspace, results in
763 	 * the following:
764 	 *
765 	 * <ul>
766 	 *  <li>one</li>
767 	 *  <ul><li>two</li></ul>
768 	 * </ul>
769 	 *
770 	 * which is invalid HTML since the <ul>s are nested directly inside
771 	 * each other.
772 	 *
773 	 * Also, the following situation will cause the kind of invalid HTML
774 	 * as above.
775 	 * <ul>
776 	 *   <li>one</li>
777 	 *   <li><ul><li>two</li></ul></li>
778 	 *   <li>[]three</li>
779 	 * </ul>
780 	 *
781 	 * Also, the following situtation:
782 	 * <ul>
783 	 *   <li>one</li>
784 	 *   <li><ul><li>two</li></ul>
785 	 *       <p>[]three</p>
786 	 *       <li>four</li>
787 	 *   </li>
788 	 * </ul>
789 	 *
790 	 * And similar situations, some of which are not so easy to reproduce.
791 	 * 
792 	 * @param event a jQuery key event
793 	 * @return false if no action needed to be taken, true if cleanup has been performed
794 	 */
795 	function deleteWorkaroundHandler(event) {
796 		if (8/*backspace*/ != event.keyCode && 46/*del*/ != event.keyCode) {
797 			return false;
798 		}
799 
800 		var rangeObj = Aloha.getSelection().getRangeAt(0);
801 		var startContainer = rangeObj.startContainer;
802 
803 		//the hack is only relevant if after the deletion has been
804 		//performed we are inside a li of a nested list
805 		var $nestedList = jQuery(startContainer).closest('ul, ol');
806 		if ( ! $nestedList.length ) {
807 			return false;
808 		}
809 		var $parentList = $nestedList.parent().closest('ul, ol');
810 		if ( ! $parentList.length ) {
811 			return false;
812 		}
813 
814 		var ranges = Aloha.getSelection().getAllRanges();
815 
816 		var actionPerformed = false;
817 		$parentList.each(function(){
818 			actionPerformed = actionPerformed || fixListNesting(jQuery(this));
819 		});
820 
821 		if (actionPerformed) {
822 			Aloha.getSelection().setRanges(ranges);
823 			for (var i = 0; i < ranges.length; i++) {
824 				ranges[i].detach();
825 			}
826 		}
827 
828 		return actionPerformed;
829 	}

832 	 * If uls or ols are nested directly inside the given list (invalid
833 	 * HTML), they will be cleaned up by being appended to the preceding
834 	 * li.
835 	 */
836 	function fixListNesting($list) {
837 		var actionPerformed = false;
838 		$list.children('ul, ol').each(function(){
839 			Aloha.Log.debug("performing list-nesting cleanup");
840 			if ( ! jQuery(this).prev('li').append(this).length ) {
841 				//if there is no preceding li, create a new one and append to that
842 				jQuery(this).parent().prepend(document.createElement('li')).append(this);
843 			}
844 			actionPerformed = true;
845 		});
846 		return actionPerformed;
847 	}
