  9 define(
 10 ['aloha/jquery', 'aloha/plugin', 'aloha/floatingmenu', 'i18n!aloha/nls/i18n'],
 11 	function(jQuery, Plugin, FloatingMenu, i18nCore) {
 12 	"use strict";
 13 	var
 14 		$ = jQuery,
 15 		GENTICS = window.GENTICS,
 16 		Aloha = window.Aloha;
 17 
 18 	/**
 19 	 * register the plugin with unique name
 20 	 */
 21 	var ribbon = Plugin.create('ribbon', {
 22 
 23 		
 24 		/**
 25 		 * Initialize the plugin and set initialize flag on true
 26 		 */
 27 		init: function () {
 28 
 29 			// Check if the ribbon is enabled
 30 			if (typeof this.settings.enable == 'undefined' || this.settings.enable === true) {
 31 				var that = this;
 32 				
 33 				// flag to mark whether ribbon is visible
 34 				this.visible = false;
 35 
 36 				// the ribbon
 37 				this.toolbar = new Ext.Toolbar({
 38 					height: '25px',
 39 					cls: 'ext-root',
 40 					id: 'aloha-ribbon'
 41 				});
 42 
 43 				// left spacer to gain some space from the left screen border
 44 				this.toolbar.add(new Ext.Toolbar.Spacer({width: '5'}));
 45 				// icon
 46 				this.icon = new Ext.Toolbar.Spacer();
 47 				this.toolbar.add(this.icon);
 48 				// fill so Aloha.Ribbon everything after it is aligned right
 49 				this.toolbar.add(new Ext.Toolbar.Fill());
 50 				// seperator before the fade out button
 51 				this.toolbar.add(new Ext.Toolbar.Separator());
 52 				// fade out button
 53 				var fadeButton = new Ext.Button({
 54 					iconCls : 'aloha-fade-out',
 55 					handler : function (button) {
 56 						var toolbar = jQuery(that.toolbar.getEl().dom);
 57 
 58 						if (button.iconCls == 'aloha-fade-out') {
 59 							toolbar.animate({
 60 								left: '-100%',
 61 								marginLeft: '34px'
 62 							});
 63 							jQuery('body').animate({
 64 								paddingTop: 0
 65 							});
 66 							button.setIconClass('aloha-fade-in');
 67 						} else {
 68 							toolbar.animate({
 69 								left: 0,
 70 								marginLeft: 0
 71 							});
 72 							jQuery('body').animate({
 73 								paddingTop: '30px'
 74 							});
 75 							button.setIconClass('aloha-fade-out');
 76 						}
 77 						that.toolbar.doLayout();
 78 					}
 79 				});
 80 				this.toolbar.add(fadeButton);
 81 				// spacer to gain some space from the right screen border
 82 				this.toolbar.add(new Ext.Toolbar.Spacer({width: '5'}));
 83 
 84 				this.toolbar.render(document.body, 0);
 85 
 86 				jQuery('body').css('paddingTop', '30px');
 87 			}
 88 		},
 89 		
 90 		/**
 91 		 * Sets the icon class for the ribbon icon
 92 		 * @param {String} iconClass CSS class for the icon
 93 		 */
 94 		setIcon: function (iconClass) {
 95 			if (typeof this.icon.cls !== 'undefined') {
 96 				this.icon.removeClass(this.icon.cls);
 97 			}
 98 
 99 			this.icon.addClass(iconClass);
100 		},
101 
102 		/**
103 		 * Adds a Aloha.ui.Button the Ribbon
104 		 * @param {Button} button Button to be added to the Ribbon
105 		 */
106 		addButton: function (button) {
107 			if (button.menu != null && typeof button.menu === 'object') {
108 				// build the drop down menu
109 				var menu = new Ext.menu.Menu();
110 				jQuery.each(button.menu, function(index, entry) {
111 					menu.addItem(new Ext.menu.Item({
112 						text: entry.label,
113 						icon: entry.icon,
114 						iconCls: entry.iconClass,
115 						handler: function() {
116 							entry.onclick.apply(entry);
117 						}
118 					}));
119 				});
120 			}
121 
122 			// configuration for the button
123 			var buttonConfig = {
124 				text : button.label,
125 				enableToggle: button.toggle,
126 				icon: button.icon,
127 				pressed : button.pressed,
128 				iconCls: button.iconClass,
129 				menu : menu,
130 				handler : function() {
131 					if (typeof button.onclick === 'function') {
132 						button.onclick.apply(button);
133 					}
134 					button.pressed = !button.pressed;
135 				}
136 			}
137 
138 			var extButton;
139 
140 			// Build a split button if we have a menu and a handler
141 			if (menu && typeof button.onclick == 'function') {
142 				// build the split button for the menu
143 				extButton = new Ext.SplitButton(buttonConfig);
144 			} else {
145 				// build a normal button
146 				extButton = new Ext.Button(buttonConfig);
147 			}
148 
149 			this.toolbar.insert(this.toolbar.items.getCount() - 3, extButton);
150 			this.toolbar.doLayout();
151 		},
152 
153 		/**
154 		 * Adds a seperator to the Ribbon.
155 		 */
156 		addSeparator: function() {
157 			this.toolbar.insert(this.toolbar.items.getCount() - 3, new Ext.Toolbar.Separator());
158 		},
159 
160 		/**
161 		 * Shows the ribbAloha.Ribbonon
162 		 */
163 		hide: function () {
164 			jQuery('#aloha-ribbon').fadeOut();
165 			this.visible = false;
166 		},
167 
168 		/**
169 		 * Hides the ribbon
170 		 */
171 		show: function () {
172 			jQuery('#aloha-ribbon').fadeIn();
173 			this.visible = true;
174 		},
175 
176 		/**
177 		 * Check whether the ribbon is visible right now
178 		 * @return true when the ribbon is visible, false when not
179 		 */
180 		isVisible: function () {
181 			return this.visible;
182 		}
183 	});
184 
185 	return ribbon;
186 });
