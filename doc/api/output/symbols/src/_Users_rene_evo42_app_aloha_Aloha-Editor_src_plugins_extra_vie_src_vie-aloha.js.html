1 (function($){
  2     $.fn.vieSemanticAloha = function(options) {
  3         
  4         // Default settings
  5         var opt = { 
  6                 beforeEditing: null
  7         };
  8         $.extend(opt, options);
  9 
 10 
 11         this.each(function() {
 12             var containerInstance = VIE.ContainerManager.getInstanceForContainer(jQuery(this));
 13             if (typeof containerInstance.editables === 'undefined') {
 14                 containerInstance.editables = {};
 15             }
 16             VIE.ContainerManager.findContainerProperties(this, false).each(function() {
 17                 var containerProperty = jQuery(this);
 18 
 19                 // Call the configured beforeEditing function that may modify 
 20                 // the content of the editable before editing is possible
 21                 if(opt.beforeEditing != null) {
 22                     opt.beforeEditing(containerProperty);
 23                 }
 24 
 25 
 26                 var propertyName = containerProperty.attr('property');
 27 
 28                 if (containerInstance.get(propertyName) instanceof Array) {
 29                     // For now we don't deal with multivalued properties in Aloha
 30                     return true;
 31                 }
 32 
 33                 containerInstance.editables[propertyName] = new GENTICS.Aloha.Editable(containerProperty);
 34                 containerInstance.editables[propertyName].vieContainerInstance = containerInstance;
 35             });
 36         })
 37     }
 38 })(jQuery);

 40 if (typeof VIE === 'undefined') {
 41     VIE = {};

 44 VIE.AlohaEditable = {
 45     refreshFromEditables: function(objectInstance) {
 46        var modifiedProperties = {};
 47 
 48         // Go through editables of the model instance
 49         jQuery.each(objectInstance.editables, function(propertyName, editableInstance) {
 50             if (!editableInstance.isModified()) {
 51                 // This editable hasn't been modified, skip
 52                 return true;
 53             }
 54 
 55             // Refresh possible RDFa objects from inside the editable
 56             jQuery('[typeof][about]', editableInstance.obj).each(function() {
 57                 var childInstance = VIE.ContainerManager.getInstanceForContainer(jQuery(this));
 58             });
 59 
 60             // Copy editable contents to the modifiedProperties object
 61             modifiedProperties[propertyName] = editableInstance.getContents();
 62         });
 63 
 64         if (jQuery.isEmptyObject(modifiedProperties))
 65         {
 66             // No modified editables for this object, skip
 67             return false;
 68         }
 69 
 70         // Set the modified properties to the model instance
 71         objectInstance.set(modifiedProperties);
 72         return true;
 73     }
 74 };
