  7 define(
  8 ['aloha/jquery','aloha/plugin', 'aloha/floatingmenu', 'i18n!numerated-headers/nls/i18n', 'i18n!aloha/nls/i18n','css!numerated-headers/css/numerated-headers.css',],
  9 function(jQuery, Plugin, FloatingMenu, i18n, i18nCore) {
 10 	"use strict";
 11 
 12    var
 13 	$ = jQuery,
 14 	GENTICS = window.GENTICS,
 15 	Aloha = window.Aloha;
 16 	
 17    return Plugin.create('numerated-headers', {
 18      
 19 	 numeratedactive : true,
 20 	 headingselector : 'h1, h2, h3, h4, h5, h6',
 21 	 
 22      /**
 23       * Initialize the plugin
 24       */
 25      init: function () {
 26        var that = this;
 27 
 28 	   if ( typeof this.settings.numeratedactive !== 'undefined' ) {
 29 			this.numeratedactive = this.settings.numeratedactive;
 30 		}
 31 
 32 	   // modifyable selector for the headers, that should be numerated
 33 	   if ( typeof this.settings.headingselector !== 'undefined' ) {
 34 			this.headingselector = this.settings.headingselector;
 35 		}
 36 	   // modifyable selector for the baseobject. Where should be numerated
 37 	   if ( typeof this.settings.baseobjectSelector !== 'undefined' ) {
 38 			this.baseobjectSelector = this.settings.baseobjectSelector;
 39 		}
 40 		
 41        	    // add button to toggle format-less pasting
 42           this.numeratedHeadersButton = new Aloha.ui.Button({
 43             'iconClass' : 'aloha-button aloha-button-numerated-headers',
 44             'size' : 'small',
 45             'onclick' : function () {
 46 		   if(that.numeratedHeadersButton.isPressed()) {
 47 			//that.numeratedHeadersButton.setPressed(false);
 48 			that.removeNumerations();
 49 		   }
 50 		   else {
 51 			//that.numeratedHeadersButton.setPressed(true);
 52 			that.createNumeratedHeaders();
 53 		   }
 54             },
 55             'tooltip' : i18n.t('button.numeratedHeaders.tooltip'),
 56             'toggle' : true,
 57 	    'pressed' : this.numeratedactive
 58           });
 59 
 60           FloatingMenu.addButton(
 61             'Aloha.continuoustext',
 62             this.numeratedHeadersButton,
 63             i18nCore.t('floatingmenu.tab.format'),
 64             1
 65           );
 66 	   
 67 	  // We need to bind to selection-changed event to recognize backspace and delete interactions
 68 	  Aloha.bind( 'aloha-selection-changed', function ( event ) {
 69 	    if ( that.numeratedHeadersButton.isPressed() ) {
 70 		that.createNumeratedHeaders();
 71 	    }
 72 	  });
 73      },
 74 
 75 	 removeNumerations : function () {
 76 		var active_editable_obj = this.getBaseElement();
 77 
 78 		if ( !active_editable_obj ) {
 79 			return;
 80 		}
 81 
 82 		var headers = active_editable_obj.find(this.headingselector);
 83 		headers.each( function() {
 84 			jQuery(this).find('span[role=annotation]').each(function(){jQuery(this).remove();});
 85 		});
 86 	 },
 87 
 88 	getBaseElement: function() {
 89 
 90 		if ( typeof this.baseobjectSelector !== 'undefined' ) {
 91 			if( jQuery(this.baseobjectSelector).length > 0 ) {
 92 				return jQuery(this.baseobjectSelector);
 93 			} else {
 94 				return false;
 95 			}
 96 		} else {
 97 			if ( typeof Aloha.activeEditable == "undefined" || Aloha.activeEditable == null ) {
 98         			return false;
 99         		} else {
100 				return Aloha.activeEditable.obj;
101 			}
102 		}
103 	},
104 
105 
106 	/*
107 	* checks if the given Object contains a note Tag that looks like this:
108 	* <span annotation=''>
109 	*
110 	* @param {Object} obj - The Object to check
111 	*/
112      	hasNote: function(obj) {
113 	
114 		if(!obj || !jQuery(obj).length > 0){	
115 			return false;
116 		}
117 
118 		obj = jQuery(obj);
119 
120 		if( obj.find('span[role=annotation]').length > 0 ) {
121 			return true;
122 		}
123 
124 		return false;
125 	},
126 
127 	
128 	/*
129 	* checks if the given Object has textual content.
130 	* A possible "<span annotation=''>" tag will be ignored
131 	*
132 	* @param {Object} obj - The Object to check
133 	*/
134 	hasContent: function(obj) {
135 	
136 		if( !obj || !jQuery(obj).length > 0 ) {	
137 			return false;
138 		}
139 
140 		obj = jQuery(obj);
141 
142 		// we have to check the content of this object without the annotation span
143 		var objCleaned = obj.clone().find('span[role=annotation]').remove().end();
144 
145 		// check for text, also in other possible sub tags
146 		if ( objCleaned.text().trim().length > 0 ) {
147 			return true;
148 		}
149 	
150 		return false;
151 	},
152 
153 
154      createNumeratedHeaders: function() {
155 
156 	var active_editable_obj = this.getBaseElement();
157 
158 	if(!active_editable_obj){
159 		return;
160 	}
161 	
162 	var that = this;
163       
164         var headers = active_editable_obj.find(this.headingselector);
165 
166 		if ( typeof headers == "undefined" || headers.length == 0 ) {
167            return;
168       }
169 
170       var base_rank = parseInt(headers[0].nodeName.substr(1));
171       var prev_rank = null;
172       var current_annotation = [];
173       var annotation_pos = 0;
174 
175       // initialize the base annotations
176       for ( var i=0; i < (6 - base_rank) + 1; i++ ) {
177           current_annotation[i] = 0; 
178       }
179       
180       headers.each(function(){
181 	// build and count annotation only if there is content in this header
182 	if( that.hasContent(this) ) {
183 
184        		var current_rank = parseInt(this.nodeName.substr(1));
185         	if( prev_rank == null ){
186           		//increment the main annotation 
187           		current_annotation[annotation_pos]++;
188         	}
189         	//starts a sub title
190         	else if ( current_rank > prev_rank ) {
191           		current_annotation[++annotation_pos]++; 
192         	}
193         	//continues subtitles
194         	else if ( current_rank == prev_rank ) {
195           		current_annotation[annotation_pos]++; 
196         	}
197         	//goes back to a main title
198         	else if ( current_rank < prev_rank ) {
199           		var current_pos = current_rank - base_rank;
200           		for( var j=annotation_pos; j > (current_pos); j-- ){
201             			current_annotation[j] = 0; //reset current sub-annotation
202           		}
203           		annotation_pos = current_pos;
204           		current_annotation[annotation_pos]++; 
205         	}
206 
207         	prev_rank = current_rank;
208 
209         	var annotation_result = current_annotation[0];
210         	for( var i = 1; i < current_annotation.length; i++ ){
211           		if(current_annotation[i] != 0){
212              			annotation_result += ("." + current_annotation[i]); 
213           		} 
214         	}
215 	
216 		if ( that.hasNote(this) ) {
217 			jQuery(this).find('span[role=annotation]').html(annotation_result); 
218 		} else {
219 			jQuery(this).prepend("<span role='annotation'>" + annotation_result + "</span> ");
220 		}
221 	} else {
222 		// no Content, so remove the Note, if there is one
223 		if ( that.hasNote(this) ) {
224 			jQuery(this).find('span[role=annotation]').remove();
225 		}
226 		
227 	}
228 
229       })
230      }
231      
232    });
233 });
