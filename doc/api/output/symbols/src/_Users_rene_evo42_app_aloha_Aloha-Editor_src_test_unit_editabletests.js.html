 10 	/**
 11 	 * Do an enter test
 12 	 * @param editable the editable
 13 	 * @param container cursor container
 14 	 * @param offset offset of the cursor position
 15 	 * @param shift true for shift enter, false for normal enter
 16 	 * @param twice true for pressing enter twice, false for once
 17 	 * @param reference result selector
 18 	 */
 19 	function doEnterTest(editable, container, offset, shift, twice, reference) {
 20 		// set cursor
 21 		TestUtils.setCursor(editable, container, offset);
 22 		// press enter
 23 		TestUtils.pressEnter(editable, shift);
 24 		// possibly again
 25 		if (twice) {
 26 			TestUtils.pressEnter(editable, shift);
 27 		}
 28 		// remove all additional end-breaks
 29 		Aloha.editables[0].obj.find('br.aloha-end-br').remove();
 30 		// get the result	
 31 		var result = Aloha.editables[0].getContents(true);
 32 		var expected = Aloha.jQuery(reference).contents();
 33 	
 34 		// compare the result with the expected result
 35 		deepEqual(result.extractHTML(), expected.extractHTML(), 'Check Operation Result');
 36 	}
